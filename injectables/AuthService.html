import { Injectable } from "@angular/core";
import { HttpClient } from "@angular/common/http";
import { environment } from "src/environments/environment";
import { Client, ResetModel } from "../@model/client";
import { Observable, BehaviorSubject } from "rxjs";
import { AuthResponse } from "../@model/responses/auth-api-response";
import { GeneralAPIResponse } from "../@model/responses/general-api-response";

@Injectable({ providedIn: "root" })
export class AuthService {
    private _presentStateSource = new BehaviorSubject<any>(false);
    public state$ = this._presentStateSource.asObservable() as Observable<any>;

    constructor(private httpClient: HttpClient) { }

    /**
     * Logins in client to the application
     * @param client the client's data
     */
    login(client: Client): Observable<AuthResponse> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}login`,
            client
        ) as Observable<AuthResponse>;
    }

    /**
     * Registers a new client to the platform this clients is one that we do not have his/her BVN details
     * @param client the client's data
     */
    registersReturningUsers(
        client: Client
    ): Observable<GeneralAPIResponse<Client>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}register`,
            client
        ) as Observable<GeneralAPIResponse<Client>>;
    }

    /**
     * Generate Client OTP
     * @param client client's email and the service the otp was generated for
     */
    generateClientOTP(client: Client): Observable<GeneralAPIResponse<string>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}generateClientOTP`,
            client
        ) as Observable<GeneralAPIResponse<string>>;
    }

    /**
     * Registers a client to the platform - this clients is one that we have his/her BVN details already
     * @param client the client's data
     */
    registerNewClient(client: Client): Observable<GeneralAPIResponse<Client>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}ntb_register`,
            client
        ) as Observable<GeneralAPIResponse<Client>>;
    }

    /**
     * Initializes a reset password more like a forgot password endpoint
     * @param resetCred the needed reset credential
     */
    initializeResetPassword(
        resetCred: ResetModel
    ): Observable<GeneralAPIResponse<string>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}passwordResetInitiate`,
            resetCred
        ) as Observable<GeneralAPIResponse<string>>;
    }

    /**
     * resets the client's password
     * @param resetCred the needed reset credential
     */
    resetPassword(
        resetCred: ResetModel
    ): Observable<GeneralAPIResponse<string>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}passwordResetComplete`,
            resetCred
        ) as Observable<GeneralAPIResponse<string>>;
    }

    /**
     * Initializes a unlock account process
     * @param client
     */
    initializeUnlock(client: Client): Observable<GeneralAPIResponse<string>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}unblockClientInitiate`,
            client
        ) as Observable<GeneralAPIResponse<string>>;
    }

    /**
     * Completes the unlock password process
     * @param client
     */
    completeUnlock(client: Client): Observable<GeneralAPIResponse<string>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}unblockClientComplete`,
            client
        ) as Observable<GeneralAPIResponse<string>>;
    }

    /**
     * Resends otp to a user
     * @param resetCred  the needed reset credential
     */
    resendToken(resetCred: ResetModel) {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}generateClientOTP`,
            resetCred
        );
    }

    /**
     * Completes users registration process
     * @param resetCred  the needed reset credential
     */
    completeRegistration(
        resetCred: ResetModel
    ): Observable<GeneralAPIResponse<Client>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}completeRegistration`,
            resetCred
        ) as Observable<GeneralAPIResponse<Client>>;
    }

    /**
     * Gets the Users token from the browser
     */
    getUserToken(): string {
        return localStorage.getItem("renmu_t");
    }

    /**
     * Sets the User token receive from the API after a successful login
     * @param token token gotten from API
     */
    setUserToken(token: string): void {
        localStorage.setItem("renmu_t", token);
    }

    /**
     * Logs the user out of the application
     */
    logOut() {
        localStorage.removeItem('renmu_t');
        localStorage.removeItem('ren_c');
        localStorage.removeItem('acctp_c');
        localStorage.removeItem('tm');
        localStorage.removeItem('entry');
    }

    /**
     * This is responsible for the state change in the auth component
     * @param loadingState Boolean that determines if something prime is being loaded
     */
    changeState(loadingState: boolean) {
        this._presentStateSource.next(loadingState);
    }

    checkIfClientsExist(payload) {
        return this.httpClient.post(`${environment.apiBaseURLSudo}clientExist`, payload);
    }

}
