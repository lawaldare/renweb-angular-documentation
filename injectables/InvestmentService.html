import { BankData } from "./../@page/single-view/banks";
import { Injectable } from "@angular/core";
import { HttpClient, HttpHeaders } from "@angular/common/http";
import { Investment } from "../@model/investment";
import { environment } from "src/environments/environment";
import { GeneralAPIResponse } from "../@model/responses/general-api-response";
import { Observable } from "rxjs";
import { getInvestmentsSuccessfullResponse } from "../@page/invest/invest/investment.mockdata";
import { heLocale } from "ngx-bootstrap/chronos/i18n/he";
import { InvestmentInterface } from "../@model/interfaces";

@Injectable({ providedIn: "root" })
export class InvestmentService {
    investments = getInvestmentsSuccessfullResponse.data.clientDeposits;

    constructor(private httpClient: HttpClient) { }

    createInvestment(inv: Investment): Observable<GeneralAPIResponse<string>> {
        return this.httpClient.post(
            `${environment.investmentBaseURL}client/createInvestment`,
            inv
        ) as Observable<GeneralAPIResponse<string>>;
    }

    createTargetSavings(
        investment: InvestmentInterface
    ): Observable<GeneralAPIResponse<string>> {
        return this.httpClient.post(
            `${environment.investmentBaseURL}client/createTargetSavings`,
            investment
        ) as Observable<GeneralAPIResponse<string>>;
    }

    createFixedDeposit(
        investment: InvestmentInterface
    ): Observable<GeneralAPIResponse<string>> {
        return this.httpClient.post(
            `${environment.investmentBaseURL}client/createFixedDeposit`,
            investment
        ) as Observable<GeneralAPIResponse<string>>;
    }

    getBanks() {
        return BankData;
    }

    getInvestments(): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.get(
            `${environment.investmentBaseURL}client/investments`
        ) as Observable<GeneralAPIResponse<any>>;
    }

    getSingleInvestment(id: string): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.get(
            `${environment.investmentBaseURL}client/investments/${id}`
        ) as Observable<GeneralAPIResponse<any>>;
    }

    getTransactionsHistory(id: string): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.get(
            `${environment.investmentBaseURL}client/investments/${id}/transactions`
        ) as Observable<GeneralAPIResponse<any>>;
    }

    getInvestmentHistory(id: string): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.get(
            `${environment.investmentBaseURL}client/investments/${id}/history`
        ) as Observable<GeneralAPIResponse<any>>;
    }

    /**
     * Save details gotten from calculator
     * @param calcDet if fixed-deposit - { amount: any; tenure: any; amountToBeEarned: any }
     */
    saveCalcValues(calcDet: any) {
        localStorage.setItem("_ren_calc_det", JSON.stringify(calcDet));
    }
    /**
     * @returns { amount: any; tenure: any; amountToBeEarned: any } if fixed-deposit
     */
    getCalcValues(): any {
        return JSON.parse(localStorage.getItem("_ren_calc_det")) as {
            amount: any;
            tenure: any;
            interest: any;
            amountToBeEarned: any;
        };
    }

    getCalType() {
        return localStorage.getItem("_ren_calc_type");
    }

    setCalcType(calcType: string) {
        localStorage.setItem("_ren_calc_type", calcType);
    }

    deleteCalcValues() {
        localStorage.removeItem("_ren_calc_det");
    }

    getAccountName(bankCode: string, accountNumber: string) {
        const httpOptions = {
            headers: new HttpHeaders({
                "Content-Type": "application/json",
                sourceAppID: environment.mbs2sourceAppID,
            }),
        };
        return this.httpClient.get(
            `${environment.rencoreBaseURL}account-number-validation/${accountNumber}/${bankCode}`
        );
    }

    editPlanName(planName: any): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.investmentBaseURL}editPlanName`,
            planName
        ) as Observable<GeneralAPIResponse<any>>;
    }

    deletePlan(investmentId): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.delete(
            `${environment.investmentBaseURL}deletePlan/${investmentId}`
        ) as Observable<GeneralAPIResponse<any>>;
    }

    sendStatement(investmentId): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.get(
            `${environment.investmentBaseURL}sendStatement/${investmentId}`
        ) as Observable<GeneralAPIResponse<any>>;
    }

    getRenflexAccount(): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.get(
            `${environment.investmentBaseURL}renflexAccount`
        ) as Observable<GeneralAPIResponse<any>>;
    }

    withdrawRenflexAccount(data): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.investmentBaseURL}withdraw/renflex`,
            Object.assign(data, { transactionRef: "Web" })
        ) as Observable<GeneralAPIResponse<any>>;
    }

    getIntrabankTransfer(details): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.investmentBaseURL}intrabankTransfer`,
            details
        ) as Observable<GeneralAPIResponse<any>>;
    }

    createRegularSavings(): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.investmentBaseURL}client/createRegularSavings`,
            {}
        ) as Observable<GeneralAPIResponse<any>>;
    }

    intrabankTransfer(data): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}transfer/intrabank`,
            Object.assign(data, { transactionRef: "Web" })
        ) as Observable<GeneralAPIResponse<any>>;
    }

    maturedWithdrawal(details: any): Observable<GeneralAPIResponse<any>> {
        // let headers = new HttpHeaders();
        // headers = headers.append("RENMONEY-API-KEY", `${environment.renmoneyAPIKey}`);

        return this.httpClient.post(
            `${environment.investmentBaseURL}withdraw/maturity`,
            Object.assign(details, { transactionRef: "Web" })
        ) as Observable<GeneralAPIResponse<any>>;
    }

    preMaturedWithdrawal(details: any): Observable<GeneralAPIResponse<any>> {
        // let headers = new HttpHeaders();
        // headers = headers.append("RENMONEY-API-KEY", `${environment.renmoneyAPIKey}`);

        return this.httpClient.post(
            `${environment.investmentBaseURL}withdraw/pre-maturity/fd`,
            Object.assign(details, { transactionRef: "Web" })
        ) as Observable<GeneralAPIResponse<any>>;
    }

    maturedTargetSavingsWithdrawal(
        details: any
    ): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.investmentBaseURL}withdraw/targetSavings`,
            details
        ) as Observable<GeneralAPIResponse<any>>;
    }

    setupRecurrentPayment(investment): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}payment/recurrent`,
            investment
        ) as Observable<GeneralAPIResponse<any>>;
    }
    saveSmartGoalInvestmentLocally(investment) {
        localStorage.setItem('sg_inv', JSON.stringify(investment));
    }

    getSmartGoalInvestmentLocally() {
        return JSON.parse(localStorage.getItem('sg_inv'));
    }
}
