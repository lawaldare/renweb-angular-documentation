import { Injectable } from "@angular/core";
import { GeneralAuth } from "../@model/client";
import { BehaviorSubject, Observable } from "rxjs";
import { ToastrService } from "ngx-toastr";

@Injectable({
    providedIn: "root",
})
export class GeneralService {
    constructor(private toastr: ToastrService) {}

    private _presentStateSource = new BehaviorSubject<any>(false);
    public state$ = this._presentStateSource.asObservable() as Observable<any>;

    /**
     * This is responsible for the state change in the auth component
     * @param loadingState Boolean that determines if something prime is being loaded
     */
    showLoader(loadingState: boolean) {
        this._presentStateSource.next(loadingState);
    }

    /**
     * Gets device details (usually attached to an already exsiting object)
     */
    getDeviceDetails(): GeneralAuth {
        let vendor = window.navigator.vendor;
        const agent = window.navigator.userAgent.split(" ");
        return {
            osVersion: window.navigator.platform,
            deviceType:
                vendor === ""
                    ? agent[agent.length - 1]
                    : agent[agent.length - 2],
            deviceId: agent.join(" "),
            location: `{0,0}`,
        };
    }

    /**
     * Copy text to clipboard
     * @param value value to copy
     */
    copyText(value: string) {
        let input = document.createElement("input");
        input.style.position = "fixed";
        input.style.left = "0";
        input.style.top = "0";
        input.style.opacity = "0";
        input.value = value;
        document.body.appendChild(input);
        input.focus();
        input.select();
        document.execCommand("copy");
        document.body.removeChild(input);
        this.toastr.success("Copied to clipboard!");
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57)) {
            return false;
        }
        return true;
    }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `₦${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "").replace("₦", "");
    }
}
