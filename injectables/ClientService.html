import { Injectable } from "@angular/core";
import { HttpClient } from "@angular/common/http";
import { environment } from "src/environments/environment";
import { Observable, BehaviorSubject } from "rxjs";
import { Client } from "../@model/client";
import { GeneralAPIResponse } from "../@model/responses/general-api-response";

@Injectable({ providedIn: "root" })
export class ClientService {


    private bvnValidated = new BehaviorSubject<boolean>(false);
    bvnValidated$ = this.bvnValidated.asObservable();

    private nokDone = new BehaviorSubject<boolean>(false);
    nokDone$ = this.nokDone.asObservable();

    private haveLiquidationBank = new BehaviorSubject<boolean>(false);
    haveLiquidationBank$ = this.haveLiquidationBank.asObservable();

    changeValidationState(value: boolean): void {
        this.bvnValidated.next(value);
    }

    changeNOKState(value: boolean): void {
        this.nokDone.next(value);
    }

    changeLiquidationBankDetailsState(value: boolean): void {
        this.haveLiquidationBank.next(value);
    }

    constructor(private httpClient: HttpClient) { }

    /**
     * Gets the client's details
     */
    getClient(): Observable<GeneralAPIResponse<Client>> {
        return this.httpClient.get(
            `${environment.apiBaseURLSudo}clientDetails`
        ) as Observable<GeneralAPIResponse<Client>>;
    }

    /**
     * Get Mambu specific client Details
     */
    getClientFromMambu(): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.get(
            `${environment.apiBaseURLSudo}client/mambuClientDetails`
        ) as Observable<GeneralAPIResponse<any>>;
    }

    /**
     * get client details from middleware
     */
    getClientByBVN(bvn): Observable<GeneralAPIResponse<null>> {
        return this.httpClient.post(`${environment.apiBaseURLSudo}clientDetailByBVN`, { bvn: bvn }) as Observable<GeneralAPIResponse<null>>;
    }

    /**
     * Create a client on mambu
     */
    createClient(client: Client): Observable<GeneralAPIResponse<null>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}client/createMambuClient`,
            client
        ) as Observable<GeneralAPIResponse<null>>;
    }

    /**
     * Update a client on mambu
     */
    updateClient(client: Client): Observable<GeneralAPIResponse<null>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}client/updateMambuClient`,
            client
        ) as Observable<GeneralAPIResponse<null>>;
    }

    /**
     * Saves the Logged in client to the browser
     * @param client the logged in client
     */
    saveClientLocally(client: Client) {
        localStorage.setItem("ren_c", JSON.stringify(client));
    }

    saveClientLocallyForAccountSetup(client: Client) {
        localStorage.setItem("acctp_c", JSON.stringify(client));
    }



    /**
     * Gets the logged in client
     * @returns type of Client
     */
    getClientLocally(): Client {
        return JSON.parse(localStorage.getItem("ren_c")) as Client;
    }


    getClientLocallyForAccountSetup(): Client {
        return JSON.parse(localStorage.getItem("acctp_c")) as Client;
    }

    removeClientLocallyForAccountSetup(): void {
        localStorage.removeItem('acctp_c');
    }

    /**
     * gets the saved client encodedkey
     */
    getClientEncodedkey(): { clientEncodedkey: string } {
        return { clientEncodedkey: this.getClientLocally().clientEncodedkey };
    }



    changePassword(passwords: any): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}client/changePassword`,
            passwords
        ) as Observable<GeneralAPIResponse<any>>;
    }

    validatePassword(password: any): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}client/validatePassword`,
            { password }
        ) as Observable<GeneralAPIResponse<any>>;
    }

    changeSecurityQuestion(data: any): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.apiBaseURLSudo}client/changeSecurityQuestion`,
            data
        ) as Observable<GeneralAPIResponse<any>>;
    }

    uploadProfilePicture(data: any): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.post(
            `${environment.investmentBaseURL}upload`,
            data
        ) as Observable<GeneralAPIResponse<any>>;
    }

    getProfilePicture(): Observable<GeneralAPIResponse<any>> {
        return this.httpClient.get(
            `${environment.investmentBaseURL}profilePicture`
        ) as Observable<GeneralAPIResponse<any>>;
    }

    /**
     * Gets or Generates the Clients NUBAN on mambu
     * @param encodedkey the client encoded key of the user to generate or retieve the nuban for
     */
    getClientNuban(encodedkey) {
        return this.httpClient.post(
            `${environment.investmentBaseURL}generateNUBAN`, { encodedkey, "nubanType": "Savings" }
        ) as Observable<GeneralAPIResponse<any>>;
    }


}
