import { Injectable } from "@angular/core";
import { HttpClient, HttpHeaders } from "@angular/common/http";
import { environment } from "src/environments/environment";
import { Observable } from "rxjs";
import { Client } from "../@model/client";
import { GeneralAPIResponse } from "../@model/responses/general-api-response";

@Injectable({ providedIn: "root" })
export class DashboardService {


    constructor(private httpClient: HttpClient) { }

    /**
     * Gets the client's details
     */
    getDataFromBVN(bvnObject: {
        bvn: string;
    }): Observable<
        GeneralAPIResponse<{
            firstName: string;
            lastName: string;
            phoneNumber: string;
            gender: string;
            residentialAddress: string;
            DOB: string;
            middleName: string;
        }>
    > {
        const httpOptions = {
            headers: new HttpHeaders({
                "Content-Type": "application/json",
                sourceAppID: environment.bvnSourceAppID
            })
        };
        return this.httpClient.post(
            `${environment.bvnMS}validate`,
            bvnObject,
            httpOptions
        ) as Observable<
            GeneralAPIResponse<{
                firstName: string;
                lastName: string;
                phoneNumber: string;
                gender: string;
                residentialAddress: string;
                DOB: string;
                middleName: string;
            }>
        >;
    }

    /**
     * Populates the PEP responses on Mambu
     * @param pepDetails pep details in an object
     */
    fillPEPdetails(pepDetails: Object): Observable<GeneralAPIResponse<null>> {
        return this.httpClient.post(
            `${environment.investmentBaseURL}pep`,
            pepDetails
        ) as Observable<GeneralAPIResponse<null>>;
    }

    /**
     * Verifies if s BVN is matches according to the set standard.
     * @param client the client object
     * @param bvnResponse response from the BVN
     */
    isBVNvalid(client: Client, bvnResponse): boolean {


        let fullName = ` ${client.firstName.trim()} ${client.lastName.trim()} `;



        const PASSVALUE = 0.3333;

        let presentValue = 0;

        if (fullName.toLowerCase().includes(` ${bvnResponse.firstName.trim().toLowerCase()} `)) presentValue += 1 / 3;

        if (fullName.toLowerCase().includes(` ${bvnResponse.lastName.trim().toLowerCase()} `)) presentValue += 1 / 3;

        return presentValue >= PASSVALUE

    }

    /**
     * Save Config details locally
     * @param data config details
     */
    saveConfig(data: any) {
        localStorage.setItem('r_conf', JSON.stringify(data));
    }

    /**
     * Get config details that have been saved locally
     */
    getConfigLocally() {
        return JSON.parse(localStorage.getItem('r_conf'));
    }

    getConfigDetails() {
        return this.httpClient.get(`${environment.configURL}`) as Observable<any>;
    }
}
