import { Injectable } from '@angular/core';
@Injectable({
    providedIn: 'root'
})
export class PasswordService {
    constructor() { }



    atLeastEightCharacterMsg: boolean = true;
    atLeastOneNumberMsg: boolean = true;
    atLeastOneUpperCaseMsg: boolean = true;
    atLeastOneLowerCaseMsg: boolean = true;
    atLeastOneSymbolMsg: boolean = true;
    // atLeastOneSpaceMsg: boolean = false;



    atLeastEightCharacter(value) {
        return /[A-Za-z\d\W_]{8,}/.test(value);
    }
    atLeastOneNumber(value) {
        return /.*\d/.test(value);
    }
    atLeastOneUpperCase(value) {
        return /.*[A-Z]/.test(value);
    }
    atLeastOneLowerCase(value) {
        return /.*[a-z]/.test(value);
    }
    atLeastOneSymbol(value) {
        return /.*[\W_]/.test(value);
    }

    // atLeastOneSpace(value) {
    //     return /!.*\S/.test(value);
    // }



    checkPassword(password) {
        this.atLeastOneNumberMsg = !this.atLeastOneNumber(password.value);
        this.atLeastOneSymbolMsg = !this.atLeastOneSymbol(password.value);
        this.atLeastOneUpperCaseMsg = !this.atLeastOneUpperCase(password.value);
        this.atLeastOneLowerCaseMsg = !this.atLeastOneLowerCase(password.value);
        this.atLeastEightCharacterMsg = !this.atLeastEightCharacter(password.value);
        // this.atLeastOneSpaceMsg = !this.atLeastOneSpace(password.value);
    }
}
