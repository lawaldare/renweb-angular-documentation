import { Injectable } from "@angular/core";
import { HttpClient, HttpHeaders } from "@angular/common/http";
import { environment } from "src/environments/environment";
import { GeneralAPIResponse } from "../@model/responses/general-api-response";
import { BehaviorSubject, Observable } from "rxjs";


@Injectable({ providedIn: "root" })
export class LoanService {

    RANDOM_NUMBER = '1234345332';

    private stepNumber = new BehaviorSubject<number>(0);
    stepNumber$ = this.stepNumber.asObservable();

    changeStepNumber(value: number): void {
        this.stepNumber.next(value);
    }


    constructor(private httpClient: HttpClient) { }

    /**
     * Get the User's Loan
     */
    getLoans(clientId): Observable<GeneralAPIResponse<string>> {
        return this.httpClient.get(
            `${environment.investmentBaseURL}loans/${clientId}`
        ) as Observable<GeneralAPIResponse<string>>;
    }

    /**
     * Create a Loan application on Mambu
     */
    createLoan(loanDetail) {
        const httpOptions = {
            headers: new HttpHeaders({
                "sourceapp": environment.mbs2sourceAppID
            })
        };
        return this.httpClient.post(
            `${environment.mambuMiddlewareBaseURL}fraud-check`, loanDetail, httpOptions
        ) as Observable<GeneralAPIResponse<string>>;
    }

    /**
     * Get Repayments for a particular loan
     * @param loanId loanId to get repayments for
     */
    getLoanRepayment(loanId): Observable<GeneralAPIResponse<string>> {
        return this.httpClient.get(
            `${environment.investmentBaseURL}loans/${loanId}/repayments`
        ) as Observable<GeneralAPIResponse<string>>;
    }

    /**
     * Save client data gathered during loan creation
     * @param client present Loan CLient
     */
    saveLoanClient(client) {
        localStorage.setItem('ren_lc', JSON.stringify(client));
    }

    saveLoanEncodedKey(key) {
        localStorage.setItem('LEK', key);
    }

    getLoanEncodedKey() {
        return localStorage.getItem('LEK');
    }

    removeLoanEncodedKey() {
        localStorage.removeItem('LEK');
    }

    saveLoanID(key) {
        localStorage.setItem('l_id', key);
    }

    getLoanID() {
        return localStorage.getItem('l_id');
    }

    removeLoanID() {
        localStorage.removeItem('l_id');
    }

    setEntryFromOnboarding(mode) {
        localStorage.setItem('entry', mode);
    }

    getEntryFromOnboarding() {
        return localStorage.getItem('entry');
    }

    removeEntryFromOnboarding() {
        localStorage.removeItem('entry');
    }

    /**
    * Save present loan detail
    * @param loan presently filled loan
    */
    savePresentLoanDetail(loan) {
        localStorage.setItem('ren_pl', JSON.stringify(loan));
    }

    /**
   * Save present loan detail
   * @param loan presently filled loan
   */
    saveAccountDetailsLocally(details) {
        localStorage.setItem('act_de', JSON.stringify(details));
    }

    /**
       * Save present loan detail
       * @param loan presently filled loan
       */
    getAccountDetailsLocally() {
        return JSON.parse(localStorage.getItem('act_de'));
    }

    /**
    * Gets an ongoing loan detail
    */
    getFormerLoanDetails() {
        return {
            loan: JSON.parse(localStorage.getItem('ren_pl')),
            client: JSON.parse(localStorage.getItem('ren_lc'))
        }
    }

    /**
     * Save client data gathered during loan creation from loan onboard Page
     * @param client present Loan CLient
     */
    saveLoanClientFromLoanPage(client) {
        localStorage.setItem('ren_lcd', JSON.stringify(client));
    }

    /**
     * Save present loan detail from loan onboard Page
     * @param loan presently filled loan
     */
    saveLoanDetailFromLoanPage(loan) {
        localStorage.setItem('ren_pld', JSON.stringify(loan));
    }

    saveTeamLists(team) {
        localStorage.setItem('tm', JSON.stringify(team));
    }

    getTeamLists() {
        return JSON.parse(localStorage.getItem('tm'));
    }

    deleteTeamLists() {
        localStorage.removeItem('tm');
    }

    saveSocialMediaData(data) {
        localStorage.setItem('smd', JSON.stringify(data));
    }

    getSocialMediaData() {
        return JSON.parse(localStorage.getItem('smd'));
    }

    deleteSocialMediaData() {
        localStorage.removeItem('smd');
    }


    emailStatusUpdate(data) {
        return this.httpClient.post(`${environment.emailBaseURL}alerts/api/v1/email/status/update`, data);
    }

    sendEmailService(data) {
        return this.httpClient.post(`${environment.emailBaseURL}alerts/api/v1/email/send`, data);
    }


    generateOfferLetter(loanID) {
        const httpOptions = {
            headers: new HttpHeaders({
                "X-Requested-With": "XMLHttpRequest",
                "RENMONEY-API-KEY": environment.renmoneyAPIKey
            })
        };
        return this.httpClient.get(`${environment.offerLetter}${loanID}/${this.RANDOM_NUMBER}/json`, httpOptions);
    }

    /**
    * Gets an ongoing loan details from
    */
    getDetailsFromLandingPage() {
        return {
            loan: JSON.parse(localStorage.getItem('ren_pld')),
            client: JSON.parse(localStorage.getItem('ren_lcd'))
        }
    }

    sendLoanDocsStatus(loanData) {
        return this.httpClient.post(`${environment.investmentBaseURL}upload/status`, loanData);
    }

    getLoanDocsStatus(loanID) {
        return this.httpClient.get(`${environment.investmentBaseURL}upload/status/${loanID}`);
    }


    /**
     * Clears the former loan detail
     */
    clearFormerLoanDetail() {
        localStorage.removeItem('ren_pl');
        localStorage.removeItem('ren_lc');
    }

    /**
     * Clears the former loan detail
     */
    clearLoanPageDetails() {
        localStorage.removeItem('ren_pld');
        localStorage.removeItem('ren_lcd');
    }

    /**
     * Get a list on banks suported by bank streaming
     */
    getMBSBanks() {
        const httpOptions = {
            headers: new HttpHeaders({
                //"Content-Type": "application/json",
                sourceAppID: environment.mbs2sourceAppID,
                Authorization: `Basic ${btoa(`${environment.gdsUsername}:${environment.gdsPassword}`)}`
            })
        };
        return this.httpClient.get(`${environment.mbsBaseURL}renmbs/mbs/api/all/v1`, httpOptions);
    }

    /**
     * Gets a list of banks supported by bank streaming (Zenno0)
     */
    getBSBanks() {
        const httpOptions = {
            headers: new HttpHeaders({
                //"Content-Type": "application/json",
                sourceAppID: environment.mbs2sourceAppID,
                Authorization: `Basic ${btoa(`${environment.gdsUsername}:${environment.gdsPassword}`)}`
            })
        };
        return this.httpClient.get(`${environment.mbsBaseURL}renmbs/api/v1/bs/banks`, httpOptions);
    }


    /**
     * Returns a list of States Supported by RenMoney
     */
    getSupportedStates() {
        return this.httpClient.get(
            `${environment.investmentBaseURL}state/sub`
        ) as Observable<GeneralAPIResponse<{ name: string, code: string }[]>>;
    }


    getEmployerStates() {
        return this.httpClient.get(
            `${environment.investmentBaseURL}state/employer`
        ) as Observable<GeneralAPIResponse<{ name: string, code: string }[]>>;
    }


    /**
      * Returns a list of Companies Supported by RenMoney
      */
    getSupportedCompanies(company: string) {
        return this.httpClient.get(
            `${environment.investmentBaseURL}companies/${company}`)
    }

    /**
     * Creates a company
     */

    createCompany(company) {
        const httpOptions = {
            headers: new HttpHeaders({
                //"Content-Type": "application/json",
                "sourceapp": environment.companySourceAppID
            })
        };
        return this.httpClient.post(`${environment.mambuMiddlewareBaseURL}company`, company, httpOptions);
    }

    addCompany(newCompanyId) {
        const httpOptions = {
            headers: new HttpHeaders({
                "sourceapp": environment.companySourceAppID
            })
        };
        return this.httpClient.get(`${environment.mambuMiddlewareBaseURL}company/${newCompanyId}`, httpOptions);
    }

    /**
     * Runs Fastrack
     * @param loanId id of the loan being currently processed
     */
    getInitialLoanGDSResponse(loanDetails) {
        const httpOptions = {
            headers: new HttpHeaders({
                //"Content-Type": "application/json",
                sourceAppID: environment.mbs2sourceAppID,
                Authorization: `Basic ${btoa(`${environment.gdsUsername}:${environment.gdsPassword}`)}`
            })
        };
        return this.httpClient.post(`${environment.decisionBaseURL}api/gds/initial/v1`, loanDetails, httpOptions);
    }


    sendInternetBankingDetail(internetBankingInfo) {
        return this.httpClient.post(`${environment.rencoreBaseURL}internet-banking`, internetBankingInfo);
    }

    storeOkraRecord(bankDetails) {
        return this.httpClient.post(environment.storeOkraRecord, bankDetails);
    }

    storeMonoRecord(bankDetails) {
        return this.httpClient.post(environment.storeMonoRecord, bankDetails);
    }

    zenoCapture(payload) {
        const httpOptions = {
            headers: new HttpHeaders({
                "RENMONEY-API-KEY": environment.zenoCaptureRenmoneyKey
            })
        };
        return this.httpClient.post(environment.zenoCapture, payload, httpOptions);
    }

    updateAccount(payload) {
        return this.httpClient.patch(`${environment.apiNewBaseURL}custom-information`, payload);
    }

    updateLoanData(loanId, payload) {
        return this.httpClient.patch(`${environment.apiNewBaseURL}loans/${loanId}`, payload);
    }


    getClientStatesAndLGAs() {
        return this.httpClient.get(`${environment.apiNewBaseURL}address/states-lgas`);
    }

    /**
     * Final Decisioning
     * @param loanId id of the loan being currently processed
     */
    getFinalLoanGDSResponse(loanId) {
        const httpOptions = {
            headers: new HttpHeaders({
                //"Content-Type": "application/json",
                sourceAppID: environment.mbs2sourceAppID,
                Authorization: `Basic ${btoa(`${environment.gdsUsername}:${environment.gdsPassword}`)}`
            })
        };

        return this.httpClient.post(`${environment.decisionBaseURL}api/gds/final/v1`, { loanId: loanId }, httpOptions);
    }

    acceptFinalOffer(loanId) {
        return this.httpClient.post(`${environment.investmentBaseURL}acceptFinalOffer`, { loanId: loanId })
    }

    sendOfferLetter(loanId, clientId) {
        return this.httpClient.post(`${environment.rencoreBaseURL}offer-letter/${loanId}/${clientId}`, {})
    }



    getCreditCheck(payload) {
        const httpOptions = {
            headers: new HttpHeaders({
                Authorization: environment.creditCheckAuth
            })
        };
        return this.httpClient.post(`${environment.emailBaseURL}middleware/api/v1/credit/check`, payload, httpOptions);
    }


    /**
     * Returns a list of Sectors
     */
    getAllSectors() {
        return this.httpClient.get(`${environment.investmentBaseURL}sector/all`)
    }

    /**
     * Returns a list of industries in a sector
     */
    getIndustry(sector) {
        return this.httpClient.get(`${environment.investmentBaseURL}industries/${sector}`)
    }

    /**
     * Returns a list of LGAs Supported in a state by RenMoney
     */

    getBalanceLeft(repayments) {
        let balance = 0;

        repayments.map((repayment) => {
            let dues =
                parseFloat(repayment.feesDue) +
                parseFloat(repayment.interestDue) +
                parseFloat(repayment.penaltyDue) +
                parseFloat(repayment.principalDue);
            let paid =
                parseFloat(repayment.feesPaid) +
                parseFloat(repayment.interestPaid) +
                parseFloat(repayment.penaltyPaid) +
                parseFloat(repayment.principalPaid);

            repayment.state = repayment.state.replace("_", " ");
            repayment.total = dues - paid;

            balance += dues - paid;
        });

        return balance;
    }
