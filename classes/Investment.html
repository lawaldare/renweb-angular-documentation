import { GeneralAuth } from './client';

export class Investment extends GeneralAuth {
    id?: string; // client encoded ID
    amount?: number;
    tenure?: number;  // in days,  must not be less than 3 months
    accountState?: string; // "ACTIVE" "MATURED"
    accountType?: string; // "REGULAR_SAVINGS"
    availableBalance?: any; // 34013.89
    earnings?:string // "19.8744301369"
    interestRate?: string | number; // "5"
    maturityDate?: Date | string; //ISO date format
    name?: string; //"Regular Savings"
    proposedInvestmentAmount?: number; // 10000
    startDate?: Date | string; // "2020-05-14T15:00:11+0000"
}
