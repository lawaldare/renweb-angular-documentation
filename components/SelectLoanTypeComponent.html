import { Component, OnInit, Output, EventEmitter, Input, AfterViewInit } from '@angular/core';
import { LoanService } from 'src/app/@core/loan.service';
import { Client } from 'src/app/@model/client';

@Component({
    selector: 'app-select-loan-type',
    templateUrl: './select-loan-type.component.html',
    styleUrls: ["../../../invest/invest/invest.shared.scss",
        "../../../invest/plan-name/plan-name.component.scss", './select-loan-type.component.scss']
})
export class SelectLoanTypeComponent implements OnInit {

    @Output() done: EventEmitter<boolean> = new EventEmitter<boolean>();
    selectedType: string;
    @Input() client: Client;
    title: any;
    constructor(private loanService: LoanService) {
    }

    ngOnInit(): void {
        // this.loanService.changeStepNumber(1);
    }



    loanTypes = [
        {
            key: 'Full Time (Private)',
            title: 'I am employed',
            description: 'I work in an organization and get paid a salary '
        },
        {
            key: 'Self-Employed',
            title: 'I am a business owner',
            description: 'I own a business - I run my own business(e.g a shop) or practice a profession (e.g fashion designer)'
        }
    ];

    optionSelected() {

        this.done.emit();
    }
}
