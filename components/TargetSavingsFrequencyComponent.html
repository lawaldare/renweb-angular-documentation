import { Component, OnInit, Input, Output, EventEmitter } from "@angular/core";
import { ToastrService } from "ngx-toastr";
import * as moment from "moment/moment";
import { InvestmentInterface } from "src/app/@model/interfaces";
import { InvestmentService } from "src/app/@core/investment.service";

@Component({
    selector: "app-target-savings-frequency",
    templateUrl: "./target-savings-frequency.component.html",
    styleUrls: [
        "../create-target-savings.component.scss",
        "./target-savings-frequency.component.scss",
    ],
})
export class TargetSavingsFrequencyComponent implements OnInit {
    minDate: Date;
    maxDate: Date;
    bsConfig = { dateInputFormat: "MMMM Do YYYY", adaptivePosition: true };
    frequencyDate: any = {};
    selectedDate;
    daySelected: string;
    @Input() investment: InvestmentInterface;
    @Output() done: EventEmitter<boolean> = new EventEmitter<boolean>();
    loading: boolean = false;
    formattedMinDate;
    nextPaymentFromToday: Date;
    days = [
        'Sunday',
        'Monday',
        'Tuesday',
        'Wednesday',
        'Thursday',
        'Friday',
        'Saturday',
        
    ];

    todaysDay: number;

    constructor(
        private toastr: ToastrService,
        private investmentService: InvestmentService
    ) {
       
    }

    ngOnInit() {

        switch (this.investment.contributionFrequency) {
            case "Monthly":
                this.nextPaymentFromToday = moment().add(1, "month").toDate();
                this.minDate = moment().add(3, 'weeks').toDate();
                this.maxDate = moment().add(1, "month").endOf("month").toDate();
                this.formattedMinDate = moment(this.nextPaymentFromToday).format("Do");
                break;
            case "Weekly":
                this.nextPaymentFromToday = moment().add(1, "week").toDate();
                this.todaysDay = moment().day();
                this.minDate = moment().add(3, "days").toDate();
                this.maxDate = moment().add(1, "week").endOf("week").toDate();
                break;
            case "Daily":
                this.setDate();
                this.setupPayment()
                break;
            default:
                break;
        }
    }

    setDate(type = "default") {
        var date;

        switch (this.investment.contributionFrequency) {
            case "Monthly":
                date = moment().add(1, "month").toDate();
                break;
            case "Weekly":
                date = moment().add(1, "week").toDate();
                break;
            default:
                date = moment().add(1, "day").toDate();
                break;
        }

        this.frequencyDate = {
            type,
            date: type == "default" ? date : this.selectedDate,
        };

        this.investment.contributionStartDate = this.frequencyDate.date;
    }

    selectDate(day: string){
        let selectedDayIndex = this.days.findIndex(x=> x=== day);
        if(selectedDayIndex+1 > this.todaysDay){
            this.selectedDate = moment().day(day).toDate();
        }else{
            this.selectedDate = moment().day(day).add(7, 'days').toDate();
        }

        
        this.setDate('custom');
        
    }

    setupPayment() {
        this.loading = true;

        var tenorPaid = moment().add(1, "month").diff(moment(), "days");

        switch (this.investment.contributionFrequency) {
            case "Weekly":
                tenorPaid = 7;
                break;
            case "Daily":
                tenorPaid = 1;
                break;
            default:
                break;
        }

        let data = {
            investmentId: this.investment._id,
            minorAmount:
                (this.investment.amount -
                    parseFloat(this.investment.contributionAmount)) *
                100,
            tenorInDays:
                moment()
                    .add(this.investment.tenor, "months")
                    .diff(moment(), "days") - tenorPaid,
            startDate: this.investment.contributionStartDate,
            frequency: this.investment.contributionFrequency.toUpperCase(),
        };

        this.investmentService.setupRecurrentPayment(data).subscribe(
            (response) => {
                this.loading = false;
                this.toastr.success(
                    "Great job, your payment setup has been completed"
                );
                this.done.emit();
            },
            (error) => {
                this.loading = false;
                this.toastr.error(
                    error.error.message ||
                        "An error occurred, please try again later"
                );
            }
        );
    }
}
