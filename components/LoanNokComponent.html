import { Component, OnInit, Input, EventEmitter, Output } from '@angular/core';
import { Client } from 'src/app/@model/client';
import { ClientService } from 'src/app/@core/client.service';

@Component({
    selector: 'app-loan-nok',
    templateUrl: './loan-nok.component.html',
    styleUrls: ['../../../modals/next-of-kin-modal/next-of-kin-modal.component.scss',
        '../../../single-view/complete/complete.component.scss',
        "../../../invest/invest/invest.shared.scss",
        "../../../invest/plan-name/plan-name.component.scss",
        './loan-nok.component.scss']
})
export class LoanNokComponent implements OnInit {

    @Input() client: Client;
    @Output() done: EventEmitter<boolean> = new EventEmitter<boolean>();
    nokDetails: any = {};
    showNok = false;

    constructor(private clientService: ClientService) { }

    ngOnInit(): void {
        this.getDetailsFromMambu();
    }

    getDetailsFromMambu() {
        this.clientService.getClientFromMambu().subscribe(
            (response) => {
                let mambuClient = response.data || {};
                if (mambuClient.nokDetails) {
                    this.nokDetails = mambuClient.nokDetails;
                }
                this.showNok = true
            },
            (error) => {
                this.showNok = true
                // this.isNOKAvailable = false;
            }
        );
    }


    addNOK(client: Client) {

        this.done.emit();
    }

}
