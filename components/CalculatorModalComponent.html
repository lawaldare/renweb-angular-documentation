@Component({
    selector: "app-calculator-modal",

export class CalculatorModalComponent implements OnInit {
    amount: number;
    maxAmount = 10000000;
    minAmount = 5000;
    timeOut: any;
    scrolledPercentage = 0;

    amountCurrency: string;
    monthlyInterest: number;
    totalAmountToBePaid: number;
    tenor: number;

    interestRatesArr = [
        {
            min: 5000,
            max: 200000 - 1,
            rates: {
                365: 12,
                270: 10,
                180: 10,
                90: 7,
            },
        },
        {
            min: 200000,
            max: 1000000 - 1,
            rates: {
                365: 12,
                270: 10.75,
                180: 10.5,
                90: 10,
            },
        },
        {
            min: 1000000,
            max: 10000000,
            rates: {
                365: 14.5,
                270: 14,
                180: 12,
                90: 11,
            },
        },
    ];
    rates: number[];
    MILLISECOND_IN_A_DAY = 86400000;
    maturityDate: number;
    interest: number;
    selectedIndex: number;
    calcValues: { amount: any; tenure: any; amountToBeEarned: any };
    firstLoad: boolean = true;

    constructor(
        private toastr: ToastrService,
        private router: Router,
        public bsModalRef: BsModalRef,
        private investmentService: InvestmentService
    ) {}

    ngOnInit() {
        this.calcValues = this.investmentService.getCalcValues();
        if (this.calcValues) {
            this.amount = this.calcValues.amount;
            this.selectedIndex = Math.abs(
                (this.calcValues.tenure * 30 - 360) / 90
            );
            this.interest =
                (this.calcValues.amountToBeEarned - this.amount) / 0.9;
            setTimeout(() => {
                this.selectTenure(360 - this.calcValues.tenure * 90);
            }, 1000);
            this.getInterestRates(this.amount, true);
        } else {
            this.amount = 20000;
            this.getInterestRates(this.amount);
        }

        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
    }

    selectTenure(durationInDays, index?: number) {
        if (index !== undefined) {
            this.selectedIndex = index;
            this.monthlyInterest = this.rates[this.selectedIndex] / 12;
            this.interest =
                ((this.monthlyInterest / 100) * this.amount * durationInDays) /
                30;
        }

        this.maturityDate =
            new Date().getTime() + durationInDays * this.MILLISECOND_IN_A_DAY;
        const tenureBoxes = document.querySelectorAll(".tenure-box");
        this.clearSelectedBoxes();
        tenureBoxes[this.selectedIndex].classList.add("selected");
    }

    clearSelectedBoxes() {
        const tenureBoxes = document.querySelectorAll(".tenure-box");
        tenureBoxes.forEach((tenureBox) => {
            tenureBox.classList.replace("selected", "-");
        });
    }

    numberChecks(amount) {
        this.getInterestRates(amount);
        this.clearSelectedBoxes();
        this.firstLoad = false;
    }

    getInterestRates(amount: number, deleteInterestRate = false) {
        delete this.monthlyInterest;
        delete this.maturityDate;
        if (!deleteInterestRate) {
            delete this.interest;
            delete this.selectedIndex;
        }
        this.interestRatesArr.forEach((rateObj) => {
            if (amount >= rateObj.min && amount <= rateObj.max) {
                this.rates = Object.values(rateObj.rates).sort(
                    (a, b) => +b - +a
                );
            }
        });
    }

    /**
     * Get Investment all investment amount ranges and corresponding rates
     */
    getInvestmentRates() {
        // TODO: find out API structure for calculator
    }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "");
    }

    calculateTotalRepayment() {}

    checkIfAmountIsValid() {
        clearTimeout(this.timeOut);
        this.amount = this.currencyToNumber(this.amountCurrency);
        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
        const amount = this.amount;

        this.timeOut = setTimeout(() => {
            if (amount > this.maxAmount) {
                this.amount = this.maxAmount;
                this.amountCurrency = this.numberToCurrencyFormat(
                    this.maxAmount
                );
                this.toastr.warning(
                    "The maximum amount you can be invested is " +
                        this.maxAmount,
                    "",
                    { timeOut: 10000 }
                );
            }
            if (amount < this.minAmount) {
                this.amount = this.minAmount;
                this.amountCurrency = this.numberToCurrencyFormat(this.amount);
                this.toastr.warning(
                    "The minimum amount you can be invested is " +
                        this.minAmount
                );
            }
            this.calculateTotalRepayment();
        }, 1650);
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57)) {
            return false;
        }
        return true;
    }

    investNow(amount, tenure, interest, amountToBeEarned) {
    
        if (amount && tenure) {
            this.investmentService.saveCalcValues({
                amount,
                tenure,
                interest,
                amountToBeEarned,
            });
        }
        this.bsModalRef.hide();
        // this.router.navigate([appRoutes.page]);
    }
}
