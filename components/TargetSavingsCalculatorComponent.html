import { Component, OnInit, Input, Output, EventEmitter } from "@angular/core";
import { ToastrService } from "ngx-toastr";
import * as moment from "moment";
import { InvestmentInterface } from "src/app/@model/interfaces";
import { Location } from "@angular/common";

@Component({
    selector: "app-target-savings-calculator",
    templateUrl: "./target-savings-calculator.component.html",
    styleUrls: [
        "../../../../calculator/investment-calculator/investment-calculator.component.scss",
        "../create-target-savings.component.scss",
        "./target-savings-calculator.component.scss",
    ],
})
export class TargetSavingsCalculatorComponent implements OnInit {
    minAmount = 1000;
    minFrequencyAmount = 100; // This is because paystack doesn't accept anything less than.
    amount: number = this.minAmount * 10;
    rates: number[] = [10, 10, 10, 10];
    selectedRate: number;
    durationInMonths: number;
    durationInDays: number;
    durationInWeeks: number;
    maturityDate: Date;
    interest: number;
    fullPayout: number;
    amountFrequency: number;
    durations = ["Monthly", "Weekly", "Daily"];
    selectedDuration;
    timeOut: any;
    amountCurrency: string;
    today: any;
    @Input() amountLabel: string;
    @Input() isWithdrawal: boolean = false;
    @Input() amountToReinvest: number;
    @Input() previousDuration: number;
    @Input() copy: string =
        "Give this plan a creative name. e.g. Rainy Day, Dangote 20xx";
    @Input() planName: string;
    @Output() done: EventEmitter<InvestmentInterface> = new EventEmitter<
        InvestmentInterface
    >();
    withholdingTaxPercentage: number = 10;
    withholdingTax: number;

    momentEquivalent = {
        "Monthly": "months",
        "Weekly": "weeks",
        "Daily": "days"
    }
    dateInMonths: moment.Moment;

    constructor(private toastr: ToastrService, private location: Location) { }

    ngOnInit() {
        this.today = moment();
        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
        if (this.isWithdrawal) {
            if (this.amountToReinvest < this.minAmount) {
                this.toastr.error(
                    `The minimum amount your can reinvest is ${this.minAmount.toLocaleString()}`
                );
                return this.location.back();
            }
            this.amount = this.amountToReinvest;
            this.populateWithdrawalData();
        }
    }

    populateWithdrawalData() {
        let previousSelectedDuration =
            ((this.previousDuration - 360) / 90) * -1;

        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
        this.checkIfAmountIsValid();

        if (this.rates[previousSelectedDuration]) {
            this.selectedRate = previousSelectedDuration;
            this.selectedDuration = this.durations[0];
            this.selectTenure();
        }
    }

    submit() {
        if (this.amount < this.minAmount) {
            return this.toastr.warning(
                `The minimum amount you can invest is ${this.minAmount.toLocaleString()}`
            );
        }

        this.done.emit({
            contributionAmount: this.amountFrequency.toFixed(2),
            contributionFrequency: this.selectedDuration,
            contributionFrequencySingular: this.selectedDuration.replace('ily', 'y').replace('ly', ''),
            amount: this.amount,
            interestRate: this.rates[this.selectedRate],
            withholdingTaxPercentage: this.withholdingTaxPercentage,
            withholdingTax: this.withholdingTax.toFixed(2),
            startDate: moment().toDate(),
            maturityDate: this.maturityDate,
            totalPayOut: this.fullPayout.toFixed(2),
            tenor: this.durationInMonths,
        });
    }

    selectTenure() {
        this.durationInMonths = (360 - this.selectedRate * 90) / 30;
        this.durationInDays = this.durationInMonths * 30;
        this.durationInWeeks = Math.floor(this.durationInDays / 7);
        this.dateInMonths = moment().add(this.durationInMonths, "months");
        this.maturityDate = moment()
            .add(this.durationInMonths, "months")
            .toDate();
        this.calculateInterest();
        this.calculateAmountFrequency();
    }

    calculateAmountFrequency() {

        switch (this.selectedDuration) {
            case "Monthly":
                this.amountFrequency = this.amount / this.durationInMonths;
                // console.log(this.dateInMonths.diff(this.today, "months")); // 3 months.
                break;
            case "Weekly":
                this.amountFrequency = this.amount / this.durationInWeeks
                // console.log(this.dateInMonths.diff(this.today, "weeks")); // 13 weeks.
                break;
            case "Daily":
                this.amountFrequency = this.amount / this.durationInDays;
                // console.log(this.dateInMonths.diff(this.today, "days")); // 91 days.
                break;
            default:
                this.amountFrequency = null;
                break;
        }
    }

    clearData() {
        this.selectedRate = null;
        this.interest = null;
        this.selectedDuration = null;
    }

    calculateInterest() {
        this.interest =
            this.amount *
            ((this.rates[this.selectedRate] * this.durationInMonths) /
                (12 * 100));

        this.withholdingTax =
            this.interest * (this.withholdingTaxPercentage / 100);

        this.fullPayout = this.amount + this.interest - this.withholdingTax;
    }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "").replace("", "");
    }

    checkIfAmountIsValid() {
        delete this.selectedRate
        clearTimeout(this.timeOut);
        this.amount = this.currencyToNumber(this.amountCurrency);
        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
        const amount = this.amount;

        this.timeOut = setTimeout(() => {
            if (amount < this.minAmount) {
                this.amount = this.minAmount;
                this.amountCurrency = this.numberToCurrencyFormat(this.amount);
                this.toastr.warning(
                    `The minimum amount you can invest is ${this.amountCurrency}`
                );
                this.selectTenure();
            }
        }, 2000);

        this.selectTenure();
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57)) {
            return false;
        }
        return true;
    }
}
