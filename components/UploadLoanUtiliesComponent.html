import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { GeneralService } from 'src/app/@core/general.service';
import { Client } from 'src/app/@model/client';
import { ClientService } from 'src/app/@core/client.service';
import { ToastrService } from 'ngx-toastr';
import { DomSanitizer } from '@angular/platform-browser';
import { LoanService } from 'src/app/@core/loan.service';


@Component({
    selector: 'app-upload-loan-utilies',
    templateUrl: './upload-loan-utilies.component.html',
    styleUrls: ["../../../invest/invest/invest.shared.scss",
        "../../../invest/plan-name/plan-name.component.scss",
        './upload-loan-utilies.component.scss']
})
export class UploadLoanUtiliesComponent implements OnInit {
    @Input() client: Client;
    @Input() uploadType: string;
    @Input() appTitle: string;
    @Input() copy: string;
    @Output() done = new EventEmitter<boolean>();
    selectedPhoto: any;
    imageUrl: any;
    loading: boolean = false;
    @Input() loanID: string;
    loanEncodedKey: string;



    constructor(private generalService: GeneralService,
        private clientService: ClientService,
        public sanitizer: DomSanitizer,
        private toastr: ToastrService, private loanService: LoanService) { }

    ngOnInit(): void {
    }

    selectPhoto(event) {
        let file = event.target.files[0];
        if (file) {
            let acceptedTypes = ["image/jpeg", "image/png"];

            if (!acceptedTypes.includes(file.type)) {
                this.selectedPhoto = null;
                return this.toastr.error(
                    "Error, only jpeg or png images are allowed please."
                );
            }

            if (file.size > 2000000) {
                this.selectedPhoto = null;
                return this.toastr.error(
                    "Error, the maximum file size you can upload is 2MB."
                );
            }

            this.selectedPhoto = file;

            this.imageUrl = this.sanitizer.bypassSecurityTrustResourceUrl(
                URL.createObjectURL(this.selectedPhoto)
            );


        }
    }

    uploadDoc() {
        this.loanEncodedKey = this.loanService.getLoanEncodedKey();
        this.loading = true;
        let reader = new FileReader();
        reader.readAsDataURL(this.selectedPhoto);
        reader.onload = () => {
            let data = {
                documentType: this.uploadType,
                type: "loan_account",
                documentMime: this.selectedPhoto.type.split("/")[1],
                encodedKey: this.loanEncodedKey,
                document: reader.result.toString().split("base64,")[1],
            };

            this.clientService.uploadProfilePicture(data).subscribe(
                (response) => {
                    this.loanService.sendLoanDocsStatus({
                        documentType: this.uploadType,
                        loanId: this.loanID,
                        uploaded: true,
                        type: 'loan_account'
                    }).subscribe(data => {
                        this.loading = false;
                        this.stopEmailServiceForInitialApproval();
                        this.toastr.success(
                            `Great, ${this.uploadType} uploaded successfully.`
                        );
                        this.done.emit()
                    }, error => {
                        console.log(error);
                    })
                },
                (error) => {
                    this.loading = false;
                    this.toastr.error(
                        error.error.message ||
                        "An error occurred, please try again later."
                    );
                }
            );
        };

        reader.onerror = (error) => {
            this.loading = false;
            this.toastr.error("An error occurred, please try again.");
        };
    }


    stopEmailServiceForInitialApproval() {
        this.loanService.emailStatusUpdate({
            "recipientEmail": this.client.email,
            "templateName": "OFFER ACCEPTED, NOT SENT TO VERIFIERS",
            "status": "Done"
        }).subscribe(() => { });
    }



}
