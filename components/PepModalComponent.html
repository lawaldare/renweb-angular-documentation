import { Component, OnInit, ChangeDetectorRef } from "@angular/core";
import { BsModalRef } from "ngx-bootstrap/modal";
import { ToastrService } from "ngx-toastr";
import { Client } from "src/app/@model/client";
import { DashboardService } from "src/app/@core/dashboard.service";
import { AuthService } from "src/app/@core/auth.service";
import { ClientService } from "src/app/@core/client.service";

@Component({
    selector: "app-pep-modal",
    templateUrl: "./pep-modal.component.html",
    styleUrls: [
        "../bvn-modal/bvn-modal.component.scss",
        "./pep-modal.component.scss"
    ]
})
export class PepModalComponent implements OnInit {
    loading = false;
    client: Client;
    pepQuestions = {
        publicOfficeHolder: "Are you currently holding a Public Office",
        relatedPublicOfficeHolder:
            "Are you an immediate family member or close associate of a person currently holding public office",
        seniorPositionGovernment:
            "Are you currently holding a senior position in a government owned company",
        memberBankStaff:
            "Are you currently holding a senior position in an international organization",
        prominentMemeberPoliticalParty:
            "Are you a prominent member of a Political party",
        relatedRoyalFamily:
            "Are you a member of a ruling Royal family or close assocaiate of such family"
    };

    pepAnswers = {
        publicOfficeHolder: "",
        relatedPublicOfficeHolder: "",
        seniorPositionGovernment: "",
        memberBankStaff: "",
        prominentMemeberPoliticalParty: "",
        relatedRoyalFamily: ""
    };

    showDisableSubmit = true;
    constructor(
        private bsModalRef: BsModalRef,
        private dashboardService: DashboardService,
        private clientService: ClientService,
        private ref: ChangeDetectorRef,
        private toastr: ToastrService
    ) { }

    ngOnInit() { }

    publicOffice: string;

    getKeys(object) {
        return Object.keys(object);
    }

    log() {
        this.checkValidity();
    }

    checkValidity() {
        this.showDisableSubmit = Object.values(this.pepAnswers).includes("");
    }

    updatePepDetail(pep) {
        // pep.clientEncodedkey = this.authService.getClientEncodedkey().clientEncodedkey;
        this.loading = true;
        Object.assign(pep, this.clientService.getClientEncodedkey());
        this.dashboardService.fillPEPdetails(pep).subscribe(
            () => {
                this.toastr.success("Details updated");
                this.client.pEPValidated = true;
                this.client.pepvalidated = true;
                this.getUserData();
            },
            error => {
                this.loading = false;
                this.toastr.error("Something went wrong, Try again");
            }
        );
    }

    getUserData() {
        this.clientService.getClient().subscribe(
            data => {
                data.data["gender"] = data.data.sex;
                this.clientService.saveClientLocally(data.data);
                this.bsModalRef.hide();
                this.loading = false;
            },
            error => {
                this.loading = false;
                this.toastr.error("Something went wrong, Try again");
            }
        );
    }

    setAnswer(pepAnswerState: boolean, question: string) {
        this.pepAnswers[question] = pepAnswerState;
        this.ref.markForCheck();
        this.log();
    }
}
