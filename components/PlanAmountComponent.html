import { Component, OnInit, Input, Output, EventEmitter } from "@angular/core";
import { NgForm } from "@angular/forms";
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';
import { ToastrService } from "ngx-toastr";
import { DialogBoxComponent } from '../../shared/dialog-box/dialog-box.component';

@Component({
    selector: "app-plan-amount",
    templateUrl: "./plan-amount.component.html",
    styleUrls: ["./plan-amount.component.scss"],
})
export class PlanAmountComponent implements OnInit {
    amount: number = 0;
    timeOut: any;
    amountCurrency: string;
    @Input() buttonTitle: string = "Continue";
    @Input() title: string = "Enter amount";
    @Input() copy: string = "How much would you like to fund?";
    @Input() minAmount: number = 1000;
    @Input() amountLabel: string = "Amount";
    @Input() availableBalance: number;
    @Output() done: EventEmitter<{ amount: number }> = new EventEmitter<{
        amount: number;
    }>();

    bsModalRef: BsModalRef;


    constructor(private toastr: ToastrService, private modalService: BsModalService) { }

    ngOnInit(): void { }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `₦${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "").replace("₦", "");
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57)) {
            return false;
        }
        return true;
    }

    checkIfAmountIsValid() {
        this.amount = this.currencyToNumber(this.amountCurrency);
        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
    }

    onSubmitAmountForm(form: NgForm) {
        let amount = this.currencyToNumber(this.amountCurrency);
        if (amount < this.minAmount) {
            this.bsModalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Notice",
                    body: `The minimum amount required is ${this.numberToCurrencyFormat(this.minAmount)} `,
                    confirmText: "Yes",
                    cancelText: "No",
                },
            });
        } else {
            this.done.emit({
                amount: this.amount,
            });
        }

    }


}
