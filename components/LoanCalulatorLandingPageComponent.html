import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { appRoutes } from 'src/app/@config/routes';
import { LoanService } from 'src/app/@core/loan.service';
import { Client } from 'src/app/@model/client';

@Component({
    selector: 'app-loan-calulator-landing-page',
    templateUrl: './loan-calulator-landing-page.component.html',
    styleUrls: ["../../@page/page/page.component.scss",
        '../../@page/create/loan/select-loan-type/select-loan-type.component.scss',
        "../investment-calculator/investment-calculator.component.scss", './loan-calulator-landing-page.component.scss']
})
export class LoanCalulatorLandingPageComponent implements OnInit {

    client: Client;
    routes;

    constructor(private loanService: LoanService, private router: Router) { }

    ngOnInit(): void {
        this.routes = appRoutes;
        this.client = this.loanService.getDetailsFromLandingPage().client;
    }


    decrementState() {
        window.history.back();
    }


    populateLoanInfo(loanInfo) {
        this.loanService.saveLoanDetailFromLoanPage(loanInfo);
        this.router.navigate([appRoutes.login]);
    }

}
