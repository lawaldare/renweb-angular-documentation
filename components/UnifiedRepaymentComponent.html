import { Component, EventEmitter, OnInit, Output } from '@angular/core';
import { NgForm } from '@angular/forms';
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';
import { ToastrService } from 'ngx-toastr';
import { LoanService } from 'src/app/@core/loan.service';
import { NoCardstatusIframeComponent } from '../no-cardstatus-iframe/no-cardstatus-iframe.component';
import { RepaymentIframeComponent } from '../repayment-iframe/repayment-iframe.component';

@Component({
    selector: 'app-unified-repayment',
    templateUrl: './unified-repayment.component.html',
    styleUrls: ["../../../invest/invest/invest.shared.scss",
        "../../../invest/plan-name/plan-name.component.scss",
        '../select-loan-type/select-loan-type.component.scss',
        '../loan-calculator/loan-calculator.component.scss', './unified-repayment.component.scss']
})
export class UnifiedRepaymentComponent implements OnInit {

    @Output() done = new EventEmitter<boolean>();


    loanId: string;
    mandateDateLink: any;
    loading: boolean = false;
    modalRef: BsModalRef;
    cardList: any;

    flowState: string = "initial";
    cardStatus: string = 'No';
    cardType: string = "";
    clientFirstName: string;
    hasDebitCard: boolean = false;
    otpQuestionsArray: any;
    remitaTransRef: string;


    constructor(private loanService: LoanService, private toastr: ToastrService, private modalService: BsModalService) { }

    ngOnInit(): void {
        this.getCardTypeList();
        this.clientFirstName = this.loanService.getFormerLoanDetails().client.firstName;
    }


    getCardTypeList() {
        this.loanService.getDebitCardList().subscribe(data => {
            this.cardList = data;
        }, error => {
            this.toastr.error(error.message)
        })
    }


    getRepaymentVendor() {
        this.loading = true;
        this.getCardTypeList();
        this.loanId = this.loanService.getFormerLoanDetails().loan.loanId;
        this.loanService.getRepaymentVendor(this.loanId).subscribe((data: any) => {
            this.loading = false;
            if (data.data && data.data.link) {
                this.mandateDateLink = data.data.link;
                this.openMandateLink();
            } else if (data.code === 'Mw119') {
                // this.toastr.warning(data.message);
                this.flowState = 'card-question';
            } else {

            }

        }, error => {
            this.loading = false;
            this.toastr.error(error.error.message);
        })
    }

    requestCardStatus() {
        this.loading = true;
        let bankName = this.loanService.getFormerLoanDetails().client.bank;
        let loanId = this.loanService.getFormerLoanDetails().loan.loanId;
        const payload: any = {};
        payload.bank_name = bankName;
        payload.loan_id = loanId;
        payload.card_status = this.cardStatus;
        if (this.cardStatus === 'Yes') {
            payload.card_type = this.cardType;
        }
        this.loanService.requestCardStatus(payload).subscribe((data: any) => {
            this.loading = false;
            this.mandateDateLink = data.data.link;
            this.openNoCardStatusMandateLink();
        }, error => {
            this.loading = false;
            this.toastr.error(error.error.message)
        })


    }

    requestMandateActivationOTP() {
        this.loading = true;
        this.loanId = this.loanService.getFormerLoanDetails().loan.loanId;
        this.loanService.requestMandateActivationOTP(this.loanId).subscribe((data: any) => {
            this.loading = false;
            if (data.status !== 'SUCCESS') {
                this.flowState = 'no-card';
            } else {
                this.remitaTransRef = data.remitaTransRef;
                this.returnOTPArray(data.authParams[0]);
            }
        }, error => {
            this.loading = false;
            this.toastr.error(error.error.message);
        })
    }

    openNoCardStatusMandateLink() {
        if (this.cardStatus === 'Yes') {
            this.flowState = 'yes-card';
        } else {
            this.flowState = 'no-card'
        }

        this.modalRef = this.modalService.show(NoCardstatusIframeComponent, {
            backdrop: "static",
            class: "modal-lg",
            initialState: { link: this.mandateDateLink, cardType: this.cardStatus }
        });
    }

    openMandateLink() {
        this.flowState = 'generate-otp';
        this.modalRef = this.modalService.show(RepaymentIframeComponent, {
            backdrop: "static",
            class: "modal-lg",
            initialState: { link: this.mandateDateLink }
        });
    }


    goToNextPage() {
        this.done.emit(true);
    }


    goBackToInitialPage() {
        this.flowState = 'initial';
    }



    returnOTPArray(obj) {
        let num = Object.keys(obj).length / 3;
        const keys = ['label', 'description', 'param'];
        let form = [];

        for (let i = 1; i <= num; i++) {
            let data = {};

            for (let key of keys) {
                data[key] = obj[`${key}${i}`];
            }

            form.push(data);
        }

        this.otpQuestionsArray = form;

        if (this.otpQuestionsArray.length > 0) {
            this.flowState = 'otp-question';
        } else {
            this.flowState = 'no-card';
        }
    }


    submitOTPAnswers(form: NgForm) {
        this.loading = true;
        let num = Object.keys(form.value).length;
        let output = [];
        let i = 1;
        for (let key in form.value) {
            let data = {};
            data[`param${i}`] = key;
            data['value'] = form.value[key];
            output.push(data);
            i++;
        }

        let payload: any = {}
        payload.loan_id = this.loanId;
        payload.remitaTransRef = this.remitaTransRef;
        payload.authParams = output;

        this.loanService.validateMandateActivationOTP(payload).subscribe((data: any) => {
            this.loading = false;
            if (data?.status?.toLowerCase() === "success") {
                this.done.emit(true);
            } else {
                this.flowState = 'no-card';
                this.toastr.error(data.status);
            }
            ;
        }, error => {
            this.loading = false;
            this.toastr.error(error.message)
        })

    }




}
