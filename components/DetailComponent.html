import { appRoutes } from 'src/app/@config/routes';
import { DialogBoxComponent } from './../../shared/dialog-box/dialog-box.component';
import { Router, ActivatedRoute } from '@angular/router';
import { Component, OnInit, TemplateRef } from '@angular/core';
import { GeneralService } from 'src/app/@core/general.service';
import { InvestmentService } from 'src/app/@core/investment.service';
import { ClientService } from 'src/app/@core/client.service';
import { ToastrService } from 'ngx-toastr';
import { Client, DialogModalButton } from 'src/app/@model/client';
import { environment } from 'src/environments/environment';
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';


@Component({
    selector: 'app-detail',
    templateUrl: './detail.component.html',
    styleUrls: ['../../invest/plan-name/plan-name.component.scss', '../../invest/plan-review/plan-review.component.scss',
        './detail.component.scss']
})
export class DetailComponent implements OnInit {

    duration: number;
    accountType: string;
    width: number;
    client: Client;
    id: any;
    investment: any
    maturityDate: any;
    percentageLost = environment.percentageLost;
    timeFinishedPercent: number;
    modalRef: BsModalRef;
    dialogOption = new DialogModalButton();
    numberOfMonthsLeft: any
    numberOfMilliSecondsLeft: number
    routes = appRoutes;



    constructor(
        private router: Router,
        private route: ActivatedRoute,
        private investmentService: InvestmentService,
        private clientService: ClientService,
        private generalService: GeneralService,
        private modalService: BsModalService

    ) { }

    ngOnInit(): void {
        this.dialogOption.isConfirmed = false;
        this.route.parent.paramMap.subscribe((params) => {
            this.id = params.get("id");
            this.getSingleInvestment(this.id);
            this.client = this.clientService.getClientLocally();
        });
        this.checkedForClosedModal();
    }

    setRulePercentage() {
        const now = new Date().getTime();
        const invStartDate = new Date(this.investment.startDate).getTime();
        this.timeFinishedPercent =
            ((now - invStartDate) / 86400000 / this.investment.duration) * 100;

        this.numberOfMilliSecondsLeft = (this.investment.duration - ((now - invStartDate) / 86400000)) * 24 * 60 * 60 * 1000;

        this.millisecondsToMonthsAndDays(this.numberOfMilliSecondsLeft);

    }

    getSingleInvestment(id: string) {
        this.generalService.showLoader(true);
        this.investmentService.getSingleInvestment(id).subscribe(
            (data) => {
                this.investment = this.attachDueInvestmentKey(
                    data.data.clientDeposit
                );
                this.accountType = this.investment.accountType;

                if (this.accountType === 'SAVINGS_PLAN') {
                    this.investment.proposedInvestmentAmount = +this.investment.targetAmount;
                }
                this.maturityDate = new Date(this.investment.maturityDate).toString().split(' ').splice(1, 3).join(' ');
                this.duration = +this.investment.duration / 30;
                const interest = this.investment.proposedInvestmentAmount * (+this.investment.interestRate * (this.investment.duration / 30) / (100 * 12));
                this.investment.withHoldingTax = interest * 0.1;
                this.investment.expectedPayout = this.investment.proposedInvestmentAmount + interest * 0.9;
                this.setRulePercentage();
                this.generalService.showLoader(false);
            },
            (error) => {
                this.generalService.showLoader(false);
            }
        );
    }

    millisecondsToMonthsAndDays(number) {
        const months = number / 2592000000;
        if (months < 1) {
            let days = (+months.toFixed(1) - Math.floor(months)) * 30;
            if (days === 30) {
                days = 0;
                this.numberOfMonthsLeft = `${days} days`;
            } else {
                this.numberOfMonthsLeft = `${days} days`;
            }
        } else {
            this.numberOfMonthsLeft = `${Math.round(months)} months`;
        }
    }

    attachDueInvestmentKey(inv) {
        const now = new Date();
        const presentDate = new Date(
            `${now.getFullYear()}-${now.getMonth()}-${now.getDate()}`
        ).getTime();
        const invMaturityDate = new Date(inv.maturityDate).getTime();
        inv.isWithdrwalDue = presentDate > invMaturityDate ? true : false;
        return inv;
    }


    openWithdraw() {
        if (this.investment.accountState !== 'MATURED') {
            this.modalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                class: 'modal-position',
                initialState: {
                    type: 'confirm',
                    header: 'Sure you want to do this?',
                    body: `By withdrawing from this plan before <br> <b>${this.maturityDate}</b>, you will lose <b>${this.percentageLost}%</b> of interest earned.`,
                    cancelText: 'not now',
                    confirmText: 'proceed',
                    dialogOption: this.dialogOption
                }
            });
            return;
        } else {
            this.router.navigate([appRoutes.investment, this.id, 'withdraw', this.investment.availableBalance]);
        }
    }


    checkedForClosedModal() {
        this.modalService.onHide.subscribe(() => {
            if (this.dialogOption.isConfirmed) {
                this.router.navigate([appRoutes.investment, this.id, 'withdraw', this.investment.availableBalance]);
            }
            this.dialogOption.isConfirmed = false;
        });
    }


}
