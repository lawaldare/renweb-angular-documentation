import { Component, OnInit } from "@angular/core";
import { Client } from "src/app/@model/client";
import { AuthService } from "src/app/@core/auth.service";
import { appRoutes } from "src/app/@config/routes";
import { Router } from "@angular/router";
import { ToastrService } from "ngx-toastr";
import { DashboardService } from 'src/app/@core/dashboard.service';
import { Base64 } from 'js-base64';
import { PasswordService } from 'src/app/@core/password.service';


@Component({
    selector: "app-signup",
    templateUrl: "./signup.component.html",
    styleUrls: ["../auth/auth-mobile.scss", "./signup.component.scss"]
})


export class SignupComponent implements OnInit {
    client = new Client();
    routes = appRoutes;
    shouldShowPassword = true;
    config: any;
    passwordNextState = "Show"; // it's either show or hide
    securityQuestions = [
        "What is your favorite food?",
        "What is the name of your first phone?",
        "What is the name of your first pet?",
        "What city were you born in?",
        "What is your mother’s maiden name?",
        "What is your father's middle name?",
        "What was your first car?"
    ];



    constructor(
        private authService: AuthService,
        private router: Router,
        private toastr: ToastrService,
        private dashboardService: DashboardService,
        public passwordService: PasswordService
    ) { }

    ngOnInit() {
        // this.getConfigDetails();
        this.client.securityQuestion = "";
    }

    // getConfigDetails() {
    //     this.config = this.dashboardService.getConfigLocally();
    //     this.securityQuestions = this.config.securityQuestions;
    // }

    register(client: Client) {
        this.authService.changeState(true);
        this.authService.registerNewClient(client).subscribe(
            data => {
                this.authService.changeState(false);
                this.router.navigate([this.routes.otpVerification, Base64.encode(client.email), 'registration']);

            },
            error => {
                this.authService.changeState(false);
                this.toastr.error(error.error.message || "Couldn't create account, try again");

            }
        );
    }



    /**
     * Changes the password type of the password field
     * it has just 2 states either "Hide" or "Show"
     * @param shouldShowPassword helps to know when to hide or show password
     * @param $event the triggered event
     */
    changePasswordState(shouldShowPassword: boolean, event) {
        if (!shouldShowPassword) {
            this.passwordNextState = "Show";
            event.target.parentElement.children[1].type = "password";
            this.shouldShowPassword = !shouldShowPassword;
            return;
        }
        this.passwordNextState = "Hide";
        event.target.parentElement.children[1].type = "text";
        this.shouldShowPassword = !shouldShowPassword;
    }
}
