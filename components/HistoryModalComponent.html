import { Component, OnInit, Input } from '@angular/core';
import { BsModalRef } from 'ngx-bootstrap/modal';
import { ActivatedRoute } from '@angular/router';
import { InvestmentService } from 'src/app/@core/investment.service';
import { GeneralService } from 'src/app/@core/general.service';
import { ToastrService } from 'ngx-toastr';

@Component({
    selector: 'app-history-modal',
    templateUrl: './history-modal.component.html',
    styleUrls: ['../../single-view/history/history.component.scss', './history-modal.component.scss']
})
export class HistoryModalComponent implements OnInit {

    @Input() id: string;
    investment: any
    transactions: any
    balancePlusEarnings: number;
    duration: number;
    loading: boolean;

    constructor(
        public bsModalRef: BsModalRef,
        private investmentService: InvestmentService,
        private toastr: ToastrService
    ) { }

    ngOnInit(): void {

    }

    ngAfterViewInit(): void {
        //Called after ngAfterContentInit when the component's view has been initialized. Applies to components only.
        //Add 'implements AfterViewInit' to the class.

        // this.getSingleInvestment(this.id);
        this.getTransactionsHistory(this.id)
    }

    sendStatement() {
        this.investmentService.sendStatement(this.id).subscribe(data => {
            this.toastr.success('Statement sent successfully')
        }, error => {
            this.toastr.error('There is a problem sending your statement')
        })
    }

    getSingleInvestment(id: string) {
        this.loading = true;
        this.investmentService.getSingleInvestment(id).subscribe(
            (data) => {
                this.investment = data.data.clientDeposit;
                this.balancePlusEarnings = this.investment.availableBalance + +this.investment.earnings;
                this.duration = +this.investment.duration / 30;
                const interest = this.investment.availableBalance * (+this.investment.interestRate * (this.investment.duration / 30) / (100 * 12));
                this.investment.withHoldingTax = +this.investment.earnings * 0.1;
                this.investment.expectedPayout = this.investment.availableBalance + interest * 0.9;
                this.loading = false;
            },
            (error) => {
                this.loading = false;
            }
        );
    }

    getTransactionsHistory(id: string) {
        this.loading = true;
        this.investmentService.getTransactionsHistory(id).subscribe(
            (data) => {
                this.loading = false;
                this.transactions = data.data.clientTransactions;
            },
            (error) => {
                this.loading = false;
            }
        );
    }

}
