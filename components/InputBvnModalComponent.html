import { Component, OnInit } from '@angular/core';
import { ClientService } from 'src/app/@core/client.service';
import { ToastrService } from 'ngx-toastr';
import { DashboardService } from 'src/app/@core/dashboard.service';
import { GeneralService } from 'src/app/@core/general.service';
import { Client } from 'src/app/@model/client';
import { BsModalRef } from 'ngx-bootstrap/modal';
import * as moment from "moment/moment";
import { environment } from 'src/environments/environment';
import { LoanService } from 'src/app/@core/loan.service';

@Component({
    selector: 'app-input-bvn-modal',
    templateUrl: './input-bvn-modal.component.html',
    styleUrls: ['../../single-view/complete/complete.component.scss', './input-bvn-modal.component.scss']
})
export class InputBvnModalComponent implements OnInit {

    client: Client;
    bvn: string;
    inputBVNModalLoader: boolean = false;
    isBvnValidated: boolean

    constructor(
        public bsModalRef: BsModalRef,
        private clientService: ClientService,
        private toastr: ToastrService,
        private bvnService: DashboardService,
        private genericService: GeneralService,
        private loanService: LoanService) { }

    ngOnInit(): void {
        Object.assign(this.client, this.clientService.getClientLocally());
    }


    verifyBVN(bvn) {
        this.inputBVNModalLoader = true;
        this.bvn = bvn;
        this.bvnService.getDataFromBVN({ bvn }).subscribe(data => {
            if (this.bvnService.isBVNvalid(this.client, data.data)) {
                this.client.bvnValidated = true;
                this.client.firstName = data.data.firstName;
                this.client.lastName = data.data.lastName;
                if (data.data.middleName) {
                    this.client.middleName = data.data.middleName;
                }
                this.client.dateOfBirth = moment(data.data.DOB, 'DD-MMM-YYYY').format('YYYY-MM-DD');
                this.client.address = data.data.residentialAddress;
                this.client.gender = data.data.gender.toUpperCase();
                this.client.bvn = bvn;
                this.client.id = bvn.toString();
                this.client.documentId = bvn.toString();
                this.client.documentType = "BVN";
                this.isBvnValidated = true;

                this.clientService.getClientByBVN(bvn).subscribe(data => {
                    if (data.data) {
                        this.inputBVNModalLoader = false;
                        this.toastr.error('A user with this BVN already exists');
                    }
                }, error => {
                    if (error.error.code === environment.statusCodes.userNotFound) {
                        this.clientService.createClient(this.client).subscribe(data => {
                            this.toastr.success('BVN successfully validated!');
                            this.clientService.changeValidationState(true);
                            this.bsModalRef.hide();
                            this.inputBVNModalLoader = false;
                            this.clientService.saveClientLocallyForAccountSetup(this.client);
                        }, error => {
                            if (error.error.code === environment.statusCodes.bvnTaken) {
                                this.clientService.updateClient(this.client).subscribe(data => {
                                    this.toastr.success('BVN successfully validated!');
                                    this.clientService.changeValidationState(true);
                                    this.clientService.saveClientLocallyForAccountSetup(this.client);
                                    this.bsModalRef.hide();
                                    this.inputBVNModalLoader = false;
                                }, error => {
                                    this.inputBVNModalLoader = false;
                                    this.toastr.error(error.error.message);
                                });
                            } else {
                                this.inputBVNModalLoader = false;
                                this.toastr.error(error.error.message);
                            }
                        });
                    } else {
                        this.inputBVNModalLoader = false;
                        this.toastr.error(error.error.message);
                    }
                })

            } else {
                this.inputBVNModalLoader = false;
                this.collectDataOfUsersHavingBVNIssues();
                this.toastr.error("Your BVN does not match the name on your account")
            }
        }, error => {
            this.inputBVNModalLoader = false;
            this.collectDataOfUsersHavingBVNIssues();
            this.toastr.error(error.message ? 'You entered an invalid BVN' : 'Something went wrong while trying to verify your BVN')
        })
    }


    collectDataOfUsersHavingBVNIssues() {
        const userdata = this.clientService.getClientLocally();
        let payload: any = {};
        payload.bvn = this.bvn;
        payload.phone_number = userdata.phoneNumber;
        payload.first_name = userdata.firstName;
        payload.last_name = userdata.lastName;
        payload.email = userdata.email;

        this.loanService.zenoCapture(payload).subscribe(data => {
            console.log(data);

        }, error => console.log(error))

    }




}
