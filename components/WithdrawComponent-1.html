import { environment } from 'src/environments/environment';
import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { GeneralService } from 'src/app/@core/general.service';
import { InvestmentService } from 'src/app/@core/investment.service';
import { LoanService } from 'src/app/@core/loan.service';
import { ClientService } from 'src/app/@core/client.service';
import { Client } from 'src/app/@model/client';

@Component({
    selector: 'app-withdraw',
    templateUrl: './withdraw.component.html',
    styleUrls: ['../../invest/plan-name/plan-name.component.scss', './withdraw.component.scss']
})
export class WithdrawComponent implements OnInit {

    balance: number;
    id: string;
    investment: any
    validAmount: number = environment.minimumWithdrawalAmount + environment.fixedDepositMinimumAvailableBalance;
    isValid: boolean;
    client: Client;


    constructor(private route: ActivatedRoute, private router: Router, private generalService: GeneralService, private investmentService: InvestmentService, private loanService: LoanService, private clientService: ClientService) { }

    ngOnInit(): void {
        this.client = this.clientService.getClientLocally();
        this.route.paramMap.subscribe((params) => {
            this.balance = +params.get("balance");
        });
        this.route.parent.paramMap.subscribe((params) => {
            this.id = params.get("id");
            this.getSingleInvestment(this.id);
        });
        this.stopEmailService();
    }

    stopEmailService() {
        this.loanService.emailStatusUpdate({
            "recipientEmail": this.client.email,
            "templateName": "FIXED DEPOSIT MATURED ON DUE DATE (FIXED DEPOSIT)",
            "status": "Done"
        }).subscribe(() => { });
    }

    getSingleInvestment(id: string) {
        this.generalService.showLoader(true);
        this.investmentService.getSingleInvestment(id).subscribe(
            (data) => {
                this.investment = data.data.clientDeposit
                this.checkingIfAmountIsValid();
                this.generalService.showLoader(false);
            },
            (error) => {
                this.generalService.showLoader(false);
            }
        );
    }

    checkingIfAmountIsValid() {
        if (this.investment.availableBalance >= this.validAmount) {
            this.isValid = true;
        } else {
            this.isValid = false;
        }
    }


    goToFullPage() {
        this.router.navigate(['p/invest', this.id, 'withdraw', 'complete'])
    }

}
