    ngOnInit() {
        this.client = this.clientService.getClientLocally();

    }


    getTheRemainingDays(date: string) {
        let today = moment(new Date());
        var loanCreatedDate = moment(date);
        let diff = today.diff(loanCreatedDate, 'days');
        if (diff < 30) {
            this.remainingDays = 30 - diff;
            this.declineCard.description = `So sorry! We can offer you loan right now, try again after the wait period. Try again in ${this.remainingDays} day(s)`;
        }
        if (this.remainingDays === undefined) {
            debugger;
            this.isDueAfterDecline = true;
        } else {
            this.isDueAfterDecline = false;
        }
    }

    getLoans() {
        this.loanSubscription = this.loanService
            .getLoans(this.client.bvn)
            .subscribe(
                (response: any) => {
                    this.response = true;
                    this.generalService.showLoader(false);
                    if (response.data.length > 0) {
                        this.isThereLoan = true;
                        this.loans = response.data;
                        // this.getLoanDocStatus(this.loans[0].loan)
                        this.accountState = this.loans[0].accountState;
                        this.gdsInitialDecision = this.loans[0].gdsInitialDecision;
                        let loanAmount = (+this.loans[0].loanAmount).toFixed(2).replace(/\d(?=(\d{3})+\.)/g, '$&,');
                        this.pendingApprovalCard.description = `Your Loan application of ₦${loanAmount} is being reviewed.`;
                        this.accounts[3].action = this.loans[0].accountState === 'PARTIAL_APPLICATION' ? 'Continue' : 'Apply Now';


                        if (this.gdsInitialDecision === 'D') {
                            this.getTheRemainingDays(this.loans[0].loanCreatedDate)
                        }

                        if (Array.isArray(this.loans) && this.loans.length > 0) {
                            let loanFilter = ["ACTIVE", "ACTIVE_IN_ARREARS"];

                            this.loans.map((loan) => {
                                if (loanFilter.includes(loan.accountState)) {
                                    this.currentLoan.loan = loan;
                                    this.currentLoan.repayments = [];

                                    this.loanService
                                        .getLoanRepayment(loan.loan)
                                        .subscribe(
                                            (response: any) => {
                                                let repayments = response.data;

                                                if (
                                                    Array.isArray(repayments) &&
                                                    repayments.length > 0
                                                ) {
                                                    this.totalLoanBalance += this.loanService.getBalanceLeft(
                                                        repayments
                                                    );

                                                    this.currentLoan.repayments = repayments;
                                                    this.currentLoan.balance = this.totalLoanBalance;
                                                }
                                            },
                                            (error) => {
                                                this.generalService.showLoader(
                                                    false
                                                );
                                                this.toastr.error(
                                                    error.error.message
                                                );
                                            }
                                        );
                                }
                            });
                        }

                    } else {
                        this.isThereLoan = false;
                        this.generalService.showLoader(false);
                    }
                },
                (error) => {
                    this.response = false;
                    this.generalService.showLoader(false);
                    this.toastr.error(error.error.message);
                }
            );
    }


    ngOnDestroy() {
        this.loanSubscription.unsubscribe();
        this.generalService.showLoader(false);
    }

}
