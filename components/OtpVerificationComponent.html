import { Component, OnInit } from "@angular/core";
import { AuthService } from "src/app/@core/auth.service";
import { ToastrService } from 'ngx-toastr';
import { ActivatedRoute, Router } from '@angular/router';
import { Client } from 'src/app/@model/client';
import { appRoutes } from 'src/app/@config/routes';
import { GeneralAPIResponse } from 'src/app/@model/responses/general-api-response';
import { ClientService } from 'src/app/@core/client.service';
import { InvestmentService } from 'src/app/@core/investment.service';
import { LoanService } from 'src/app/@core/loan.service';
import { Base64 } from 'js-base64';



@Component({
    selector: "app-otp-verification",
    templateUrl: "./otp-verification.component.html",
    styleUrls: ["../auth/auth-mobile.scss", "./otp-verification.component.scss"]
})
export class OtpVerificationComponent implements OnInit {
    otp: string;
    client = new Client();
    pageToBeRedirectedTo: any;
    serviceToCall: any;
    toastMessage: string;



    constructor(private authService: AuthService,
        private toastr: ToastrService,
        private router: Router,
        private investmentService: InvestmentService,
        private loanService: LoanService,
        private clientService: ClientService,
        private route: ActivatedRoute) { }

    ngOnInit() {
        this.route.paramMap.subscribe(param => {
            this.client.service = param.get('type');
            this.client.emailOrPhone = Base64.decode(param.get('emailOrPhone'));
        })
    }



    verifyClient() {
        this.authService.changeState(true);
        let serviceToCall;
        if (this.client.service == 'password') {
            serviceToCall = undefined;
            this.pageToBeRedirectedTo = `${appRoutes.changePassword}/${Base64.encode(this.client.emailOrPhone)}/${Base64.encode(this.client.otp)}`;
        }
        if (this.client.service == 'registration') {

            serviceToCall = this.authService.completeRegistration(this.client);
            // const loanCalcDetails = this.loanService.getFormerLoanDetails();
            const loanCalcDetails = this.loanService.getDetailsFromLandingPage();
            if (loanCalcDetails.loan) {
                // this.pageToBeRedirectedTo = appRoutes.continueLoanFromDashboard;
                this.loanService.setEntryFromOnboarding(true);
                this.pageToBeRedirectedTo = appRoutes.createLoan;
            } else {
                const calcType = this.investmentService.getCalType();
                if (!calcType) this.pageToBeRedirectedTo = appRoutes.onboard;
                if (calcType) this.pageToBeRedirectedTo = `${appRoutes.onboard}/${calcType.toLowerCase()}`;
            }

            this.toastMessage = 'Awesome! Account was created successfully';
        }
        if (this.client.service == 'unblock') {

            serviceToCall = this.authService.completeUnlock(this.client);
            this.pageToBeRedirectedTo = appRoutes.login;
            this.toastMessage = 'Account was unblocked successfully';
        }

        if (serviceToCall) {
            serviceToCall.subscribe((data) => {
                if (this.client.service === 'registration') {
                    this.clientService.saveClientLocally(data.data);
                    this.authService.setUserToken(data.sessionToken);
                }
                this.authService.changeState(false);
                this.toastr.success(this.toastMessage ? this.toastMessage : 'Verification successfull');
                this.router.navigate([this.pageToBeRedirectedTo])
            }, error => {
                this.authService.changeState(false);
                this.toastr.error(error.error.message || "Couldn't verify OTP, try again");
            })
        } else {
            this.router.navigate([this.pageToBeRedirectedTo]);
            this.authService.changeState(false);
        }

    }

    resendOtp() {
        this.authService.changeState(true);
        this.authService.resendToken(this.client)
            .subscribe(() => {
                this.authService.changeState(false);
                this.toastr.success('You will receive a code (one-time-password) from us shortly');
            }, error => {
                this.authService.changeState(false);
                this.toastr.error(error.error.message || 'Something went wrong, please try again');
            })
    }
}
