import { LoanService } from 'src/app/@core/loan.service';
import { appRoutes } from './../../../../@config/routes';
import { Router } from '@angular/router';
import { Client } from './../../../../@model/client';
import { Component, OnInit, Input, EventEmitter, Output } from '@angular/core';
import { LoanInterface } from '../create-loan.component';

@Component({
    selector: 'app-loan-state',
    templateUrl: './loan-state.component.html',
    styleUrls: ['../../../invest/plan-name/plan-name.component.scss', '../../../invest/plan-review/plan-review.component.scss', '../../../../@auth/signup/signup.component.scss', './loan-state.component.scss']
})
export class LoanStateComponent implements OnInit {

    @Input() isApproved: boolean;
    @Input() client: Client;

    @Output() done: EventEmitter<boolean> = new EventEmitter<boolean>();


    constructor(private router: Router, private loanService: LoanService) { }

    ngOnInit(): void {

    }

    goToPage() {
        if (this.isApproved) {
            this.done.emit();
        } else {
            this.router.navigate([appRoutes.page]);
            this.loanService.clearFormerLoanDetail();
            this.loanService.clearLoanPageDetails();
            this.loanService.removeLoanEncodedKey();
            this.loanService.removeEntryFromOnboarding();
            this.loanService.deleteTeamLists();
            this.loanService.removeLoanID();

        }
    }

}
