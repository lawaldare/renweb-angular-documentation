import { appRoutes } from './../../../@config/routes';
import { BankStatementNoticeComponent } from './bank-statement-notice/bank-statement-notice.component';

import { Client, DialogModalButton } from 'src/app/@model/client';
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';
import { ClientService } from 'src/app/@core/client.service';
import { Router, ActivatedRoute } from '@angular/router';
import { Location } from "@angular/common";
import { ThingsToKnowComponent } from './things-to-know/things-to-know.component';
import { LoanRequirementsComponent } from './loan-requirements/loan-requirements.component';
import { DashboardService } from 'src/app/@core/dashboard.service';
import { RenConfig } from './loan-calculator/loan-config-mockdata';
import { LoanService } from 'src/app/@core/loan.service';
import { ToastrService } from 'ngx-toastr';
import { DialogBoxComponent } from '../../shared/dialog-box/dialog-box.component';

@Component({
    selector: "app-create-loan",
    templateUrl: "./create-loan.component.html",
    styleUrls: ["./create-loan.component.scss"],
})
export class CreateLoanComponent implements OnDestroy {
    state = 0;
    id: string;
    standalone: string;
    loanDetail: LoanInterface = {};
    client: Client;
    modalRef: BsModalRef;
    isNOKAvailable: boolean = false;
    dialogOption = new DialogModalButton();
    accountDetails: any;
    loanSubscription: any;
    loans: any;
    canCreateLoan: boolean = false;
    status: string;
    loanEncodedKey: string;

    states = {
        employmentType: 0,
        infoPage: 1,
        calculator: 2,
        bvn: 3,
        personalInfo: 4,
        morePersonalInfo: 5,
        nok: 6,
        companyDetails: 7,
        liquidationBank: 8,
        loanSummary: 9,
        loanState: 10,
        mobileBankingLogin: 11,
        verifyId: 12,
        uploadSelfie: 13,
        uploadUtility: 14,
        repayment: 15,
        signOffer: 16,
        loanFinal: 17

        // repayment: 2,
    };

    modalId: string;
    isApproved: boolean;
    mbsBanks: any
    loading: boolean = false;
    formerLoanDetail: { loan: any; client: any; };
    isFromCalcalatorLanding: boolean = false;
    gdsInitialDecision: string;
    remainingDays: number;
    isDueAfterDecline: boolean;
    entryMode: boolean;


    constructor(
        public locationRef: Location,
        public clientService: ClientService,
        private modalService: BsModalService,
        private dashboardService: DashboardService,
        private loanService: LoanService,
        private toastr: ToastrService,
        private router: Router,
        private route: ActivatedRoute,
        private generalService: GeneralService,
    ) {

    }

    ngOnInit() {

        this.client = this.clientService.getClientLocally();

        this.goToNextState(0);

        this.entryMode = Boolean(this.loanService.getEntryFromOnboarding());

        let tempDetails = this.loanService.getDetailsFromLandingPage();

        if (tempDetails.loan) {
            if (this.client.bvnValidated) {
                this.goToNextState(4);
            } else {
                this.goToNextState(3);
            }
            this.formerLoanDetail = this.loanService.getDetailsFromLandingPage();
            this.loanDetail = this.formerLoanDetail.loan;
            Object.assign(this.client, this.formerLoanDetail.client);
        } else {
            this.goToNextState(0);
            this.formerLoanDetail = this.loanService.getFormerLoanDetails();
        }


        if (this.formerLoanDetail.loan && !this.entryMode) {
            this.showDialog();
        }

        this.checkForStateChange();

        this.getLoans();

    }

    ngOnDestroy() {
        if (this.isFromCalcalatorLanding) {
            this.loanService.clearLoanPageDetails();
        }
    }


    checkForStateChange() {
        this.route.queryParamMap.subscribe(paramData => {
            this.state = +paramData.get('state');
            this.id = paramData.get('id');
            this.standalone = paramData.get('standalone');
        })
    }

    goToNextState(state) {
        if (state <= 9) {
            this.router.navigate([appRoutes.createLoan], { queryParams: { state: state } });
        } else if (state === 10) {
            this.router.navigate([appRoutes.createLoan], { queryParams: { state: state, status: this.status } });
        } else {
            this.router.navigate([appRoutes.createLoan], { queryParams: { state: state, id: this.loanDetail.loanId } });
        }
    }

    incrementState() {
        this.loanDetail.presentState = this.state;
        this.loanService.savePresentLoanDetail(this.loanDetail);
        this.goToNextState(this.state + 1);

    }

    employementTypeCompleted() {
        this.saveClientDetails();
    }

    paymentSetup() {
        this.finalGDS();
    }

    showDialog() {

        this.modalRef = this.modalService.show(DialogBoxComponent, {
            backdrop: "static",
            initialState: {
                type: "confirm",
                header: "Continue your pending application?",
                body: "You have an unfinished application. Want to pick up from where you stopped?",
                confirmText: "Yes",
                cancelText: "No",
                dialogOption: this.dialogOption,
                id: 'dialog'
            },
        });
    }

    saveClientDetails() {
        this.loanService.saveLoanClient(this.client);
    }


    ngAfterViewInit(): void {
        this.checkForClosedModal();
    }

    openThingsToKnowModal() {
        this.modalRef = this.modalService.show(ThingsToKnowComponent, {
            backdrop: "static",
            initialState: { clientName: this.client.firstName, id: this.modalId }
        });
    }

    startEmailServiceForInitialApproval() {
        const client = this.loanService.getFormerLoanDetails().client;
        let payload: any = {};
        payload.recipientEmail = client.email;
        payload.emailSubject = 'Just three more steps!';
        payload.recipientFirstName = client.firstName;
        payload.deepLinkUrl = `${environment.environmentLink}p/create/loan?state=11&id=${this.loanDetail.loanId}`;
        payload.templateName = "OFFER ACCEPTED, NOT SENT TO VERIFIERS";

        this.loanService.sendEmailService(payload).subscribe(() => { })
    }


    // startEmailService() {

    getLoanState() {
        this.loading = true;
        let employmentStatus = this.loanService.getFormerLoanDetails().client.employmentStatus;
        this.loanService.newInitialGDS(this.loanDetail.loanId, employmentStatus)
            .subscribe((data: { gdsResponseFlag: string }) => {
                this.loanDetail.presentState = 11;
                this.loanService.savePresentLoanDetail(this.loanDetail);

                if (data.gdsResponseFlag == 'A') {
                    this.startEmailServiceForInitialApproval();
                    this.isApproved = true;
                    this.status = 'accept';
                    // this.openRequirementModal();
                } else {
                    this.loanService.clearFormerLoanDetail();
                    this.loanService.clearLoanPageDetails();
                    this.loanService.removeEntryFromOnboarding();
                    this.loanService.deleteTeamLists();
                    this.loanService.removeLoanID();
                    this.status = 'declined';
                    this.isApproved = false;
                }
                this.incrementState();

                this.loading = false;
            }, error => {
                if (environment.production) {
                    this.toastr.error('Oops, something seems wrong. Try again.');
                } else {
                    this.toastr.error(error?.error?.message);
                }
                this.loading = false;

            })
    }

    finalGDS() {
        this.loading = true;
        this.loanService.newFinalGDS(this.loanDetail.loanId)
            .subscribe((data: { gdsResponseFlag: string }) => {
                this.isApproved = true;
                this.incrementState();
                this.loading = false;
            }, error => {
                this.toastr.error(error?.error?.message);
                this.loading = false;
            })
    }


    openRequirementModal() {
        this.modalRef = this.modalService.show(LoanRequirementsComponent, {
            backdrop: "static",
            initialState: { clientName: this.client.firstName, id: this.modalId }
        });
    }


    deleteEntryMode() {
        if (this.loanService.getEntryFromOnboarding()) {
            this.loanService.removeEntryFromOnboarding();
        }
    }


    decrementState() {
        if (this.state === this.states.bvn + 1) {
            this.goToNextState(this.state - 2);
        } else if (this.state === this.states.nok + 1) {
            this.goToNextState(this.state - 2);
        } else if (this.state > 0) {
            this.goToNextState(this.state - 1);
        } else {
            this.router.navigate([appRoutes.dashboard])
        }

        // this.loanService.stepNumber$.subscribe(stepNumber => {
        //     this.loanService.changeStepNumber(stepNumber - 1);
        // })

    }

    populateLoanInfoFromBVN() {
        let tempDetails = this.loanService.getDetailsFromLandingPage();
        if (tempDetails.loan) {
            this.loanService.savePresentLoanDetail(tempDetails.loan);
            Object.assign(this.loanDetail, tempDetails.loan);
        }
    }

    updateClient(client: Client) {
        Object.assign(this.client, client);
        this.clientService.saveClientLocally(this.client);
    }

    populateLoanInfo(loanInfo) {
        this.loanService.savePresentLoanDetail(loanInfo);
        Object.assign(this.loanDetail, loanInfo);
    }

    populateAccountDetailInfo(accountDetails) {
        this.accountDetails = accountDetails;
        this.loanService.saveAccountDetailsLocally(accountDetails);
    }


    showNokIfSelfEmployed() {
        if (this.client.employmentStatus === 'Full Time (Private)') {
            this.state++;
            this.goToNextState(this.state + 1);
        }
        this.incrementState();
    }

    checkIfBVNIsValidated() {
        if (this.client.bvnValidated) {
            this.state++;
            this.goToNextState(this.state + 1);
        } else {
            this.incrementState();
        }
    }


    checkForClosedModal() {
        this.modalService.onHide.subscribe(() => {
            if (this.dialogOption.isConfirmed) {
                this.loanDetail = this.formerLoanDetail.loan;
                Object.assign(this.client, this.formerLoanDetail.client);
                if (this.loanDetail.presentState) {
                    this.goToNextState(this.loanDetail.presentState + 1);
                }


            } else if (this.dialogOption.isCancelled) {
                if (this.dialogOption.id === 'dialog') {
                    this.goToNextState(0);
                    delete this.dialogOption.id;
                }
            }
            this.dialogOption.isConfirmed = false;
        });
    }

    sendOfferLetter() {
        this.loading = true;
        this.loanService.sendOfferLetter(this.loanDetail.loanId, this.client.bvn).subscribe(data => {
            this.loading = false;
        }, error => {
            this.toastr.error(error.message);
            this.loading = false;
        })
    }

    getLoans() {
        this.generalService.showLoader(true);
        this.loanService.getLoans(this.client.bvn).subscribe((response: any) => {
            this.generalService.showLoader(false);
            if (response.data.length > 0) {
                this.loans = response.data;
                this.gdsInitialDecision = this.loans[0].gdsInitialDecision;
                this.getTheRemainingDays(this.loans[0].loanCreatedDate);
                if (this.gdsInitialDecision === 'D' && !this.isDueAfterDecline === true) {
                    this.toastr.warning(`So sorry! We can not offer you loan right now, try again after the wait period. Try again in ${this.remainingDays} day(s)`);
                    this.router.navigate([appRoutes.dashboard]);
                }
            };
        }, (error) => {
            this.generalService.showLoader(false);
            this.toastr.error(error.error.message);
        });

    }

    getTheRemainingDays(date: string) {
        let today = moment(new Date());
        var loanCreatedDate = moment(date);
        let diff = today.diff(loanCreatedDate, 'days');
        if (diff < 30) {
            this.remainingDays = 30 - diff;
        }

        if (this.remainingDays === undefined) {
            this.isDueAfterDecline = true;
        } else {
            this.isDueAfterDecline = false;
        }
    }

}

export interface LoanInterface {
    amount?: number;
    tenure?: number;
    purpose?: string;
    loanId?: number;
    presentState?: number;
}
