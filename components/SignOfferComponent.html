    sendOfferLetter() {
        this.innerLoading = true;
        this.loanService.sendOfferLetter(this.loanDetail.loanId, this.client.bvn).subscribe(data => {
            this.innerLoading = false;
            this.toastr.success('Please sign your loan offer letter to proceed. An email with the loan offer has been re-sent');
        }, error => {
            this.toastr.error(error.message);
            this.innerLoading = false;
        })
    }

    checkOfferLetterStatus() {
        this.innerLoading = true;
        this.loanService.getLoans(this.client.bvn).subscribe((response: any) => {
            if (response.data.length > 0) {
                this.loan = response.data;
                if (this.loan[0].contractSignedDate) {
                    this.innerLoading = false;
                    this.done.emit();
                    // this.loanService.acceptFinalOffer(this.loanDetail.loanId).subscribe(data => {
                    //     this.innerLoading = false;
                    //     this.done.emit();
                    // }, error => {
                    //     this.toastr.error(error.message);
                    //     this.innerLoading = false;
                    // })
                } else {
                    this.sendOfferLetter();
                }

            } else {
                this.innerLoading = false;
            }
        },
            (error) => {
                this.innerLoading = false;
                this.toastr.error(error.error.message);
            }
        );
    }
