import { Component, OnInit, Input, Output, EventEmitter, AfterViewInit } from '@angular/core';
import { Client } from 'src/app/@model/client';
import { LoanService } from 'src/app/@core/loan.service';
import { ToastrService } from 'ngx-toastr';
import { environment } from 'src/environments/environment';
import { ClientService } from 'src/app/@core/client.service';

@Component({
    selector: 'app-verify-id',
    templateUrl: './verify-id.component.html',
    styleUrls: ["../../../invest/invest/invest.shared.scss",
        "../../../invest/plan-name/plan-name.component.scss",
        './verify-id.component.scss']
})
export class VerifyIdComponent implements OnInit, AfterViewInit {

    @Input() client: Client;
    @Output() done = new EventEmitter<any>();
    idCard: string;
    defaultMessage: string = 'We are unable to verify the details on your ID card right now. Please try again or contact us via 0700 5000 500 or hello@renmoney.com';

    idNumber;
    loading: boolean = false;
    idTypes = Object.keys(environment.allowedIDTypes);
    selectedIDType: string;
    // selectedIDType = this.loanService.getFormerLoanDetails().client.documentType;
    clientDocumentType: string;
    constructor(private loanService: LoanService,
        private toastr: ToastrService, private clientService: ClientService) { }

    ngOnInit(): void {
        this.selectedIDType = this.loanService.getFormerLoanDetails().client.documentType;
    }

    ngAfterViewInit() {
        this.selectedIDType = this.loanService.getFormerLoanDetails().client.documentType;
    }


    selectID(id) {
        this.client.documentType = id;
    }

    verify(idNumber) {
        this.loading = true;
        const { firstName, lastName, phoneNumber, bvn, documentType, dob } = this.client;
        const idType = environment.allowedIDTypes[documentType];
        const verifyObj = {
            idNumber,
            phoneNumber,
            country: "NG",
            clientId: bvn,
            idType,
            firstName,
            lastName,
            dob
        }
        this.client.id = this.client.bvn;
        if (this.client.gender) {
            this.client.gender = this.client.gender.toUpperCase();
        }
        if (this.client.sex) {
            this.client.sex = this.client.sex.toUpperCase();
        }
        this.loanService.verifyID(verifyObj).subscribe((data: any) => {
            if (data.code === 'SmID1013' || data.status === 'FAILED') {
                this.toastr.warning(data.message);
                this.loading = false;
            } else if (data.code === 'SmID107') {
                this.toastr.error(data.friendlyMessage);
                this.loading = false;
            } else {
                this.clientService.updateClient(this.client).subscribe(data => {
                    this.loading = false;
                    this.done.emit();
                }, error => {
                    this.toastr.error(error.error);
                })
            }

        }, error => {
            this.loading = false;
            this.toastr.error(error.friendlyMessage || this.defaultMessage);
        })
    }

}
