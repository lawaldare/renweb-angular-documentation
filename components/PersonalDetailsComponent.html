import { Component, OnInit, Input, Output, EventEmitter, AfterViewInit } from '@angular/core';
import { Client } from 'src/app/@model/client';
import { DashboardService } from 'src/app/@core/dashboard.service';
import * as moment from 'moment';
import { DialogBoxComponent } from 'src/app/@page/shared/dialog-box/dialog-box.component';
import { BsModalService } from 'ngx-bootstrap/modal';
import { LoanInterface } from '../create-loan.component';
import { environment } from 'src/environments/environment';
import { LoanService } from 'src/app/@core/loan.service';
import { ToastrService } from 'ngx-toastr';


@Component({
    selector: 'app-personal-details',
    templateUrl: './personal-details.component.html',
    styleUrls: ["../../../invest/invest/invest.shared.scss",
        "../../../invest/plan-name/plan-name.component.scss",
        '../select-loan-type/select-loan-type.component.scss',
        '../loan-calculator/loan-calculator.component.scss',
        './personal-details.component.scss']
})
export class PersonalDetailsComponent implements OnInit, AfterViewInit {

    @Input() client: Client;
    @Output() done = new EventEmitter<Client>();
    @Input() loanDetail: LoanInterface;

    dmi;
    dateFromCalender = "";
    residentialStatuses: string[] = [];
    bsConfig = { dateInputFormat: "DD/MM/YYYY", adaptivePosition: true };
    modalRef: any;
    loanType: string;
    teamList: any;
    socialMediaData: any;
    loan: any;
    loading: boolean = false;


    constructor(private toastr: ToastrService, private dashboardService: DashboardService, private modalService: BsModalService, private loanService: LoanService) { }

    ngOnInit(): void {
        this.residentialStatuses = this.dashboardService.getConfigLocally().residentialStatus;
        this.teamList = this.loanService.getTeamLists();
        this.socialMediaData = this.loanService.getSocialMediaData();

        console.log(this.socialMediaData);


    }

    ngAfterViewInit() {
        if (this.client.dateMovedIn) {
            this.dmi = new Date(this.client.dateMovedIn);
        }
    }

    setDOB(event) {
        this.dateFromCalender = event;
        this.dmi = this.dateFromCalender;
    }

    updatePersonalInfo(client) {
        this.loading = true;
        if (this.dateFromCalender) {
            this.client.dateMovedIn = moment(this.dmi, 'DD-MMM-YYYY').format('YYYY-MM-DD');
            this.createLoan();
        } else {
            this.modalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Oops!",
                    body: `Please enter a valid date`,
                    confirmText: "Done",
                },
            });
        }

    }


    createLoan() {
        const config = this.dashboardService.getConfigLocally();
        this.loanType = this.client.employmentStatus === 'Full Time (Private)' ? config.plLoanType : config.mblLoanType;
        const loanObj = {
            client_id: this.client.bvn,
            client_type: "CLIENT",
            loan_type: this.loanType,
            amount: this.loanDetail.amount,
            tenor: this.loanDetail.tenure,
            customInformation: [
                {
                    "customFieldID": "Processing_Branch_Loan_Accounts",
                    "value": "Ikoyi"
                },
                {
                    "customFieldSetGroupIndex": 0,
                    "customFieldID": "Loan_Type_Loan_Accounts",
                    "value": "New"
                },
                {
                    "customFieldID": "Channel_Loan",
                    "value": "Web"
                },
                {
                    "customFieldSetGroupIndex": 0,
                    "customFieldID": "Loan_Purpose",
                    "value": environment.production ? this.loanDetail.purpose : this.loanDetail.purpose.toUpperCase()
                },
                {
                    "customFieldSetGroupIndex": 0,
                    "customFieldID": "Repayment_Method_Loan_Accounts",
                    "value": "Direct Debit"
                },
                {
                    "customFieldSetGroupIndex": 0,
                    "customFieldID": "Beneficiary_Account_Type_Loan_Ac",
                    "value": "Default"
                },
            ]
        };
        if (this.teamList && this.teamList.salesSubTeam) {
            loanObj.customInformation.push({
                "customFieldID": "salesSubTeam",
                "value": this.teamList.salesSubTeam
            })
        }
        if (this.teamList && this.teamList._loansalesAgent) {
            loanObj.customInformation.push({
                "customFieldID": "_loansalesAgent",
                "value": this.teamList._loansalesAgent
            })
        }
        if (this.teamList && this.teamList._salesteamLead) {
            loanObj.customInformation.push({
                "customFieldID": "_salesteamLead",
                "value": this.teamList._salesteamLead
            })
        }


        this.loanService.getLoans(this.client.bvn).subscribe((response: any) => {
            if (response.data.length > 0) {
                // let loanId = this.loanService.getLoanID();
                let loanId = response.data[0].loan;
                let payload: any = {};
                payload.loan_amount = this.loanService.getFormerLoanDetails().loan.amount;
                payload.loan_tenor = this.loanService.getFormerLoanDetails().loan.tenure;
                payload.loan_purpose = this.loanService.getFormerLoanDetails().loan.purpose;

                this.loanService.updateLoanData(loanId, payload).subscribe(data => {
                    Object.assign(this.loanDetail, { loanId: loanId });
                    this.loading = false;
                    this.done.emit(this.client);
                }, error => {
                    this.loading = false;
                    this.toastr.error(error?.message, 'Oops!');
                })

            } else {
                this.loanService.createLoan(loanObj).subscribe((data: any) => {
                    this.loanService.saveLoanEncodedKey(data.data.loanAccount.encodedKey)
                    Object.assign(this.loanDetail, { loanId: data.data.loanAccount.id });
                    this.loanService.saveLoanID(data.data.loanAccount.id);
                    this.loanService.deleteTeamLists();
                    if (this.socialMediaData) {
                        this.sendSocialMediaData();
                    }
                    // this.loanService.deleteSocialMediaData();
                    this.loading = false;
                    this.done.emit(this.client);
                }, error => {
                    this.loading = false;
                    this.toastr.error(error?.message, 'Oops!');
                })
            }
        },
            (error) => {
                this.loading = false;
                this.toastr.error(error.error.message);
            }
        );


    }

    sendSocialMediaData() {
        const socialMediaData = {
            "id": this.client.bvn,
            "type": "clients",
            "fields": this.socialMediaData
        }
        this.loanService.updateAccount(socialMediaData).subscribe(() => { });
    }

}
