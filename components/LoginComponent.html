import { Component, OnInit } from "@angular/core";
import { AuthService } from "src/app/@core/auth.service";
import { Client } from "src/app/@model/client";
import { Router, ActivatedRoute } from "@angular/router";
import { ToastrService } from "ngx-toastr";
import { appRoutes } from "src/app/@config/routes";
import { environment } from 'src/environments/environment';
import { ClientService } from 'src/app/@core/client.service';
import { LoanService } from 'src/app/@core/loan.service';
import { Base64 } from 'js-base64';


@Component({
    selector: "app-login",
    templateUrl: "./login.component.html",
    styleUrls: ["../auth/auth-mobile.scss", '../signup/signup.component.scss', "./login.component.scss"]
})
export class LoginComponent implements OnInit {

    client = new Client();
    routes = appRoutes;
    shouldShowPassword = true;
    passwordNextState = 'Show'; // it's either show or hide
    returnUrl: string;
    id: string;
    clientExists: boolean = false;
    headerText: string = 'Log in or sign up';
    subHeaderText: string = "Let's get you started!";

    constructor(private authService: AuthService,
        private loanService: LoanService,
        private router: Router,
        private route: ActivatedRoute,
        private toastr: ToastrService,
        private clientService: ClientService
    ) { }

    ngOnInit() {
        this.getReturnUrl();
    }

    getReturnUrl() {
        this.route.queryParamMap.subscribe((param) => {
            this.returnUrl = param.get("returnUrl");
        });
    }


    login(client: Client) {
        client.username = client.emailOrPhone;
        this.authService.changeState(true);
        this.authService.login(client).subscribe(data => {
            // if (data.data.phoneValidated) {
            this.authService.changeState(false);
            this.authService.setUserToken(data.sessionToken);
            this.clientService.saveClientLocally(data.data);
            const loanCalcDetails = this.loanService.getDetailsFromLandingPage();
            if (loanCalcDetails.loan) {
                this.loanService.setEntryFromOnboarding(true);
                this.router.navigate([this.routes.createLoan]);
            } else if (this.returnUrl) {
                this.router.navigateByUrl(this.returnUrl);
            } else {
                this.router.navigate([this.routes.page]);
            }


            // } else {
            //     this.sendOtp(client.emailOrPhone);
            // }

        },
            error => {
                this.authService.changeState(false);

                if (error.error.code === environment.statusCodes.pendingOtp) {
                    this.sendOtp(client.emailOrPhone)
                }

                if (error.error.code === 'MwE500') {
                    //do nothing;
                } else {
                    this.toastr.error(error.error.message || "Oh snap! Please check your login details and try again");
                }


                if (error.error.code === environment.statusCodes.blockedUser) {
                    this.router.navigate([this.routes.unblock, Base64.encode(error.error.data.email), Base64.encode(error.error.data.securityQuestion)]);
                }
            })

    }


    checkIfClientExists(client: Client) {
        this.authService.changeState(true);
        this.authService.checkIfClientsExist({ email: client.emailOrPhone }).subscribe((data: any) => {
            if (data.data) {
                this.authService.changeState(false)
                this.clientExists = true;
                this.headerText = 'Log in to your dashboard';
                this.subHeaderText = 'Welcome back! Enter your password';
            }
        }, error => {
            this.authService.changeState(false);
            if (error.error.code === environment.statusCodes.userNotFound) {
                this.router.navigate([this.routes.register]);
            }
        })
    }


    sendOtp(emailOrPhone) {
        this.authService.changeState(true);
        this.router.navigate([this.routes.otpVerification, Base64.encode(emailOrPhone), 'registration']);
        this.authService.generateClientOTP({ emailOrPhone: emailOrPhone, service: 'registration' })
            .subscribe(() => {
                this.authService.changeState(false);
                this.toastr.success('an OTP has been sent to your phone number');
            }, error => {
                this.authService.changeState(false);
                this.toastr.error("OTP wasn't sent, Kindly click on the resend button to send OTP");
            })


    }

    /**
     * Changes the password type of the password field
     * it has just 2 states either "Hide" or "Show"
     * @param shouldShowPassword helps to know when to hide or show password
     * @param $event the triggered event
     */
    changePasswordState(shouldShowPassword: boolean, event) {
        if (!shouldShowPassword) {
            this.passwordNextState = "Show";
            event.target.parentElement.children[1].type = "password";
            this.shouldShowPassword = !shouldShowPassword;
            return;
        }
        this.passwordNextState = "Hide";
        event.target.parentElement.children[1].type = "text";
        this.shouldShowPassword = !shouldShowPassword;
    }

}
