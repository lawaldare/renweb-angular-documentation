import { Component, OnInit } from "@angular/core";
import { DialogModalButton } from "src/app/@model/client";
import { NgForm } from "@angular/forms";
import { ClientService } from "src/app/@core/client.service";
import { ToastrService } from "ngx-toastr";
import { BsModalRef } from "ngx-bootstrap/modal";

@Component({
    selector: "app-change-password-modal",
    templateUrl: "./change-password-modal.component.html",
    styleUrls: ["../../../@auth/signup/signup.component.scss", "./change-password-modal.component.scss"],
})
export class ChangePasswordModalComponent implements OnInit {
    id: string = "changePassword";
    oldPassword: string;
    newPassword: string;
    shouldShowPassword = true;
    shouldShowNewPassword = true;
    loading: boolean = false;
    passwordState = "Show";
    newPasswordState = "Show";
    dialogOption = new DialogModalButton();

    constructor(
        public clientService: ClientService,
        public modalRef: BsModalRef,
        public toastr: ToastrService
    ) { }

    ngOnInit(): void { }

    showPassword(password, showPassword, event) {
        if (!this[showPassword]) {
            this[password] = "Show";
            event.target.parentElement.children[1].type = "password";
            this[showPassword] = !this[showPassword];
            return;
        }
        this[password] = "Hide";
        event.target.parentElement.children[1].type = "text";
        this[showPassword] = !this[showPassword];
    }

    changeUserPassword(form: NgForm) {
        this.loading = true;
        this.clientService.validatePassword(form.value.oldPassword).subscribe(data => {
            this.clientService.changePassword(form.value).subscribe(
                (data) => {
                    this.loading = false;
                    this.toastr.success(data.message);
                    this.dialogOption = Object.assign(this.dialogOption, {
                        isConfirmed: true,
                        id: this.id,
                    });
                    this.modalRef.hide();
                    form.resetForm();
                },
                (error) => {
                    this.dialogOption = Object.assign(this.dialogOption, {
                        isConfirmed: false,
                        id: this.id,
                    });
                    this.loading = false;
                    this.toastr.error(error.error.message);
                }
            );
        }, error => {
            this.loading = false;
            this.toastr.error('Current password incorrect');
        })

    }
}
