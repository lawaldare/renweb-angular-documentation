import { appRoutes } from 'src/app/@config/routes';
import { ClientService } from "../../../@core/client.service";
import { Component, OnInit, AfterViewInit } from "@angular/core";
import { Client } from "../../../@model/client";
import { GeneralService } from 'src/app/@core/general.service';
import { BankData } from '../../single-view/banks';
import { ToastrService } from 'ngx-toastr';
import { InvestmentService } from 'src/app/@core/investment.service';
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';
import { InputBvnModalComponent } from '../../modals/input-bvn-modal/input-bvn-modal.component';
import { AddBankModalComponent } from '../../modals/add-bank-modal/add-bank-modal.component';
import { Subscription } from 'rxjs';

@Component({
    selector: "app-account-payment",
    templateUrl: "./payment.component.html",
    styleUrls: ["./payment.component.scss"],
})
export class PaymentComponent implements OnInit, AfterViewInit {
    client = new Client();
    addOrChangeCard = false;
    bsModalRef: BsModalRef;
    selectedOption = 'card';
    cardDetails: any;
    bank: any;
    accountDetails: any = {};
    banks = BankData;
    statusMessage: string;
    loading: boolean;
    accountNameFetched: boolean;
    investmentSubscription: Subscription;
    clientSubscription: Subscription;
    isBVNValidated: boolean = false;
    accountSetUpRoute = appRoutes.accountSetUp;

    constructor(private clientService: ClientService,
        private investmentService: InvestmentService,
        private toastr: ToastrService,
        private modalService: BsModalService,
        private generalService: GeneralService) { }

    ngOnInit(): void {
        this.getClientCardDetails();


    }

    ngAfterViewInit() {
        this.getMambuDetails();
    }

    getClientCardDetails() {
        this.client = this.clientService.getClientLocally();
        if (this.client.bvnValidated) {
            this.isBVNValidated = true;
        }

        this.cardDetails = this.client.cardDetails;
        if (this.cardDetails) {
            this.cardDetails.cardType = this.cardDetails.cardType.split(" ")[0];
            this.cardDetails.bin = this.cardDetails.bin.substring(0, 4);
        }
    }


    getMambuDetails() {
        this.generalService.showLoader(true);
        this.clientSubscription = this.clientService.getClientFromMambu()
            .subscribe(
                (data) => {
                    if (data.data) {
                        if (data.data.bankLiquidationDetails) {
                            this.accountDetails.bank = data.data.bankLiquidationDetails.nameOfBank;
                            this.accountDetails.accountNumber = data.data.bankLiquidationDetails.accountNumber;
                            this.getAccountName(this.banks.find(
                                (x) => x.Name.toLowerCase() === this.accountDetails.bank.toLowerCase()
                            ).Code, this.accountDetails.accountNumber);
                        }
                    }


                    this.generalService.showLoader(false);
                },
                (error) => {

                    this.generalService.showLoader(false);
                }
            );
    }

    getAccountName(bankCode, accountNumber) {
        this.loading = true;
        this.generalService.showLoader(true);
        this.investmentSubscription = this.investmentService
            .getAccountName(
                bankCode,
                accountNumber
            )
            .subscribe(
                (data: any) => {
                    this.generalService.showLoader(false);
                    if (data.message === "Invalid Account Number") {
                        this.statusMessage =
                            "Invalid bank and account number combination";
                        this.loading = false;
                        return;
                    }
                    this.accountDetails.accountName = data.data.account_name;
                    this.accountNameFetched = true;
                    this.loading = false;
                },
                (error) => {
                    this.loading = false;
                    this.generalService.showLoader(false);
                    this.statusMessage = "Kindly try again";
                    this.toastr.error(
                        "Error occured during account name enquiry"
                    );
                }
            );
    }


    openAddBank() {
        this.bsModalRef = this.modalService.show(AddBankModalComponent, {
            backdrop: "static",
            initialState: { client: this.client, accountDetails: this.accountDetails }
        });
    }

    paymentSuccess() {
        this.toastr.success('Card successfully tokenized');
        this.addOrChangeCard = false;
        location.reload();
        this.getUserData();
    }

    paymentFailed() {
        this.toastr.error('Something went wrong while tokenizing card, try again');
        this.addOrChangeCard = false;
    }

    getUserData() {
        this.generalService.showLoader(true);
        this.clientService.getClient().subscribe(
            (data) => {
                this.generalService.showLoader(false);
                data.data["gender"] = data.data.sex;
                this.clientService.saveClientLocally(data.data);


            },
            () => {
                this.generalService.showLoader(false);
                this.toastr.error(
                    "Something went wrong while getting the client"
                );
            }
        );
    }

    ngOnDestroy() {
        // this.investmentSubscription.unsubscribe();
        // this.clientSubscription.unsubscribe();
        this.generalService.showLoader(false)
    }




}
