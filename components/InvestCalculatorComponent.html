import { DashboardService } from 'src/app/@core/dashboard.service';
import { Component, OnInit, Output, EventEmitter, Input } from "@angular/core";
import { appRoutes } from "src/app/@config/routes";
import { ToastrService } from "ngx-toastr";
import { Router } from "@angular/router";
import { InvestmentService } from "src/app/@core/investment.service";
import { InvestmentInterface } from "src/app/@model/interfaces";
import { BsModalRef, BsModalService } from "ngx-bootstrap/modal";
import { DialogBoxComponent } from '../../shared/dialog-box/dialog-box.component';


@Component({
    selector: "app-deposit-calculator",
    templateUrl: "./invest-calculator.component.html",
    styleUrls: [
        "../../../@page/page/page.component.scss",
        "../invest/invest.shared.scss",
        "../../../calculator/investment-calculator/investment-calculator.component.scss",
        "../plan-review/plan-review.component.scss",
        "./invest-calculator.component.scss",
    ],
})
export class InvestCalculatorComponent implements OnInit {
    @Output() done: EventEmitter<InvestmentInterface> = new EventEmitter<
        InvestmentInterface
    >();
    @Input() copy: string;
    @Input() showCopy = true;
    @Input() isPartialWithdrawal: boolean = false;
    @Input("amount") amountToBeInvested: number;

    amount: number;
    maxAmount = 10000000;
    minAmount = 10000;
    timeOut: any;
    scrolledPercentage = 0;

    bsModalRef: BsModalRef;


    amountCurrency: string;
    monthlyInterest: number;
    totalAmountToBePaid: number;
    tenor: number;
    config: any;

    interestRatesArr: any;
    rates: any[];
    MILLISECOND_IN_A_DAY = 86400000;
    maturityDate: number;
    interest: number;
    selectedIndex: number;
    calcValues: { amount: any; tenure: any; amountToBeEarned: any };
    firstLoad: boolean = true;
    width: number;

    constructor(
        private toastr: ToastrService,
        private router: Router,
        private investmentService: InvestmentService,
        private dashboardService: DashboardService,
        private modalService: BsModalService
    ) { }

    ngOnInit() {
        this.getConfigDetails();
        this.width = 100 / Object.keys(this.interestRatesArr[0].rates).length;
        if (this.isPartialWithdrawal) {
            this.amount = +this.amountToBeInvested;
        } else {
            this.amount = 20000;
        }
        this.getCalcValues();

        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
    }

    getCalcValues() {
        this.calcValues = this.investmentService.getCalcValues();
        if (this.calcValues && !this.isPartialWithdrawal) {
            this.amount = this.calcValues.amount;
            this.selectedIndex = Math.abs(
                (this.calcValues.tenure * 30 - 360) / 90
            );
            this.interest =
                (this.calcValues.amountToBeEarned - this.amount) / 0.9;
            setTimeout(() => {
                this.selectTenure(360 - this.calcValues.tenure * 90);
                this.monthlyInterest = this.rates[this.selectedIndex] / 12;
            }, 1000);
            this.getInterestRates(this.amount, true);
        } else {
            this.getInterestRates(this.amount);
        }
    }

    selectTenure(durationInDays, index?: number) {
        if (index !== undefined) {
            this.selectedIndex = index;
            this.monthlyInterest = this.rates[this.selectedIndex] / 12;
            this.interest =
                ((this.monthlyInterest / 100) * this.amount * durationInDays) /
                30;
        }

        this.maturityDate =
            new Date().getTime() + durationInDays * this.MILLISECOND_IN_A_DAY;
        const tenureBoxes = document.querySelectorAll(".tenure-box");
        this.clearSelectedBoxes();
        tenureBoxes[this.selectedIndex].classList.add("selected");
    }

    clearSelectedBoxes() {
        const tenureBoxes = document.querySelectorAll(".tenure-box");
        tenureBoxes.forEach((tenureBox) => {
            tenureBox.classList.replace("selected", "-");
        });
    }

    numberChecks(amount) {
        this.getInterestRates(amount);
        this.clearSelectedBoxes();
        this.firstLoad = false;
    }

    getInterestRates(amount: number, deleteInterestRate = false) {
        delete this.monthlyInterest;
        delete this.maturityDate;
        if (!deleteInterestRate) {
            delete this.interest;
            delete this.selectedIndex;
        }

        this.interestRatesArr.forEach((rateObj) => {
            if (amount >= rateObj.min && amount <= rateObj.max) {
                this.rates = Object.values(rateObj.rates).sort(
                    (a, b) => +b - +a
                );
            }
        });

        if (this.rates && this.selectedIndex) this.monthlyInterest = this.rates[this.selectedIndex] / 12;
    }

    /**
     * Get Investment all investment amount ranges and corresponding rates
     */
    getInvestmentRates() {
        // TODO: find out API structure for calculator
    }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "").replace("", "");
    }

    calculateTotalRepayment() { }

    checkIfAmountIsValid() {
        clearTimeout(this.timeOut);
        this.amount = this.currencyToNumber(this.amountCurrency);
        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
        const amount = this.amount;

        this.timeOut = setTimeout(() => {
            if (amount < this.minAmount) {
                this.amount = this.minAmount;
                this.amountCurrency = this.numberToCurrencyFormat(this.amount);
                this.toastr.warning(
                    "The minimum amount you can invest is " +
                    this.minAmount
                );
            }
            if (amount > this.maxAmount) {
                this.amount = this.maxAmount;
                this.amountCurrency = this.numberToCurrencyFormat(this.amount);
                this.toastr.warning(
                    "The maximum amount you can invest is " +
                    this.maxAmount
                );
            }
            this.calculateTotalRepayment();
        }, 1650);
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57)) {
            return false;
        }
        return true;
    }

    showCurrency() {
        this.amount = this.currencyToNumber(this.amountCurrency);
        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
    }

    investNow(amount, tenure, interest, amountToBeEarned) {
        if (amount < this.minAmount) {
            this.bsModalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Notice",
                    body: `The minimum amount you can invest is ${this.minAmount}`,
                    confirmText: "Yes",
                    cancelText: "No",
                },
            });
        } else if (amount > this.maxAmount) {
            this.bsModalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Notice",
                    body: `The maximum amount you can invest is ${this.maxAmount}`,
                    confirmText: "Yes",
                    cancelText: "No",
                },
            });
        } else {
            if (amount && tenure) {
                this.investmentService.saveCalcValues({
                    amount,
                    tenure,
                    interest,
                    amountToBeEarned,
                });
            }

            let withholdingTax = (interest * 0.1).toFixed(2);

            this.done.emit({
                amount,
                withholdingTaxPercentage: 10,
                withholdingTax,
                tenor: tenure,
                interestRate: this.monthlyInterest * 12,
                totalPayOut: amountToBeEarned,
                startDate: new Date(),
            });
        }


    }

    getInterestConfigRates(bounds, rates) {
        let outputArray = [];
        for (let i = 0; i < bounds.length; i++) {
            let obj = {};
            if (i === 0) {
                obj['min'] = this.config.fixedInvestmentMinimum;
            } else {
                obj['min'] = bounds[i - 1];
            }
            obj['max'] = bounds[i] - 1;
            obj['rates'] = {};

            for (let r = 0; r < rates.length; r++) {
                let temp = {};
                if (r === rates.length - 1) {
                    temp[+rates[r]['numberOfMonths'] * 30 + 5] = rates[r][`bound${i + 1}`];
                    Object.assign(obj['rates'], temp);
                } else {
                    temp[+rates[r]['numberOfMonths'] * 30] = rates[r][`bound${i + 1}`];
                    Object.assign(obj['rates'], temp);
                }

            }
            outputArray.push(obj);
        }

        this.interestRatesArr = outputArray;
    }


    getConfigDetails() {
        this.config = this.dashboardService.getConfigLocally();
        this.getInterestConfigRates(this.config['fixedInvestmentBounds'], this.config['fixedInvestmentRates']);
    }


}
