    ngOnInit() {
        this.route.queryParamMap.subscribe((params) => {
            if (params.get("reload")) {
                location.href = appRoutes.dashboard;
            }
        });

        this.client = this.clientService.getClientLocally();
        if (!this.client.nuban && this.client.clientEncodedkey) {
            this.retrieveNuban();
        } else {
            this.getInvestments();
        }




        this.actions = [
            {

            },
        ];
    }

    getClientProfile() {

            (response) => {

                let mambuClient = response.data || {};
                if (
                    !mambuClient.nokDetails ||
                    !mambuClient.bankLiquidationDetails ||
                    !this.client.bvnValidated
                ) {
                    this.accountSetupComplete = false;
                }
            },
            (error) => {
                this.accountSetupComplete = false;
            }
        );
    }

    retrieveNuban() {
        this.clientService.getClientNuban(this.client.clientEncodedkey)
            .subscribe((data) => {
                this.client.nuban = data.data.nuban;
                this.getInvestments();
            }, error => {
                this.toastr.error(error.error.message);
                this.getInvestments();
            })
    }

    getInvestments() {
        this.loading = true;
        this.investmentSubscription = this.investmentService.getInvestments().subscribe((data) => {
            this.loading = false;
            const clientDeposits = data.data.clientDeposits;
            if (clientDeposits.length > 0) {
                this.isThereTransaction = true;
                clientDeposits.forEach((item) => {

                    if (item.accountType != "REGULAR_SAVINGS") {
                        this.portfolioBalance += parseFloat(
                            item.availableBalance
                        );
                    }

                    // Smart Goal total and count
                    if (item.accountType == "SAVINGS_PLAN") {

                        this.totalTargetSavingsBalance += parseFloat(
                            item.availableBalance
                        );
                        this.targetSavingsEarnings += parseFloat(
                            item.earnings
                        );

                        this.totalTargetSavings++;
                    }

                    // Fixed deposit total and count
                    if (item.accountType == "FIXED_DEPOSIT") {

                        this.totalFixedDepositBalance += parseFloat(
                            item.availableBalance
                        );
                        this.fixedDepositEarnings += parseFloat(
                            item.earnings
                        );

                        this.totalFixedDeposits++;
                    }


                    // Pocket balance
                    if (item.accountType == "REGULAR_SAVINGS") {
                        this.isThereTransaction = true;
                        if (item.id == this.client.nuban) {
                            this.totalRenflexSavingsBalance = parseFloat(
                                item.availableBalance
                            ) ? parseFloat(
                                item.availableBalance
                            ) : 0;
                            this.renflexSavingsEarnings = parseFloat(
                                item.earnings
                            ) ? parseFloat(
                                item.earnings
                            ) : 0;

                            this.portfolioBalance += parseFloat(
                                item.availableBalance
                            ) ? parseFloat(
                                item.availableBalance
                            ) : 0;

                            this.renflexSavings = item;
                        }
                    }
                });

            } else {
                this.isThereTransaction = false;
            }
        },
            (error) => {
                this.loading = false;
                // this.toastr.error(error.error.message);
            }
        );
    }

    getTheRemainingDays(date: string) {
        let today = moment(new Date());
        var loanCreatedDate = moment(date);
        let diff = today.diff(loanCreatedDate, 'days');

        if (diff < 30) {
            this.remainingDays = 30 - diff;
            this.declineCard.description = `So sorry! We can not offer you loan right now, try again after the wait period. Try again in ${this.remainingDays} day(s)`
        }
        if (this.remainingDays === undefined) {
            this.isDueAfterDecline = true;
        } else {
            this.isDueAfterDecline = false;
        }
    }

    getLoans() {
        this.generalService.showLoader(true);
        this.loanSubscription = this.loanService
            .getLoans(this.client.bvn)
            .subscribe(
                (response: any) => {
                    this.response = true;
                    this.generalService.showLoader(false);
                    if (response.data.length > 0) {
                        this.isThereLoan = true;
                        this.loans = response.data;
                        // this.getLoanDocStatus(this.loans[0].loan)
                        this.accountState = this.loans[0].accountState;
                        this.gdsInitialDecision = this.loans[0].gdsInitialDecision;
                        let loanAmount = (+this.loans[0].loanAmount).toFixed(2).replace(/\d(?=(\d{3})+\.)/g, '$&,');
                        this.cards[2].description = `You are about to borrow ₦${loanAmount}.`;
                        this.pendingApprovalCard.description = `Your Loan application of ₦${loanAmount} is being reviewed.`

                        if (this.gdsInitialDecision === 'D') {
                            this.getTheRemainingDays(this.loans[0].loanCreatedDate)
                        }

                        if (Array.isArray(this.loans) && this.loans.length > 0) {
                            let loanFilter = ["ACTIVE", "ACTIVE_IN_ARREARS"];

                            this.loans.map((loan) => {
                                if (loanFilter.includes(loan.accountState)) {
                                    this.currentLoan.loan = loan;
                                    this.currentLoan.repayments = [];

                                    this.loanService
                                        .getLoanRepayment(loan.loan)
                                        .subscribe(
                                            (response: any) => {
                                                let repayments = response.data;

                                                if (
                                                    Array.isArray(repayments) &&
                                                    repayments.length > 0
                                                ) {
                                                    this.totalLoanBalance += this.loanService.getBalanceLeft(
                                                        repayments
                                                    );

                                                    this.currentLoan.repayments = repayments;
                                                    this.currentLoan.balance = this.totalLoanBalance;
                                                }
                                            },
                                            (error) => {
                                                this.generalService.showLoader(
                                                    false
                                                );
                                                this.toastr.error(
                                                    error.error.message
                                                );
                                            }
                                        );
                                }
                            });
                        }

                    } else {
                        this.isThereLoan = false;
                        this.generalService.showLoader(false);
                    }
                },
                (error) => {
                    this.generalService.showLoader(false);
                    this.toastr.error(error.error.message);
                }
            );
    }

    openPortfolioModal() {
        this.bsModalRef = this.modalService.show(PortfolioComponent, {
            initialState: {
                client: this.client,
                portfolioBalance: this.portfolioBalance,
                totalTargetSavingsBalance: this.totalTargetSavingsBalance,
                targetSavingsEarnings: this.targetSavingsEarnings,
                totalFixedDepositBalance: this.totalFixedDepositBalance,
                fixedDepositEarnings: this.fixedDepositEarnings,
                totalRenflexSavingsBalance: this.totalRenflexSavingsBalance,
                renflexSavingsEarnings: this.renflexSavingsEarnings,
            },
        });
    }

    openLoanRepaymentsModal() {
        this.bsModalRef = this.modalService.show(LoanRepaymentsModalComponent, {
            initialState: {
                ...this.currentLoan,
            },
        });
    }

    openHistoryModal() {
        if (this.isThereTransaction) {
            this.bsModalRef = this.modalService.show(HistoryModalComponent, {
                class: 'bg-transparent',
                initialState: {
                    id: this.renflexSavings.id,
                },
            });
        } else {
            this.bsModalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Transaction",
                    body: "No transaction",
                    confirmText: "Yes",
                    cancelText: "No",
                },
            });
        }

    }

    ngOnDestroy() {
        this.investmentSubscription.unsubscribe();
        this.loanSubscription.unsubscribe();
        this.generalService.showLoader(false);
    }
}
