import { Component, OnInit, ViewChild, ViewChildren, AfterViewInit, ElementRef, QueryList } from '@angular/core';

@Component({
  selector: 'app-stepper',
  templateUrl: './stepper.component.html',
  styleUrls: ['./stepper.component.scss']
})
export class StepperComponent implements OnInit, AfterViewInit {


  stepsArray = [
    {
      title: 'Account Created',
    },
    {
      title: 'Review Investment',
    },
    {
      title: 'Fund Investment',
    },
    {
      title: 'Secure Account',
    }
  ]

  currentStep : number = 1;
  numberOfSteps: number = this.stepsArray.length;
  stepper: any;
  steps: any;

  @ViewChild('stepper', {static: false}) _stepper: ElementRef;
  @ViewChildren('step') _steps: QueryList<any>;



  constructor() { }

  ngOnInit(): void {

  }


  ngAfterViewInit():void{
    this.stepper = this._stepper.nativeElement;
    this.steps = this._steps['_results'];
    this.addClass(this.steps[0]['nativeElement'], 'editing');
  }


  nextStep(){
    this.currentStep++;
    if(this.currentStep > this.numberOfSteps){
      this.currentStep = this.numberOfSteps;
    }

    Array.from(this.steps).forEach((step, index) => {
      let stepNumber = index + 1;
      if(stepNumber === this.currentStep) {
        this.addClass(step['nativeElement'], 'editing');
      } else {
        this.removeClass(step['nativeElement'], 'editing');
      }

      if(stepNumber < this.currentStep){
        this.addClass(step['nativeElement'], 'done');
      } else {
        this.removeClass(step['nativeElement'], 'done');
      }
    })
  }

hasClass(elem, className) {
    return new RegExp(' ' + className + ' ').test(' ' + elem.className + ' ');
}

addClass(elem, className) {
    if (!this.hasClass(elem, className)) {
        elem.className += ' ' + className;
    }
}

removeClass(elem, className) {
    var newClass = ' ' + elem.className.replace( /[\t\r\n]/g, ' ') + ' ';
    if (this.hasClass(elem, className)) {
        while (newClass.indexOf(' ' + className + ' ') >= 0 ) {
            newClass = newClass.replace(' ' + className + ' ', ' ');
        }
        elem.className = newClass.replace(/^\s+|\s+$/g, '');
    }
}

}

<div class="stepper-horizontal" #stepper>
