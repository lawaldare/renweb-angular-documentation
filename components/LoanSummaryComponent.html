import { ClientService } from './../../../../@core/client.service';
import { environment } from 'src/environments/environment';
import { Component, OnInit, EventEmitter, Output, Input } from '@angular/core';
import { Client } from 'src/app/@model/client';
import { DashboardService } from 'src/app/@core/dashboard.service';
import { LoanService } from 'src/app/@core/loan.service';
import { ToastrService } from 'ngx-toastr';
import { LoanInterface } from '../create-loan.component';

@Component({
    selector: 'app-loan-summary',
    templateUrl: './loan-summary.component.html',
    styleUrls: ['../../../shared/page-description/page-description.component.scss', '../../../invest/plan-name/plan-name.component.scss', '../../../invest/plan-review/plan-review.component.scss', './loan-summary.component.scss']
})
export class LoanSummaryComponent implements OnInit {

    @Output() done: EventEmitter<any> = new EventEmitter<any
    >();
    @Input() client: Client;
    @Input() loanDetail: LoanInterface;
    @Input() accountDetails: {
        bank?: string,
        accountNumber?: string,
        bankCode?: string;
        accountName?: string
    } = {};
    loanType: string;
    loading: boolean;
    teamList: any

    constructor(private dashboardService: DashboardService,
        private loanService: LoanService,
        private toastr: ToastrService, private clientService: ClientService) { }

    ngOnInit(): void {
        if (!this.accountDetails?.bank) {
            this.accountDetails = this.loanService.getAccountDetailsLocally();
        }

        this.teamList = this.loanService.getTeamLists();
    }



    updateClientOnMambu() {
        this.loading = true;
        this.client = this.loanService.getFormerLoanDetails().client;
        this.client.id = this.client.bvn;
        if (this.client.gender) {
            this.client.gender = this.client.gender.toUpperCase();
        }
        if (this.client.sex) {
            this.client.sex = this.client.sex.toUpperCase();
        }

        this.clientService.updateClient(this.client).subscribe(data => {
            if (this.client.employmentStatus === 'Full Time (Private)') {
                let payload: any = {};
                payload.loanId = this.loanService.getFormerLoanDetails().loan.loanId;
                payload.clientId = this.client.id;
                payload.cbReport = '1';
                this.loanService.getCreditCheck(payload).subscribe(data => {
                    this.loading = false;
                    this.done.emit(this.loanDetail);
                }, error => {
                    this.toastr.error(error.error.message);
                    this.loading = false;
                })
            } else {
                this.loading = false;
                this.done.emit(this.loanDetail);
            }

        }, error => {
            this.loading = false;
            this.toastr.error(error.error.message)
        })


    }
