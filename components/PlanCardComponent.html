import {
    Component,
    Input,
    Output,
    EventEmitter,
    HostListener,
} from "@angular/core";
import { appRoutes } from 'src/app/@config/routes';
import { Router } from '@angular/router';

@Component({
    selector: "app-plan-card",
    templateUrl: "./plan-card.component.html",
    styleUrls: ["./plan-card.component.scss"],
})
export class PlanCardComponent {
    @Input() name: string;
    @Input() oldPlanName: string;
    @Input() type: string;
    @Input() amount: number;
    @Input() interest: string | number;
    @Input() maturityDate: Date;
    @Input() investmentId: string;
    @Input() columnOnly: boolean = false; //to only use flex-column, default is flex-sm-row flex-column


    constructor(private router: Router) {

    }


    @HostListener("click")
    goToInvestmentPage() {

        this.router.navigate([appRoutes.investment, this.investmentId]);

    }
}
