import { InvestmentService } from 'src/app/@core/investment.service';
import { NgForm } from '@angular/forms';
import { Component, OnInit, TemplateRef } from '@angular/core';
import { Location } from '@angular/common';
import { BsModalService, BsModalRef } from 'ngx-bootstrap/modal';
import { ToastrService } from 'ngx-toastr';
import { ActivatedRoute } from '@angular/router';
import Swal from 'sweetalert2/dist/sweetalert2.js';




@Component({
    selector: 'app-single-view',
    templateUrl: './single-view.component.html',
    styleUrls: ['./single-view.component.scss']
})
export class SingleViewComponent implements OnInit {

    // showOptionPlan: boolean = false;
    modalRef: BsModalRef;
    loading: boolean = false;
    investmentId: any
    investment: any;
    isZero: boolean = false;
    planName: string;
    oldPlanName: string


    constructor(private location: Location,
        private modalService: BsModalService,
        private investmentService: InvestmentService,
        private toastr: ToastrService,
        private route: ActivatedRoute,
    ) { }

    ngOnInit(): void {
        this.route.paramMap.subscribe((params) => {
            this.investmentId = params.get("id");
            this.getSingleInvestment(this.investmentId);

        });
    }

    getSingleInvestment(id: string) {
        this.investmentService.getSingleInvestment(id).subscribe(
            (data) => {
                this.investment = this.attachDueInvestmentKey(
                    data.data.clientDeposit
                );
                this.oldPlanName = this.investment.planName ? this.investment.planName : this.investment.name;
                this.checkAvailableBalance(+this.investment.availableBalance);
            },
            (error) => {
            }
        );
    }

    checkAvailableBalance(num) {
        if (num === 0) {
            this.isZero = true;
        } else {
            this.isZero = false;
        }
    }

    attachDueInvestmentKey(inv) {
        const now = new Date();
        const presentDate = new Date(
            `${now.getFullYear()}-${now.getMonth()}-${now.getDate()}`
        ).getTime();
        const invMaturityDate = new Date(inv.maturityDate).getTime();
        inv.isWithdrwalDue = presentDate > invMaturityDate ? true : false;
        return inv;
    }


    goBack() {
        this.location.back();
    }

    // displayOptionPlans() {
    //     this.showOptionPlan = !this.showOptionPlan
    // }

    displayOptionPlans(event) {
        const element = event.currentTarget.parentElement.children[1].children[2];
        if (element.className === '') {
            element.className = 'visible';
        } else {
            element.className = '';
        }
    }



    editPlanName(planNameModal: TemplateRef<any>) {
        this.modalRef = this.modalService.show(planNameModal);
        // this.showOptionPlan = false;
    }

    deletePlan() {
        // this.showOptionPlan = false;
        Swal.fire({
            title: 'Are you sure?',
            text: `You will not be able to recover ${this.oldPlanName} plan`,
            icon: 'warning',
            showCancelButton: true,
            confirmButtonText: 'Yes, delete it!',
            cancelButtonText: 'No, keep it'
        }).then((result) => {
            if (result.value) {
                this.investmentService.deletePlan(this.investmentId).subscribe(data => {
                    Swal.fire(
                        'Deleted!',
                        'Plan deleted successfully',
                        'success'
                    )

                }, error => {
                    this.toastr.error(
                        'There is a problem deleting the plan'
                    );
                })

            }
        })
    }

    onSubmitPlanNameForm(form: NgForm) {
        this.loading = true;
        this.investmentService.editPlanName(form.value).subscribe(data => {
            this.loading = false;
            this.toastr.success('Plan name edited successfully')
            form.resetForm();
            this.modalRef.hide();
        }, error => {
            this.loading = false;
            form.resetForm();
            this.modalRef.hide();
            this.toastr.error(
                'There is a problem editing the plan name'
            );
        })


    }
