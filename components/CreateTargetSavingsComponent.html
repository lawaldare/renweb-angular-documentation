import { Component, OnInit } from "@angular/core";
import { Location } from "@angular/common";
import { InvestmentInterface } from "src/app/@model/interfaces";
import { BsModalRef, BsModalService } from "ngx-bootstrap/modal";
import { DialogBoxComponent } from "../../shared/dialog-box/dialog-box.component";
import { DialogModalButton, Client } from "src/app/@model/client";
import { ClientService } from "src/app/@core/client.service";
import { Router, ActivatedRoute } from "@angular/router";
import { appRoutes } from "src/app/@config/routes";
import { InvestmentService } from 'src/app/@core/investment.service';

@Component({
    selector: "app-create-target-savings",
    templateUrl: "./create-target-savings.component.html",
    styleUrls: ["./create-target-savings.component.scss"],
})
export class CreateTargetSavingsComponent implements OnInit {
    client: Client = {};
    state = 0;
    modalRef: BsModalRef;
    investment: InvestmentInterface = {
        planType: "targetSavings",
        useContributionAmount: true,
    };
    dialogOption = new DialogModalButton();
    states = {
        planName: 0,
        calculator: 1,
        planReview: 2,
        bvn: 3,
        planFund: 4,
        targetSavingsWithdraw: 5,
        targetSavingsPartialWithdraw: 6,
        repayment: 7,
        frequency: 8,
        pep: 9
    };

    constructor(
        public location: Location,
        public clientService: ClientService,
        private modalService: BsModalService,
        private investmentService: InvestmentService,
        private router: Router,
        private route: ActivatedRoute
    ) { }

    ngOnInit() {
        this.client = this.clientService.getClientLocally();
        this.checkForClosedModal();
        this.checkForStateChange();
        this.goToNextState(0);
    }

    checkForStateChange() {
        this.route.queryParamMap.subscribe(paramData => {
            this.state = +paramData.get('state');
        })
    }

    goToNextState(state) {
        this.router.navigate([appRoutes.createTargetSavings], { queryParams: { state: state } });
    }

    incrementState() {
        this.goToNextState(this.state + 1);
    }

    getInvestment() {
        this.investment = this.investmentService.getSmartGoalInvestmentLocally();
    }


    decrementState() {
        if (this.state > 0) {
            this.goToNextState(this.state - 1);
        }
        else this.router.navigate([appRoutes.viewTargetSavings])
    }

    assignPlanName(planName: string) {
        this.investment = { ...this.investment, planName };
        this.incrementState();
    }

    assignCalculatorVariables(data: Object) {
        this.investment = { ...this.investment, ...data };
        this.incrementState();
    }

    bvnValidated() {
        this.incrementState();
    }

    planApproved() {
        // if (this.client.bvnValidated) {
        //     this.incrementState();
        // }
        this.modalRef = this.modalService.show(DialogBoxComponent, {
            backdrop: "static",
            initialState: {
                type: "confirm",
                header: "Accept Terms & Conditions",
                body: `I accept to save ${parseFloat(this.investment.contributionAmount).toLocaleString()} in my ${this.investment.planName
                    } Smart goal plan ${this.investment.contributionFrequency
                    }. I understand that full interest will be earned only if I meet all ${this.investment.contributionFrequency
                    } savings.`,
                confirmText: "Continue",
                cancelText: "Cancel",
                dialogOption: this.dialogOption,
            },
        });
    }

    paymentDone(event) {
        if (!event) {
            this.incrementState();
            this.investment = this.investmentService.getSmartGoalInvestmentLocally();
        } else {
            this.goToNextState(this.state + 4);
            this.investment = this.investmentService.getSmartGoalInvestmentLocally();
        }

        // if (this.investment.paymentMade) {
        //     this.modalRef = this.modalService.show(DialogBoxComponent, {
        //         backdrop: "static",
        //         initialState: {
        //             type: "info",
        //             id: "payment",
        //             header: "Awesome!",
        //             body: `You have successfully funded your plan.`,
        //             confirmText: "Done",
        //             dialogOption: this.dialogOption,
        //         },
        //     });
        // }
    }

    completeWithdraw(event) {
        this.goToNextState(this.state + event)
    }

    checkForClosedModal() {
        this.modalService.onHide.subscribe(() => {
            if (
                this.dialogOption.isConfirmed &&
                this.dialogOption.id === "payment"
            ) {
                this.incrementState();
            } else if (this.dialogOption.isConfirmed) {
                this.clientService.getClientLocally().bvnValidated
                    ? (this.goToNextState(this.state + 2))
                    : this.incrementState();
            }
            this.dialogOption.isConfirmed = false;
        });
    }



    recurrentSetup() {
        if (!this.client.pepvalidated) {
            this.incrementState();
        }
        this.router.navigate([appRoutes.dashboard]);
    }

    pepDone() {
        this.router.navigate([appRoutes.dashboard]);
    }
}
