import {
    Component,
    OnInit,
    Input,
    AfterViewInit,
    Output,
    EventEmitter,
    ChangeDetectionStrategy,
    TemplateRef,
} from "@angular/core";
import { CurrencyPipe } from "@angular/common";
import { Client } from "src/app/@model/client";
import { ClientService } from "src/app/@core/client.service";
import { InvestmentInterface } from "src/app/@model/interfaces";
import { GeneralService } from "src/app/@core/general.service";
import { InvestmentService } from "src/app/@core/investment.service";
import { ToastrService } from "ngx-toastr";
import { Router } from "@angular/router";
import { Investment } from 'src/app/@model/investment';
import { BsModalService, BsModalRef } from 'ngx-bootstrap/modal';
import { DialogBoxComponent } from '../../shared/dialog-box/dialog-box.component';

@Component({
    selector: "app-plan-fund",
    templateUrl: "./plan-fund.component.html",
    styleUrls: ['../../withdraw/withdraw.component.scss',
        "../../single-view/partial-withdrawal/partial-withdrawal.component.scss",
        "../plan-name/plan-name.component.scss",
        "./plan-fund.component.scss",
    ],
})
export class PlanFundComponent implements OnInit, AfterViewInit {

    amountToTransfer: string;
    availableBalance: number;
    modalRef: BsModalRef;
    tempAmountToTransfer: number;
    amountToTransferDub: number;
    timeOut: any;





    @Input() investment: InvestmentInterface = { paymentOption: '' };
    @Input() paymentType = 'Investment'
    @Input() isItSmartGoal: boolean;
    @Input() showNotification: boolean;
    @Input() pageTitle: string = "How will you like to fund your plan?";
    @Input() description: string = "Select a preferred payment method";
    @Output() done: EventEmitter<boolean> = new EventEmitter<boolean>();
    @Input() availableOptions = ["debitCard", "bankTransfer", "renflex"];

    selectedPaymentOption: any = {};
    client: Client = {};
    cardDetails: { last4: string; bank: string; cardType: string; bin: string };
    paymentOptions = [
        {
            type: "debitCard",
            hidden: false,
            left: {
                title: "Debit Card",
                icon: "payment-card",
                svgWidth: 16,
                svgHeight: 16,
                svgViewBox: "0 -2 21 21",
            },
        },
        {
            type: "bankTransfer",
            hidden: true,
            left: {
                title: "Bank Transfer",
                icon: "payment-bank",
                svgWidth: 18,
                svgHeight: 18,
                svgViewBox: "0 0 22 22",
            },
        },
        {
            type: "renflex",
            hidden: true,
            left: {
                title: "Renmoney Account",
                icon: "renflex",
                svgWidth: 18,
                svgHeight: 18,
                svgViewBox: "0 0 18 18",
            },
            right: {
                title: "Balance",
                description: {
                    default: 0,
                    value: new CurrencyPipe("en-US").transform(0, "₦"),
                },
            },
        },
    ];
    paymentOption: any = {};
    loading: boolean = true;
    loadingP: boolean = false;


    renflexAccount: any;

    constructor(
        public generalService: GeneralService,
        private clientService: ClientService,
        private toastr: ToastrService,
        private router: Router,
        private investmentService: InvestmentService,
        private modalService: BsModalService,
    ) { }

    ngOnInit(): void {
        this.paymentOptions = this.paymentOptions
            .filter((option) => this.availableOptions.includes(option.type))
            .map((option) => ({ ...option, hidden: false }));

        this.client = this.clientService.getClientLocally();

        this.cardDetails = this.client.cardDetails;
        if (this.cardDetails) {
            this.cardDetails.cardType = this.cardDetails.cardType.split(" ")[0];
        }

        if (this.investment) {
            if (this.investment._id) {
                this.loading = false;
            }

            this.selectedPaymentOption.type = this.investment.paymentOption;
        }

        if (this.availableOptions.includes("renflex")) {
            if (!this.client.nuban) {
                this.retrieveNuban();
            } else {
                this.getSingleInvestment(this.client.nuban);
            }
        }

        if (this.showNotification) {
            this.showDialog();
        }
    }

    showDialog() {
        this.modalRef = this.modalService.show(DialogBoxComponent, {
            backdrop: "static",
            initialState: {
                type: "confirm",
                header: "Notice",
                body: "We will never share your card information with anyone. The ₦10 payment is to verify that your card can be charged",
                confirmText: "Okay",
            },
        });
    }

    retrieveNuban() {
        this.clientService.getClientNuban(this.client.clientEncodedkey)
            .subscribe((data) => {
                this.client.nuban = data.data.nuban;
                this.getSingleInvestment(this.client.nuban);
            }, () => {
                this.toastr.error("Couldn't retrieve your pocket account");
            })
    }

    /**
     * Get Renflex account
     * @param id Nuban number which is also Regular savings ID
     */
    getSingleInvestment(id) {

        this.loading = true;
        this.investmentService.getSingleInvestment(id).subscribe(
            (data) => {
                this.checkForRegularSavings(data.data.clientDeposit);

                this.loading = false;
            },
            (error) => {
                this.loading = false;
                this.toastr.error("Couldn't retrieve your pocket account");
            }
        );
    }

    checkForRegularSavings(data: Investment) {
        this.renflexAccount = data
        let renflexOptionIndex = this.paymentOptions.findIndex(
            (item) => item.type == "renflex"
        );

        if (this.renflexAccount && renflexOptionIndex != -1) {
            this.paymentOptions[
                renflexOptionIndex
            ].right.description.value = new CurrencyPipe("en-US").transform(
                this.renflexAccount.availableBalance || 0,
                "₦"
            );
            // this.availableBalance = this.renflexAccount.availableBalance;

            if (this.isItSmartGoal) {
                if (+this.investment.contributionAmount <= this.renflexAccount.availableBalance) {
                    return (this.paymentOptions[renflexOptionIndex].hidden = false);
                } else {
                    return (this.paymentOptions[renflexOptionIndex].hidden = true);
                }
            } else {
                if (this.investment.amount <= this.renflexAccount.availableBalance) {
                    return (this.paymentOptions[renflexOptionIndex].hidden = false);
                } else {
                    return (this.paymentOptions[renflexOptionIndex].hidden = true);
                }
            }



        }
    }

    transfer() {
        // console.log(this.tempAmountToTransfer);
        if (this.isItSmartGoal) {
            this.done.emit(false);
        } else {
            let payload: any = {};
            payload.debitAccountId = this.renflexAccount.id;
            payload.creditAccountId = this.investment._id;
            // payload.minorAmount = +this.tempAmountToTransfer * 100;
            payload.minorAmount = this.investment.amount * 100;
            payload.narration = `Funded ${this.investment.planName} from regular savings`;
            // this.loadingP = true;
            this.loading = true;
            this.investmentService.intrabankTransfer(payload).subscribe(
                (data) => {
                    this.loading = false;
                    this.investment.paymentMade = true;
                    // this.modalRef.hide();
                    this.done.emit(true);
                },
                (error) => {
                    this.loading = false;
                    // this.toastr.error(`There is a problem transferring funds to your ${this.investment.planName} account`)
                    this.toastr.error(error.error.message);
                }
            );

        }

    }

    /**
     * Renflex specific functionality End
     */


    ngAfterViewInit() {
        if (!this.investment._id) {
            this.loading = true;
            switch (this.investment.planType) {
                case "fixedDeposit":
                    this.createFixedDeposit();
                    break;
                case "targetSavings":
                    this.createTargetSavings();
                    break;
                case 'loan':
                    this.loading = false;
                    break;

                default:
                    break;
            }
        }
    }

    paymentSuccess() {
        this.loading = false;
        this.client.mambuStatus = "OnMambu";
        this.clientService.saveClientLocally(this.client);
        if (this.investment) {
            this.investmentService.deleteCalcValues();
            if (this.investment) {
                this.investment.paymentMade = true;
            }
        }
        this.done.emit(true);
    }

    paymentModalClosed() {
        this.loading = false;
    }

    optionSelected() {
        this.loading = true;
        let paymentOption = this.selectedPaymentOption.type;
        if (paymentOption == "debitCard" || paymentOption == "recurrent") {
            this.investment.paymentOption = null;
            setTimeout(() => {
                this.investment.paymentOption = paymentOption;
            }, 100);
        } else if (paymentOption == "bankTransfer") {
            this.investment.paymentOption = paymentOption;
            this.loading = false;
            this.done.emit(true);
        } else {
            // this.modalRef = this.modalService.show(fullWithdrawalPassword);
            this.transfer();
            // this.loading = false;
            this.loading = true;
        }
    }

    paymentFailed() {
        this.loading = false;
        this.toastr.error("Payment not successful");
    }

    createFixedDeposit() {
        let inv: any = this.investment;
        inv.tenor = inv.tenor * 30;
        delete inv.totalPayOut;
        this.investmentService.createInvestment(inv).subscribe(
            (data: any) => {
                this.loading = false;
                this.investment._id = data.data.encodedKey;

            },
            () => {
                this.loading = true;
                this.toastr.error(
                    "Error while creating investment, kindly go back"
                );
            }
        );
    }

    createTargetSavings() {
        let investment = {
            ...this.investment,
            frequency: this.investment.contributionFrequency.toUpperCase(),
            tenor: this.investment.tenor * 30,
            targetAmount: this.investment.amount * 100,
        };

        this.investmentService.createTargetSavings(investment).subscribe(
            (data: any) => {
                this.loading = false;
                this.investment._id = data.data.id;
                this.investmentService.saveSmartGoalInvestmentLocally(this.investment);
            },
            () => {
                this.loading = true;
                this.toastr.error(
                    "Error while creating investment, kindly go back"
                );
            }
        );
    }

    calculateRemainingBalance(num) {
        num = this.currencyToNumber(this.amountToTransfer);
        if (num <= this.renflexAccount.availableBalance) {
            this.availableBalance = this.renflexAccount.availableBalance - num;
        } else {
            this.availableBalance = 0;
        }
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57)) {
            return false;
        }
        return true;
    }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `₦${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "").replace("₦", "");
    }

    checkIfAmountIsValid() {
        clearTimeout(this.timeOut);
        this.tempAmountToTransfer = this.currencyToNumber(this.amountToTransfer);
        this.amountToTransfer = this.numberToCurrencyFormat(this.tempAmountToTransfer);
        const amount = this.tempAmountToTransfer;
        this.amountToTransferDub = this.tempAmountToTransfer;
        this.toastr.clear();
        this.timeOut = setTimeout(() => {
            if (amount > this.renflexAccount.availableBalance) {
                this.amountToTransfer = this.renflexAccount.availableBalance;
                // this.notValid = false;
                this.toastr.warning(
                    "The maximum amount you can withdraw is " +
                    this.renflexAccount.availableBalance
                );
            } else if (this.tempAmountToTransfer < 100) {
                this.tempAmountToTransfer = 100;
                this.amountToTransfer = this.numberToCurrencyFormat(this.tempAmountToTransfer);
                // this.notValid = false;
                this.availableBalance = this.renflexAccount.availableBalance - 100;
                this.toastr.warning(
                    "The minimum amount you can withdraw is ₦100"
                );
            }
        }, 1650);
    }

    checkingForClosedModal() {
        this.modalService.onHide.subscribe(() => {
            this.modalRef.hide();
        });
    }
}
