import { Component, OnInit } from "@angular/core";
import { Location } from "@angular/common";
import { DialogModalButton, Client } from "../../../@model/client";
import { DialogBoxComponent } from "../../shared/dialog-box/dialog-box.component";
import { InvestmentInterface } from "../../../@model/interfaces";
import { BsModalRef, BsModalService } from "ngx-bootstrap/modal";
import { ClientService } from "../../../@core/client.service";
import { appRoutes } from "src/app/@config/routes";
import { Router, ActivatedRoute } from "@angular/router";

@Component({
    selector: "app-create-fixed-deposit",
    templateUrl: "./create-fixed-deposit.component.html",
    styleUrls: [
        "../create.component.scss",
        "./create-fixed-deposit.component.scss",
    ],
})
export class CreateFixedDepositComponent implements OnInit {
    state = 0;
    investment: InvestmentInterface = {
        planType: "fixedDeposit",
    };
    client: Client;
    modalRef: BsModalRef;
    dialogOption = new DialogModalButton();
    states = {
        planName: 0,
        depositCalculator: 1,
        planReview: 2,
        bvn: 3,
        planFund: 4,
        bankTransfer: 5,
        pep: 6,
    };

    constructor(
        public locationRef: Location,
        public clientService: ClientService,
        private modalService: BsModalService,
        private router: Router,
        private route: ActivatedRoute
    ) { }

    ngOnInit() {
        this.checkForClosedModal();
        this.client = this.clientService.getClientLocally();
        this.checkForStateChange();
        this.goToNextState(0);
    }

    checkForStateChange() {
        this.route.queryParamMap.subscribe(paramData => {
            // increment state value
            this.state = +paramData.get('state');
        })
    }

    namePlanDone(planName: string) {
        this.investment = { ...this.investment, planName };
        this.incrementState();
    }

    depositCompleted(calculatorValues: Object) {
        this.investment = { ...this.investment, ...calculatorValues };
        this.incrementState();
    }

    planApproved() {
        this.modalRef = this.modalService.show(DialogBoxComponent, {
            backdrop: "static",
            initialState: {
                type: "confirm",
                header: "Accept Terms & Conditions",
                body: `I accept that ${this.investment.amount.toLocaleString()} in my ${this.investment.planName
                    } Fixed Deposit plan should be locked for ${this.investment.tenor
                    } months. Untimely withdrawal will attract 30% penalty on the interest earned till date.`,
                confirmText: "Continue",
                cancelText: "Cancel",
                dialogOption: this.dialogOption,
            },
        });
    }

    bvnValidated() {
        this.incrementState();
    }

    paymentDone() {
        let paymentOption = this.investment.paymentOption;

        if (this.investment.paymentMade) {
            this.modalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    id: "payment",
                    header: "Awesome!",
                    body: `You have successfully funded your plan.`,
                    confirmText: "Done",
                    dialogOption: this.dialogOption,
                },
            });
        }

        if (!this.investment.paymentMade) {
            if (paymentOption == "bankTransfer") {
                this.goToNextState(this.states.bankTransfer);
            }
        }
    }

    checkForClosedModal() {
        this.modalService.onHide.subscribe(() => {
            if (
                this.dialogOption.isConfirmed &&
                this.dialogOption.id === "payment"
            ) {
                debugger;
                this.afterPayment();
            } else if (this.dialogOption.isConfirmed) {
                this.clientService.getClientLocally().bvnValidated
                    ? (this.goToNextState(this.state + 2))
                    : this.incrementState();
            }
            this.dialogOption.isConfirmed = false;
        });
    }

    afterPayment() {
        if (!this.client.pepvalidated) {
            return (
                this.goToNextState(this.states.pep)
            );
        }

        return this.router.navigate([appRoutes.dashboard], {
            queryParams: { reload: true },
        });
    }

    goToNextState(state) {
        this.router.navigate([appRoutes.createFixedDeposit], { queryParams: { state: state } });
    }

    pepDone() {
        this.router.navigate([appRoutes.dashboard]);
    }

    private incrementState() {
        this.goToNextState(this.state + 1);
    }

    decrementState() {
        if (this.state > 0) {
            this.goToNextState(this.state - 1);
        }
        else this.router.navigate([appRoutes.viewFixedDeposits])
    }

    initiatePaymentOption(data) {
        if (data.selectedOption == "bankTransfer") {
            this.goToNextState(this.states.pep);
        }
    }

    switchPayment() {
        this.goToNextState(this.states.planFund);

    }
}

<app-sub-page pageTitle="Fixed Deposit" [tabType]="state < 3  ? 'backButton' : ''"
    (backButtonAction)="decrementState()">
    <ng-container>

        <app-plan-name [hidden]="state!== states.planName"
            copy='Give this plan a creative name. e.g. Rainy Day, Dangote 20xx' (done)='namePlanDone($event)'>
        </app-plan-name>

        <app-deposit-calculator *ngIf="state===states.depositCalculator " (done)='depositCompleted($event)'
            copy='Tell us how much you will like to deposit and for how long?'></app-deposit-calculator>

        <app-plan-review [hidden]="state!== states.planReview" (done)='planApproved()' [investment]='investment'>
        </app-plan-review>

        <app-input-bvn [hidden]="state!== states.bvn" (done)='bvnValidated()'></app-input-bvn>

        <app-plan-fund [investment]="investment" (done)='paymentDone()' [isItSmartGoal]="false"
            *ngIf="state=== states.planFund" [availableOptions]="['debitCard', 'renflex']" [showNotification]="false">
        </app-plan-fund>

        <app-bank-transfer *ngIf="state=== states.bankTransfer" (back)="switchPayment()" (done)='paymentDone()'
            [investment]="investment">
        </app-bank-transfer>

        <app-pep-component (done)='pepDone()' [hidden]='state !== states.pep'></app-pep-component>

    </ng-container>

</app-sub-page>
