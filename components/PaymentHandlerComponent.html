import {
    Component,
    OnInit,
    EventEmitter,
    Input,
    Output,
    ChangeDetectionStrategy,
} from "@angular/core";
import { PaymentService } from "src/app/@core/payment.service";
import { GeneralService } from "src/app/@core/general.service";
import { ToastrService } from "ngx-toastr";
import { Client } from "src/app/@model/client";
import { environment } from "src/environments/environment";
import { RaveOptions } from "angular-rave";
import { PaystackOptions } from 'angular4-paystack';

declare let PaystackPop: any;

@Component({
    selector: "app-payment-handler",
    templateUrl: "./payment-handler.component.html",
    styleUrls: ["./payment-handler.component.scss"],
    changeDetection: ChangeDetectionStrategy.OnPush,
})
export class PaymentHandlerComponent implements OnInit {
    @Input() client: Client;
    @Input() account;
    @Input() amount: number;
    @Input() paymentType: string = "Investment";
    @Input() billingType: string = "debitCard";

    @Output() success: EventEmitter<boolean> = new EventEmitter<boolean>();
    @Output() failed: EventEmitter<boolean> = new EventEmitter<boolean>();
    @Output() closed: EventEmitter<boolean> = new EventEmitter<boolean>();

    public ravePaymentOptions: RaveOptions = {
        PBFPubKey: environment.payment.rave,
        customer_email: "",
        customer_firstname: "",
        customer_lastname: "",
        currency: "NGN",
        custom_description: "Payment for ",
        amount: 0,
        txref: "tran-ren-",
        meta: [
            {
                metaname: "mambuId",
                metavalue: "0",
            },
        ],
    };

    paystackPaymentOption = {
        email: "",
        amount: 0,
        ref: "tran-ren-",
    };
    isUserFirstPayment: boolean;
    paystackPublicKey = environment.payment.paystack;
    currency: any;
    transRef = "";

    options: PaystackOptions

    constructor(
        private paymentService: PaymentService,
        private toastr: ToastrService
    ) { }

    ngOnInit() {
        this.populatePaymentInfo();

    }

    ngOnChanges() {
        this.checkIfCardIsAdded();
    }

    populatePaymentInfo() {
        if (this.ravePaymentOptions.customer_email && this.account._id) {
            this.checkIfCardIsAdded();
            return;
        }
        this.ravePaymentOptions.customer_email = this.client.email;
        this.paystackPaymentOption.email = this.client.email;
        this.ravePaymentOptions.customer_firstname = this.client.firstName;
        this.ravePaymentOptions.customer_lastname = this.client.lastName;
        this.paystackPaymentOption.amount = this.ravePaymentOptions.amount = this.amount;
        this.ravePaymentOptions.meta[0].metavalue = this.account._id;
        this.transRef = this.getTransactionRef();
        this.ravePaymentOptions.txref = this.transRef;

        // this.checkIfCardIsAdded();
    }

    checkIfCardIsAdded() {
        if (this.ravePaymentOptions.customer_email && this.account._id) {
            this.isUserFirstPayment = !this.client.cardDetails;
            this.pay();
        } else {
            this.populatePaymentInfo();
        }
    }

    pay() {
        if (this.account && this.ravePaymentOptions.PBFPubKey !== "") {
            if (this.account._id) {
                if (this.billingType === "debitCard") {
                    this.showPaymentPopUp();
                } else {
                    //TODO: change back to make payment instantly a soon as the endpoint is working.
                    // this.showPaymentPopUp();
                    this.makePaymentInstantly();
                }
            }
        }
    }

    showPaymentPopUp() {
        document
            .getElementById(`payWith${environment.selectedPaymentGateway}`)
            .click();
    }

    paymentCallback(event) {
        if (event.status || event.success) {
            this.tokenizeCard({
                transactionReference: this.transRef,
                cardProcessor: environment.selectedPaymentGateway,
                loanOrInvestmentId: this.account._id,
                paymentType: this.paymentType,
            });
        } else {
            this.failed.emit(true);
        }
    }

    makePaymentInstantly() {
        const payment = {
            loanOrInvestmentId: "",
            amount: 0,
            paymentType: this.paymentType,
            cardProcessor: environment.selectedPaymentGateway,
        };
        payment.loanOrInvestmentId = this.account._id;
        (payment.amount = this.amount * 100), //amount in kobo;
            this.paymentService.makePayment(payment).subscribe(
                () => {
                    this.success.emit(true);
                },
                (error) => {
                    this.failed.emit(true);

                }
            );
    }

    tokenizeCard(cardTokenPayload) {
        if (cardTokenPayload.loanOrInvestmentId === "tokenizeCard") {
            delete cardTokenPayload.loanOrInvestmentId;
        }
        this.paymentService.saveCardToken(cardTokenPayload).subscribe(
            (data) => {
                PaystackPop = {};
                this.success.emit(true);
            },
            (error) => {
                PaystackPop = {};
                this.failed.emit(true);
            }
        );
    }

    getTransactionRef() {
        const ref = `${new Date().getTime().toString(36)}-ref-${this.account._id
            }-${new Date().toLocaleDateString().replace(/\//g, "-") +
            "-" +
            new Date().toTimeString().split(" ")[0].replace(/\:/g, "-")
            }`;
        return ref;
    }

    payWithPaystack() {
        let key;
        if (this.paymentType.toLowerCase() != 'loan') {
            key = environment.payment.paystack

        } else {
            key = environment.payment.loanPaystack
        }
        const handler = PaystackPop.setup({
            key,
            email: this.client.email,
            amount: this.amount * 100, //amount in kobo
            currency: "NGN",
            ref: this.transRef,
            channels: ["card"],
            metadata: {
                custom_fields: [
                    {
                        paymentType: this.paymentType,
                        clientId: this.client.clientEncodedkey,
                        loanOrInvestmentId: this.account._id,
                    },
                ],
            },
            callback: (response) => {
                this.paymentCallback(response);
            },
            onClose: () => {
                this.closed.emit(true);
            },
        });
        handler.openIframe();
    }
