import { Component, OnInit, EventEmitter, Output, Input } from "@angular/core";
import { NgForm } from "@angular/forms";

@Component({
    selector: "app-plan-name",
    templateUrl: "./plan-name.component.html",
    styleUrls: ["./plan-name.component.scss"],
})
export class PlanNameComponent implements OnInit {
    planName: string;
    @Input() copy: string;
    @Output() done: EventEmitter<string> = new EventEmitter<string>();
    constructor() {}

    ngOnInit(): void {}

    createPlanName(form: NgForm) {
        const plan = form.value.planName;
        this.done.emit(plan);
    }
}
