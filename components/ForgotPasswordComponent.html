import { Component, OnInit } from '@angular/core';
import { Client, ResetModel } from 'src/app/@model/client';
import { appRoutes } from 'src/app/@config/routes';
import { AuthService } from 'src/app/@core/auth.service';
import { Router } from '@angular/router';
import { ToastrService } from 'ngx-toastr';
import { Base64 } from 'js-base64';


@Component({
    selector: 'app-forgot-password',
    templateUrl: './forgot-password.component.html',
    styleUrls: ["../auth/auth-mobile.scss", './forgot-password.component.scss']
})
export class ForgotPasswordComponent implements OnInit {

    client = new ResetModel();
    routes = appRoutes;

    constructor(private authService: AuthService,
        private router: Router,
        private toastr: ToastrService) { }

    ngOnInit() {
    }

    passwordResetInitialize(client: ResetModel) {
        this.authService.changeState(true);
        this.authService.initializeResetPassword(client)
            .subscribe(data => {
                this.authService.changeState(false);
                this.router.navigate([this.routes.otpVerification, Base64.encode(client.emailOrPhone), 'password']);
                this.toastr.success('You will receive a code (one-time-password) from us shortly');
            },
                error => {
                    this.authService.changeState(false);
                    this.toastr.error(error.error.message || "Couldn't verify account, try again");
                })
    }

}
