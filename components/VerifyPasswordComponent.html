import { Component, OnInit, Output } from "@angular/core";
import { NgForm } from "@angular/forms";
import { BsModalRef } from "ngx-bootstrap/modal";
import { ClientService } from "src/app/@core/client.service";
import { ToastrService } from "ngx-toastr";
import { DialogModalButton } from "src/app/@model/client";

@Component({
    selector: "app-verify-password",
    templateUrl: "./verify-password.component.html",
    styleUrls: ["./verify-password.component.scss"],
})
export class VerifyPasswordComponent implements OnInit {
    id: string = "verifyPassword";
    loading: boolean = false;
    password: string;
    shouldShowPassword = true;
    passwordState = "Show";
    dialogOption = new DialogModalButton();

    constructor(
        public clientService: ClientService,
        public modalRef: BsModalRef,
        private toastr: ToastrService
    ) {}

    ngOnInit(): void {}

    verifyUserPassword(form: NgForm) {
        this.loading = true;

        this.clientService.validatePassword(form.value.password).subscribe(
            (data) => {
                this.loading = false;
                form.resetForm();
                this.dialogOption = Object.assign(this.dialogOption, {
                    isConfirmed: true,
                    id: this.id,
                });
                this.modalRef.hide();
            },
            (error) => {
                this.dialogOption = Object.assign(this.dialogOption, {
                    isConfirmed: false,
                    id: this.id,
                });
                this.loading = false;
                this.toastr.error(error.error.message);
            }
        );
    }

    showPassword(event) {
        if (!this.shouldShowPassword) {
            this.passwordState = "Show";
            event.target.parentElement.children[1].type = "password";
            this.shouldShowPassword = !this.shouldShowPassword;
            return;
        }
        this.passwordState = "Hide";
        event.target.parentElement.children[1].type = "text";
        this.shouldShowPassword = !this.shouldShowPassword;
    }
}
