import { ClientService } from "../../../@core/client.service";
import { Component, OnInit } from "@angular/core";
import { Client } from "../../../@model/client";
import { DomSanitizer } from "@angular/platform-browser";
import { ToastrService } from "ngx-toastr";
import { GeneralService } from "src/app/@core/general.service";

@Component({
    selector: "app-account-profile",
    templateUrl: "./profile.component.html",
    styleUrls: ["./profile.component.scss"],
})
export class ProfileComponent implements OnInit {
    selectedPhoto: any;
    client = new Client();
    loading: boolean = false;
    profilePhoto;
    imageUrl: any;

    constructor(
        private clientService: ClientService,
        public sanitizer: DomSanitizer,
        public toastr: ToastrService,
        public generalService: GeneralService
    ) { }

    ngOnInit(): void {
        this.generalService.showLoader(true);
        this.client = this.clientService.getClientLocally();
        this.getProfilePhoto();
    }

    getProfilePhoto() {
        this.clientService.getProfilePicture().subscribe(
            (response) => {
                if (response.message && response.message.split('"')[1]) {
                    this.profilePhoto = this.sanitizer.bypassSecurityTrustResourceUrl(
                        response.message.split('"')[1]
                    );
                }
                this.selectedPhoto = null;
                this.generalService.showLoader(false);
            },
            (error) => {
                this.generalService.showLoader(false);
            }
        );
    }

    selectPhoto(event) {
        let file = event.target.files[0];
        if (file) {
            let acceptedTypes = ["image/jpeg", "image/png"];

            if (!acceptedTypes.includes(file.type)) {
                this.selectedPhoto = null;
                return this.toastr.error(
                    "Error, only jpeg or png images are allowed please."
                );
            }

            if (file.size > 2000000) {
                this.selectedPhoto = null;
                return this.toastr.error(
                    "Error, the maximum file size you can upload is 2MB."
                );
            }

            this.selectedPhoto = file;

            this.imageUrl = this.sanitizer.bypassSecurityTrustResourceUrl(
                URL.createObjectURL(this.selectedPhoto)
            );

            this.uploadImage();
        }
    }

    uploadImage() {
        this.generalService.showLoader(true);
        let reader = new FileReader();
        reader.readAsDataURL(this.selectedPhoto);
        reader.onload = () => {
            let data = {
                documentType: "photo",
                type: "client",
                documentMime: this.selectedPhoto.type.split("/")[1],
                encodedKey: this.client.clientEncodedkey,
                profilePicture: "yes",
                document: reader.result.toString().split("base64,")[1],
            };

            this.clientService.uploadProfilePicture(data).subscribe(
                (response) => {
                    this.generalService.showLoader(false);
                    this.toastr.success(
                        "Great, profile image uploaded successfully."
                    );
                },
                (error) => {
                    this.generalService.showLoader(false);
                    this.toastr.error(
                        error.error.message ||
                        "An error occurred, please try again later."
                    );
                }
            );
        };

        reader.onerror = (error) => {
            this.generalService.showLoader(false);
            this.toastr.error("An error occurred, please try again.");
        };
    }
}
