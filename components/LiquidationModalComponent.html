import { Component, OnInit } from '@angular/core';
import { BsModalRef } from 'ngx-bootstrap/modal';
import { Client } from 'src/app/@model/client';
import { ClientService } from 'src/app/@core/client.service';
import { ToastrService } from 'ngx-toastr';
import { GeneralService } from 'src/app/@core/general.service';
import { InvestmentService } from 'src/app/@core/investment.service';
import { environment } from 'src/environments/environment';

@Component({
    selector: 'app-liquidation-modal',
    templateUrl: './liquidation-modal.component.html',
    styleUrls: ['../../single-view/complete/complete.component.scss', './liquidation-modal.component.scss']
})
export class LiquidationModalComponent implements OnInit {
    loading: boolean = false;
    client: Client;
    accountDetails = { accountName: "", accountNumber: "", bankCode: "", bank: "" };
    banks: any[];
    statusMessage: string;
    timeout: any;
    selectedBankCode: string;
    accountNumber: string;
    modalRef: BsModalRef;
    bank: any;



    constructor(
        public bsModalRef: BsModalRef,
        private clientService: ClientService,
        private toastr: ToastrService,
        private generalService: GeneralService,
        private investmentService: InvestmentService,) { }

    ngOnInit(): void {
        this.getBanks();
        this.client = this.clientService.getClientLocallyForAccountSetup();
        this.getMambuDetails();
        // this.getAccountName();
    }

    getBanks() {
        this.banks = this.investmentService.getBanks();
    }

    getBank(code) {
        this.bank = this.banks.find(x => x.Code === code).Name;
        this.accountDetails.bank = this.bank;
    }


    update() {
        this.loading = true;
        // console.log(this.accountDetails);
        // const client: any = {};
        // client.firstName = this.client.firstName;
        // client.lastName = this.client.lastName;
        // client.middleName = this.client.middleName;
        // client.dob = this.client.dateOfBirth;
        // client.address = this.client.address;
        // client.gender = this.client.sex.toUpperCase();

        // client.bvn = this.client.bvn;
        // client.id = client.bvn;
        // client.documentType = "BVN";

        if (this.accountDetails.accountName.toLowerCase().split(" ").includes(this.client.firstName.toLowerCase()) || this.accountDetails.accountName.toLowerCase().split(" ").includes(this.client.lastName.toLowerCase())) {
            this.updateBankData(this.banks.find((x) => x.Code === this.accountDetails.bankCode).Name, this.accountDetails.accountNumber);
        } else {
            this.loading = false;
            this.statusMessage =
                "Name doesn't match any of your provided names";
            this.toastr.error("Name doesn't match any of your provided names");
        }
    }

    getAccountName(bankCode) {
        clearTimeout(this.timeout);
        delete this.statusMessage;
        this.timeout = setTimeout(() => {
            if (
                this.accountDetails.accountNumber.length === 10 &&
                this.accountDetails.bank
            ) {
                this.loading = true;
                this.investmentService
                    .getAccountName(
                        bankCode,
                        this.accountDetails.accountNumber
                    )
                    .subscribe(
                        (data: any) => {
                            if (data.message === "Invalid Account Number") {
                                this.statusMessage =
                                    "Invalid bank and account number combination";
                                this.loading = false;
                                return;
                            }
                            this.accountDetails.accountName = data.data.account_name;
                            this.loading = false;
                        },
                        (error) => {
                            this.loading = false;
                            this.statusMessage = "Kindly try again";
                            this.toastr.error(
                                "Error occured during account name enquiry"
                            );
                        }
                    );
            }
        }, 1650);
    }

    updateBankData(bankName, accountNumber) {
        this.client.id = this.client.bvn;
        this.client.bank = bankName;
        this.client.accountNumber = accountNumber;
        this.clientService.saveClientLocally(this.client);
        this.clientService.updateClient(this.client).subscribe(
            (data) => {
                this.loading = false;
                this.clientService.changeLiquidationBankDetailsState(true);
                this.clientService.removeClientLocallyForAccountSetup();
                this.bsModalRef.hide();
                // this.getUserData();
                // this.getMambuDetails();
            },
            (error) => {
                // if (error.error.code === environment.statusCodes.bvnTaken) {
                //     this.clientService.updateClient(this.client).subscribe(data => {
                //         this.clientService.changeLiquidationBankDetailsState(true);
                //         this.clientService.removeClientLocallyForAccountSetup();
                //         this.loading = false;
                //         this.bsModalRef.hide();
                //     }, error => {
                //         this.loading = false;
                //         this.toastr.error(error.error.message);
                //     })
                // }
                this.loading = false;
                this.toastr.error(error.error.message);
            }
        );
    }


    getMambuDetails() {
        this.generalService.showLoader(true);
        this.clientService.getClientFromMambu().subscribe(
            (data) => {
                if (data.data) {
                    if (data.data.bankLiquidationDetails) {
                        this.bank = data.data.bankLiquidationDetails.nameOfBank;
                        this.accountDetails.bank = this.bank;
                        this.accountNumber = data.data.bankLiquidationDetails.accountNumber;
                        this.accountDetails.accountNumber = this.accountNumber;
                        this.getAccountName(this.banks.find(
                            (x) => x.Name.toLowerCase() === this.bank.toLowerCase()
                        ).Code)
                    }
                }
                this.generalService.showLoader(false);
            },
            (error) => {
                this.toastr.error(
                    "Oops",
                    error.error.message || "Error occured while adding to bank"
                );
                this.generalService.showLoader(false);
            }
        );
    }

    getUserData() {
        this.clientService.getClient().subscribe(
            (data) => {
                this.loading = false;
                data.data["gender"] = data.data.sex;
                this.clientService.saveClientLocally(data.data);
            },
            (error) => {
                this.loading = false;
                this.toastr.error(
                    "Something went wrong while getting the client"
                );
            }
        );
    }


}
