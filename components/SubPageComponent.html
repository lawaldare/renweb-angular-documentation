import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from "@angular/core";
import { LoanService } from 'src/app/@core/loan.service';

@Component({
    selector: "app-sub-page",
    templateUrl: "./sub-page.component.html",
    styleUrls: ["./sub-page.component.scss"],
    changeDetection: ChangeDetectionStrategy.OnPush,
})
export class SubPageComponent {
    @Input('pageTitle') title; //the title of the header component
    @Input() tabType; //tabs, backButton
    @Input() isCalculatorPage = false;
    @Input() tabs: Tabs; //an array of tabs to be displayed
    @Output() backButtonAction = new EventEmitter(); //a function or action that should happen when the back button is clicked
    stepNumber: number;


    constructor(public loanService: LoanService) {
    }
}

interface Tabs {
    title: string;
    link: string;
}

<div class="inner-page {{isCalculatorPage ? 'p-0 pt-3 mw-100':'mx-auto' }} ">
    <section id="header">
        <div class="d-flex justify-content-between">
            <h4 class="text-blue-primary weight-bolder">
                {{ title }}
            </h4>
            <!-- <p>Step {{ loanService.stepNumber$ | async }} of 6</p> -->
            <ng-content select='.header-option'></ng-content>
        </div>

        <hr>

        <ng-container [ngSwitch]="tabType">
            <div *ngSwitchCase="'tabs'" class="d-flex border-bottom pointer flex-md-row flex-column">
                <a class="account px-md-4 py-md-0 py-3 pb-md-2 pb-0 tab pointer" [routerLink]="tab.link"
                    routerLinkActive='active' *ngFor="let tab of tabs;">
                    {{ tab.title }}
                </a>
            </div>
            <div *ngSwitchCase="'backButton'" (click)='backButtonAction.emit()'
                class="text-grey-light pointer pl-3 back-button">
                <i class="fas fa-chevron-left mr-2"></i> Back
            </div>
        </ng-container>

    </section>

    <ng-content></ng-content>

</div>
