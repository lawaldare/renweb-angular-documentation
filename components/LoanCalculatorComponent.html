    ngOnInit(): void {
        this.config = this.dashboardService.getConfigLocally();

        this.getLimits();

            // this.tenors = this.getTenors(limits.minTenure, limits.maxTenure);
            if (+this.salary >= 200000 && this.state === "Lagos") {
                this.options.stepsArray = [{ value: 3 }, { value: 4 }, { value: 5 }, { value: 6 }, { value: 7 }, { value: 8 }, { value: 9 }, { value: 10 }, { value: 11 }, { value: 12 }, { value: 18 }, { value: 24 }];

            }

            this.tenorUnit = 'Months';
            this.selectedTenure = this.value;
            this.outcome = this.employedCalculator(this.salary, this.selectedTenure, 'rbRates', 'new');
        } else {
            const limits = this.getLimits();
            this.options.stepsArray = this.getTenors(limits.minTenure, limits.maxTenure);
            this.value = this.options.stepsArray[this.options.stepsArray.length - 1].value;
            this.tenorUnit = 'Weeks';
            this.selectedTenure = this.value;
            this.outcome = this.selfEmployedCalculator(this.salary, this.selectedTenure, 'rbRates', 'new');
        }

        this.setMaximumLoanableAmount();

        if (this.employmentType === 'Full Time (Private)') {
            if (this.value > 12) {

                // this.loanDetail.purpose = 'Housing';

        this.selectedLoanPurpose = purpose;

    onTenureSelected(event) {
        this.selectedTenure = event;
        this.displayNotice = true;
        if (this.employmentType === 'Full Time (Private)') {
            if (this.selectedTenure > 12) {
                // this.isTenureLessThen12Months = false;

                this.loanPurposeArray = this.config.loanPurpose;

            }
        }
        this.calculateTotalRepayment();
    }

    getLimits(): { maxTenure, minTenure } {

        const rates = this.config.loanCalculatorLogic.rates[this.employmentType === 'Full Time (Private)' ? 'employed' : 'selfEmployed'].sort((a, b) => a.min - b.min);
        if (this.employmentType === 'Full Time (Private)') {
            this.maxAmount = this.config.plMaximumAmount;
            this.minAmount = this.config.plMinimumAmount;
        } else {
            this.maxAmount = this.config.mblMaximumAmount;
            this.minAmount = this.config.mblMinimumAmount;
        }
        const sortedRates = rates[0].new['rbRates'].filter(x => x.rate > 0).sort((a, b) => a.min - b.min);
        const minTenure = sortedRates[0].min;
        const maxTenure = sortedRates[sortedRates.length - 1].max;
        // const minAmount = rates[0].min;
        return {
            maxTenure, minTenure
        }
    }

    getTenors(minValue: number, maxValue: number) {
        let returnedArray = [];
        for (let i = minValue; i <= maxValue; i++) {
            let obj: any = {};
            obj.value = i
            returnedArray.push(obj);
        }
        return returnedArray;
    }


    setMaximumLoanableAmount() {

        this.setOutcome();
        if (this.loanDetail.amount) {
            if (this.loanDetail.amount < this.minAmount) {
                this.amount = this.minAmount;
            } else if (this.loanDetail.amount > this.outcome.loan_amount) {
                this.amount = this.outcome.loan_amount;
            } else {
                this.amount = this.loanDetail.amount
            }
        } else {
            this.amount = this.minAmount < this.outcome.loan_amount ? this.minAmount : this.outcome.loan_amount;
        }
        this.amountCurrency = this.generalService.numberToCurrencyFormat(this.outcome.loan_amount);

        if (this.amount < this.minAmount) {
            this.showSlider = false;
            return;

        this.setScrolledPercentage();
        if (this.employmentType === 'Full Time (Private)') {
            this.getEmployedRepayment();
        } else {
            this.getSelfEmployedRepayment();
        }

        // this.getInterestRates(this.amount);
    }

        } this.outcome = this.selfEmployedCalculator(this.salary, this.selectedTenure, 'rbRates', 'new');

        // this.amountCurrency = this.generalService.numberToCurrencyFormat(this.outcome.loan_amount);

        this.amount = +this.generalService.currencyToNumber(this.amountCurrency);

    checkIfAmountIsValid() {
        clearTimeout(this.timeOut);
        this.amount = +this.generalService.currencyToNumber(this.amountCurrency);
        this.amountCurrency = this.generalService.numberToCurrencyFormat(this.amount);
        const amount = this.amount;

                this.amount = this.minAmount;

                this.toastr.warning(
                    "The minimum loan amount is " +

            if (amount > this.outcome.loan_amount) {
                this.amount = this.outcome.loan_amount;

                // this.toastr.warning(
                //     "The maximum amount you can invest is " +
                //     this.outcome.loan_amount
                // );
            }
            this.updateTotalRepayment();
        }, 700);
    }

    setScrolledPercentage() {
        // this.scrolledPercentage = ((this.amount - this.minAmount) * 100) / (this.outcome.loan_amount - this.minAmount);
    }

    calculateTotalRepayment() {
        this.setOutcome();
        if (this.employmentType === 'Full Time (Private)') {
            this.getEmployedRepayment();
        } else {

        }

        if (this.employmentType === 'Full Time (Private)') {
            this.getEmployedRepayment();
        } else {
            this.getSelfEmployedRepayment();
        }
    }


    getBand(amount, tenor, type) {
        let band = this.config.loanCalculatorLogic.bands[type].find((band) => {
            if (tenor >= band.min && tenor <= band.max) {
                return band;
            }
        });

        if (band && band.rates) {
            return band.rates.find((rate) => {
                if (amount >= rate.min && amount <= rate.max) {
                    return rate;
                }
            });
        }
    }

    getRate(amount, tenor, type, calculator_rate_type, loan_type) {
        let band = this.config.loanCalculatorLogic.rates[type].find((band) => {
            if (amount >= band.min && amount <= band.max) {
                return band;
            }
        });

        if (band && band[loan_type] && band[loan_type][calculator_rate_type]) {
            return band[loan_type][calculator_rate_type].find((rate) => {
                if (tenor >= rate.min && tenor <= rate.max) {
                    return rate;
                }
            });
        }
    }

    pv(salary, rate, tenor) {
        let amount = 0;

        for (var i = 1; i <= tenor; i++) {
            amount += salary / Math.pow(1 + rate / 100, i);
        }

        return amount;
    }

    PMT(rate, nperiod, pv, fv = 0, type = 0) {
        if (rate == 0) return -(pv + fv) / nperiod;

        var pvif = Math.pow(1 + rate, nperiod);
        var pmt = rate / (pvif - 1) * -(pv * pvif + fv);

        if (type == 1) {
            pmt /= (1 + rate);
        };

        pmt = pmt * -1;
        return pmt.toFixed(2);
    }

    employedCalculator(
        monthly_income,
        tenor,
        calculator_rate_type,
        loan_type
    ) {
        const band = this.getBand(monthly_income, tenor, "employed");
        const monthly_repayment =
            monthly_income * (((band && band.rate) || 0) / 100) || 0;
        const total_repayment = monthly_repayment * tenor || 0;
        const rateData = this.getRate(
            total_repayment,
            tenor,
            "employed",
            calculator_rate_type,
            loan_type
        );
        let loan_amount =
            +(this.pv(monthly_repayment, (rateData && rateData.rate) || 0, tenor) || 0).toFixed();

        loan_amount = loan_amount > this.maxAmount ? this.maxAmount : loan_amount;
        return {
            loan_amount,
            total_repayment,
            monthly_repayment,

        };
    }


    selfEmployedCalculator(
        monthly_income,
        tenor,
        calculator_rate_type,
        loan_type
    ) {
        tenor = tenor - 3;
        const band = this.getBand(monthly_income, tenor, "selfEmployed");
        const monthly_repayment =
            (monthly_income * (band && (band.rate || 0) / 100) || 0) / 4;
        const total_repayment = monthly_repayment * tenor || 0;

        const rateData = this.getRate(
            total_repayment,
            tenor,
            "selfEmployed",
            calculator_rate_type,
            loan_type
        );
        const weeklyRate = (((rateData && rateData.rate) || 0) / 30) * 7;
        let loan_amount = +(this.pv(monthly_repayment, weeklyRate || 0, tenor) || 0).toFixed();
        loan_amount = loan_amount > this.maxAmount ? this.maxAmount : loan_amount;
        return {
            loan_amount,
            total_repayment,
            monthly_repayment,
            rate: weeklyRate
        };
    }

    saveCalcValues(amount, tenure = this.selectedTenure, purpose) {
        if (amount < this.minAmount) {
            this.bsModalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Notice",
                    body: `Hey, to get the loan amount less than ${this.generalService.numberToCurrencyFormat(this.minAmount)}, kindly apply through our mobile app`,
                    confirmText: "Yes",
                    cancelText: "No",
                },
            });
        } else if (amount > this.outcome.loan_amount) {
            this.bsModalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Notice",
                    body: `The maximum loan amount is ${this.generalService.numberToCurrencyFormat(this.outcome.loan_amount)}`,
                    confirmText: "Okay",
                },
            });
        } else if (this.selectedTenure > 12 && amount < 500000) {
            this.bsModalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Notice",
                    body: `The minimum loan amount for tenures above 12 months is 500,000`,
                    confirmText: "Okay",
                },
            });
        } else {
            this.done.emit({ amount, tenure, purpose });
        }
    }


    getSelfEmployedRepayment() {
        const rate = +this.config.loanCalculatorLogic.rates['selfEmployed'].find((band) => {

            this.amount = +this.generalService.currencyToNumber(this.amountCurrency);
            if (this.amount >= band.min && this.amount <= band.max) {
                return band;
            }
        })['new']['rbRates'].find(rateBand => rateBand.min <= this.selectedTenure && rateBand.max >= this.selectedTenure).rate;

        const weeklyRate = (rate / 100) * 7 / 30;
        const repaymentsCount = this.selectedTenure - 3;
        const totalAmountRepaid = +this.PMT(weeklyRate, repaymentsCount, this.amount, 0) * repaymentsCount;
        this.frequentRepayment = totalAmountRepaid / repaymentsCount;

    }

    getEmployedRepayment() {
        const rate = +this.config.loanCalculatorLogic.rates['employed'].find((band) => {
            this.amount = +this.generalService.currencyToNumber(this.amountCurrency);

            if (this.amount >= band.min && this.amount <= band.max) {
                return band;
            }
        })['new']['rbRates'].find(rateBand => rateBand.min <= this.selectedTenure && rateBand.max >= this.selectedTenure).rate;


        const totalAmountRepaid = +this.PMT(rate / 100, this.selectedTenure, this.amount, 0) * this.selectedTenure;
        this.frequentRepayment = totalAmountRepaid / this.selectedTenure;

    }







}
