import { Component, OnInit, AfterViewInit, Output, EventEmitter, Input } from '@angular/core';
import { Client } from 'src/app/@model/client';

@Component({
    selector: 'app-calender',
    templateUrl: './calender.component.html',
    styleUrls: ['./calender.component.scss']
})
export class CalenderComponent implements OnInit {

    @Output() done: EventEmitter<any> = new EventEmitter<any>();
    @Input() client: Client;


    daysInNumber: number[];
    yearsInNumber: number[];
    monthsInNumber = [
        { name: 'Jan', value: '01' },
        { name: 'Feb', value: '02' },
        { name: 'Mar', value: '03' },
        { name: 'Apr', value: '04' },
        { name: 'May', value: '05' },
        { name: 'Jun', value: '06' },
        { name: 'Jul', value: '07' },
        { name: 'Aug', value: '08' },
        { name: 'Sep', value: '09' },
        { name: 'Oct', value: '10' },
        { name: 'Nov', value: '11' },
        { name: 'Dec', value: '12' },
    ];

    startYear = 1960;
    endYear = new Date().getFullYear();
    daySelected = '';
    monthSelected = '';
    yearSelected = '';

    constructor() { }

    ngOnInit(): void {
        this.generateDays();
        this.generateYears(this.startYear, this.endYear);
        if (this.client && this.client.dob) {
            this.splitDate(this.client.dob);
        }
    }

    generateDays() {
        let output = [];
        for (let i = 1; i < 32; i++) {
            output.push(i);
        }
        this.daysInNumber = output;
    }

    generateYears(start, end) {
        let output = [];
        for (let i = start; i < end + 1; i++) {
            output.push(i);
        }
        this.yearsInNumber = output;

    }

    splitDate(date: string) {
        let dateSplitted = date.split('-');
        this.daySelected = dateSplitted[2];
        this.monthSelected = dateSplitted[1];
        this.yearSelected = dateSplitted[0];
    }

    changeDay(event) {
        this.monthSelected = "";
        this.daySelected = event;
        if (event <= 29) {
            this.monthsInNumber = [{ name: 'Jan', value: '01' }, { name: 'Feb', value: '02' }, { name: 'Mar', value: '03' }, { name: 'Apr', value: '04' }, { name: 'May', value: '05' }, { name: 'Jun', value: '06' }, { name: 'Jul', value: '07' }, { name: 'Aug', value: '08' }, { name: 'Sep', value: '09' }, { name: 'Oct', value: '10' }, { name: 'Nov', value: '11' }, { name: 'Dec', value: '12' },
            ];
        } else if (event > 29 && event <= 30) {
            this.monthsInNumber = [{ name: 'Jan', value: '01' }, { name: 'Mar', value: '03' }, { name: 'Apr', value: '04' }, { name: 'May', value: '05' }, { name: 'Jun', value: '06' }, { name: 'Jul', value: '07' }, { name: 'Aug', value: '08' }, { name: 'Sep', value: '09' }, { name: 'Oct', value: '10' }, { name: 'Nov', value: '11' }, { name: 'Dec', value: '12' },
            ];
        } else if (event > 30) {
            this.monthsInNumber = [{ name: 'Jan', value: '01' }, { name: 'Mar', value: '03' }, { name: 'May', value: '05' }, { name: 'Jul', value: '07' }, { name: 'Aug', value: '08' }, { name: 'Oct', value: '10' }, { name: 'Dec', value: '12' },
            ];
        }
        this.generateFullISODate(this.daySelected, this.monthSelected, this.yearSelected);
    }

    changeMonth(event) {
        this.monthSelected = event;
        this.generateFullISODate(this.daySelected, this.monthSelected, this.yearSelected);
    }

    changeYear(event) {
        this.yearSelected = event;
        this.generateFullISODate(this.daySelected, this.monthSelected, this.yearSelected);
    }

    generateFullISODate(day, month, year) {
        let dateString = `${year}-${month}-${day}`;
        let dateOutput = new Date(dateString);
        if (this.daySelected && this.monthSelected && this.yearSelected) {
            this.done.emit(dateOutput)
        } else {
            this.done.emit("");
        }
    }

}

<div class="container row">

    <div class="form-group col-sm-4 pl-0 date-row-left">
