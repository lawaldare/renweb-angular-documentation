import { Component, OnInit, Input, Output, EventEmitter, ChangeDetectionStrategy } from "@angular/core";
import { ToastrService } from "ngx-toastr";
import { Client } from "src/app/@model/client";
import { DashboardService } from "src/app/@core/dashboard.service";
import { ClientService } from "src/app/@core/client.service";
import { Router } from "@angular/router";

@Component({
    selector: "app-pep-component",
    templateUrl: "./pep.component.html",
    styleUrls: ["./pep.component.scss"],
    changeDetection: ChangeDetectionStrategy.OnPush,

})
export class PepComponent implements OnInit {
    loading = false;
    client: Client;
    @Output() done: EventEmitter<boolean> = new EventEmitter<boolean>();


    pepQuestions = [
        {
            question: "Are you currently holding a Public Office",
            answers: ["YES", "NO"],
            title: "publicOfficeHolder",
        },
        {
            question:
                "Are you an immediate family member or close associate of a person currently holding public office",
            answers: ["YES", "NO"],
            title: "relatedPublicOfficeHolder",
        },
        {
            question:
                "Are you currently holding a senior position in a government owned company",
            answers: ["YES", "NO"],
            title: "seniorPositionGovernment",
        },
        {
            question:
                "Are you currently holding a senior position in an international organization",
            answers: ["YES", "NO"],
            title: "memberBankStaff",
        },
        {
            question: "Are you a prominent member of a Political party",
            answers: ["YES", "NO"],
            title: "prominentMemeberPoliticalParty",
        },
        {
            question:
                "Are you a member of a ruling Royal family or close associate of such family",
            answers: ["YES", "NO"],
            title: "relatedRoyalFamily",
        },
    ];

    pepAnswers: any;

    constructor(
        private dashboardService: DashboardService,
        private clientService: ClientService,
        private toastr: ToastrService,
        private router: Router
    ) { }

    ngOnInit() {
        this.pepQuestions.map((question) => {
            this.pepAnswers = { ...this.pepAnswers, [question.title]: "" };
        });
        this.client = this.clientService.getClientLocally();
    }

    checkValidity() {
        return Object.values(this.pepAnswers).includes("");
    }

    getAnswer(value) {
        return value == "YES" ? "true" : "false";
    }

    submit() {
        this.loading = true;
        Object.assign(
            this.pepAnswers,
            this.clientService.getClientEncodedkey()
        );
        this.dashboardService.fillPEPdetails(this.pepAnswers).subscribe(
            () => {
                this.toastr.success("Details updated");
                this.client.pEPValidated = true;
                this.client.pepvalidated = true;
                this.getUserData();
            },
            (error) => {
                this.loading = false;
                this.toastr.error("Something went wrong, Try again");
            }
        );
    }

    getUserData() {
        this.clientService.getClient().subscribe(
            (data) => {
                data.data["gender"] = data.data.sex;
                this.clientService.saveClientLocally(data.data);
                this.loading = false;
                this.done.emit(true);
            },
            (error) => {
                this.loading = false;
                this.toastr.error("Something went wrong, Try again");
            }
        );
    }
}
