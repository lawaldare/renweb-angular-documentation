import { NgForm } from '@angular/forms';
import { Component, OnInit } from '@angular/core';
import { LoanService } from 'src/app/@core/loan.service';
import { BsModalService } from 'ngx-bootstrap/modal';
import { ToastrService } from 'ngx-toastr';
import { DialogBoxComponent } from 'src/app/@page/shared/dialog-box/dialog-box.component';
import * as moment from 'moment';
import { appRoutes } from 'src/app/@config/routes';
import { Client } from 'src/app/@model/client';
import { ActivatedRoute, Router } from '@angular/router';
import { DashboardService } from 'src/app/@core/dashboard.service';
import { RenConfig } from 'src/app/@page/create/loan/loan-calculator/loan-config-mockdata';

@Component({
    selector: 'app-loan-calculator-page',
    templateUrl: './loan-calculator.component.html',
    styleUrls: [
        "../../@page/page/page.component.scss",
        '../../@page/create/loan/select-loan-type/select-loan-type.component.scss',
        "../investment-calculator/investment-calculator.component.scss",
        './loan-calculator.component.scss']
})
export class LoanCalculatorPageComponent implements OnInit {

    presentPage = 0;

    bsConfig = { dateInputFormat: "DD/MM/YYYY", adaptivePosition: true };
    maxDate: Date;
    minDate: Date;
    stateCode: string = "";
    clientLGA: string = "";
    dob;
    states: { name: string; code: string }[];
    LGAs: { name: string }[];
    amountCurrency: string;
    modalRef: any;
    routes;
    client: Client = {};
    salary: string = "";
    employementStatus: string;
    dateFromCalender = "";
    config: any;
    salesAgent: string;
    teamLead: string;
    subTeam: string;
    salesAgentName: string;
    teamLeadName: string;
    subTeamName: string;
    teamList: any;


    constructor(private loanService: LoanService,
        private router: Router,
        private route: ActivatedRoute,
        private modalService: BsModalService,
        private toastr: ToastrService,
        private dashboardService: DashboardService) {
        // const maximumDOBAllowed = moment().subtract(22, "years").toDate();
        this.maxDate = new Date();
        this.maxDate.setDate(this.maxDate.getDate())
    }


    setDOB(event) {
        this.dateFromCalender = event;
        this.dob = this.dateFromCalender;
    }


    ngOnInit() {
        this.routes = appRoutes;
        this.getSupportedStates();
        this.dashboardService.saveConfig(RenConfig);
        // this.goToNextState(this.presentPage);
        this.config = this.dashboardService.getConfigLocally();
        this.route.paramMap.subscribe((params) => {
            this.salesAgent = params.get("sales-agent");
            this.teamLead = params.get("team-lead");
            this.subTeam = params.get("sub-team");
            this.salesAgentName = this.hex2bin(this.salesAgent);
            this.teamLeadName = this.hex2bin(this.teamLead);
            this.subTeamName = this.hex2bin(this.subTeam);

            let teamList: any = {};
            if (this.teamLead) {
                teamList._salesteamLead = this.teamLeadName;
            }
            if (this.salesAgent) {
                teamList._loansalesAgent = this.salesAgentName;
            }
            if (this.subTeam) {
                teamList.salesSubTeam = this.subTeamName;
            }
            this.loanService.saveTeamLists(teamList);
        });
        this.goToNextState(this.presentPage)
    }


    hex2bin(encodedKey: string) {
        let ret = []
        let i = 0
        let l
        encodedKey += '';
        for (l = encodedKey.length; i < l; i += 2) {
            let c = parseInt(encodedKey.substr(i, 1), 16)
            let k = parseInt(encodedKey.substr(i + 1, 1), 16)
            if (isNaN(c) || isNaN(k)) return false
            ret.push((c << 4) | k)
        }
        return String.fromCharCode.apply(String, ret)
    }

    goToNextState(presentPage) {
        if (this.salesAgent && this.teamLead && this.subTeam) {
            this.router.navigate([`${appRoutes.onBoardingLoanCalculator}/${this.salesAgent}/${this.teamLead}/${this.subTeam}`], { queryParams: { presentPage: presentPage } });
        } else if (this.salesAgent && this.teamLead && !this.subTeam) {
            this.router.navigate([`${appRoutes.onBoardingLoanCalculator}/${this.salesAgent}/${this.teamLead}`], { queryParams: { presentPage: presentPage } });
        } else if (!this.salesAgent && !this.teamLead && !this.subTeam) {
            this.router.navigate([appRoutes.onBoardingLoanCalculator], { queryParams: { presentPage: presentPage } });
        }
    }


    decrementState() {
        this.presentPage--;
        this.goToNextState(this.presentPage);
    }

    updatePersonalInfo(form: NgForm) {
        if (this.dateFromCalender) {
            const maximumDOBAllowed = moment().subtract(22, "years").toDate().getTime();
            this.client.dateOfBirth = moment(this.dob, 'DD-MMM-YYYY').format('YYYY-MM-DD');
            if (maximumDOBAllowed < this.dob.getTime()) {
                this.modalRef = this.modalService.show(DialogBoxComponent, {
                    backdrop: "static",
                    initialState: {
                        type: "info",
                        header: "Oops!",
                        body: "Your loan cannot be processed at this time because you do not meet our minimum age criteria of 22 years. Please reapply once you become eligible.",
                        confirmText: "Done",
                    },
                });
            } else if (form.value.employementStatus === 'Full Time (Private)' && this.currencyToNumber(form.value.salary) < this.config.plMinimumSalary) {
                this.modalRef = this.modalService.show(DialogBoxComponent, {
                    backdrop: "static",
                    initialState: {
                        type: "info",
                        header: "Oops!",
                        body: "You are not eligible to apply for a loan due to your monthly income",
                        confirmText: "Done",
                    },
                });
            } else if (form.value.employementStatus === 'Self-Employed' && this.currencyToNumber(form.value.salary) < this.config.mblMinimumSalary) {
                this.modalRef = this.modalService.show(DialogBoxComponent, {
                    backdrop: "static",
                    initialState: {
                        type: "info",
                        header: "Oops!",
                        body: "You are not eligible to apply for a loan due to your monthly income",
                        confirmText: "Done",
                    },
                });
            } else {
                Object.assign(this.client, { addressState: this.states.find(x => x.code === this.stateCode).name });
                Object.assign(this.client, { addressLGA: this.LGAs.find(x => x.name === this.clientLGA).name });
                // this.loanService.saveLoanClient(this.client)
                this.loanService.saveLoanClientFromLoanPage(this.client);
                this.presentPage++;
                this.goToNextState(this.presentPage)
            }
        } else {
            this.modalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Oops!",
                    body: `Please enter a valid date of birth`,
                    confirmText: "Done",
                },
            });
        }


    }


    populateLoanInfo(loanInfo) {
        // this.loanService.savePresentLoanDetail(loanInfo);
        this.loanService.saveLoanDetailFromLoanPage(loanInfo);
        this.router.navigate([appRoutes.login]);
        // this.openRequirementModal();

    }
    getSupportedStates() {
        this.loanService.getSupportedStates()
            .subscribe(data => {
                this.states = data.data;

            }, error => {
                this.toastr.error(error.message);
            })
    }

    getSupportedStateLGAs(stateCode: string) {
        this.loanService.getSupportedLGAsforSelectedState(stateCode)
            .subscribe(data => {
                this.LGAs = data.data;

            }, error => {
                this.toastr.error(error.message);
            })
    }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "").replace("", "");
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57)) {
            return false;
        }
        return true;
    }

    checkIfAmountIsValid(amount) {
        Object.assign(this.client, { salary: this.currencyToNumber(amount) });
        this.amountCurrency = this.numberToCurrencyFormat(this.client.salary);
    }

}
