import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { NgForm } from '@angular/forms';
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';
import { ToastrService } from 'ngx-toastr';
import { ClientService } from 'src/app/@core/client.service';
import { InvestmentService } from 'src/app/@core/investment.service';
import { Client } from 'src/app/@model/client';
import { InvestmentInterface } from 'src/app/@model/interfaces';
import { DialogBoxComponent } from 'src/app/@page/shared/dialog-box/dialog-box.component';


@Component({
    selector: 'app-target-savings-partial-withdraw',
    templateUrl: './target-savings-partial-withdraw.component.html',
    styleUrls: ['../../../invest/plan-name/plan-name.component.scss', '../../../withdraw/withdraw.component.scss', '../../../single-view/partial/partial.component.scss', './target-savings-partial-withdraw.component.scss']
})
export class TargetSavingsPartialWithdrawComponent implements OnInit {

    client: Client;
    loading: boolean = false;
    renflexAccount: any;
    availableBalance: number;
    availableBalanceAfterWithdrawal: number;
    amountToTransferDub: number;
    partialWithdrawalFormValue: any;
    passwordNextState: string = 'Show';
    shouldShowPassword: boolean = true;
    showDepositCalculator: boolean = false;
    isTargetSavings: boolean;
    amount: any;
    tenure: any;
    notValid: boolean = true;
    amountToTransfer: string;
    tempAmountToTransfer: number;
    timeOut: any;
    bsModalRef: BsModalRef;


    @Output() done: EventEmitter<boolean> = new EventEmitter<boolean>();
    @Input() investment: InvestmentInterface;

    constructor(private investmentService: InvestmentService, private toastr: ToastrService, private clientService: ClientService, private modalService: BsModalService) { }

    ngOnInit(): void {
        this.client = this.clientService.getClientLocally();
        if (!this.client.nuban) {
            this.retrieveNuban();
        } else {
            this.getRenmoneyAccount(this.client.nuban);
        }
    }




    retrieveNuban() {
        this.clientService.getClientNuban(this.client.clientEncodedkey)
            .subscribe((data) => {
                this.client.nuban = data.data.nuban;
                this.getRenmoneyAccount(this.client.nuban);
            }, () => {
                this.toastr.error("Couldn't retrieve your pocket account");
            })
    }


    getRenmoneyAccount(id) {
        this.loading = true;
        this.investmentService.getSingleInvestment(id).subscribe(
            (data) => {
                this.renflexAccount = data.data.clientDeposit;
                console.log(this.renflexAccount);

                this.availableBalance = this.renflexAccount.availableBalance;
                this.loading = false;
            },
            (error) => {
                this.loading = false;
                this.toastr.error("Couldn't retrieve your pocket account");
            }
        );
    }



    calculateRemainingBalance(num) {
        num = this.currencyToNumber(this.amountToTransfer);
        if (num <= this.renflexAccount.availableBalance) {
            this.availableBalance = this.renflexAccount.availableBalance - num;
        } else {
            this.availableBalance = 0;
        }
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57) && charCode !== 46) {
            return false;
        }
        return true;
    }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `₦${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "").replace("₦", "");
    }



    checkIfAmountIsValid() {
        this.tempAmountToTransfer = this.currencyToNumber(this.amountToTransfer);
        this.amountToTransfer = this.numberToCurrencyFormat(this.tempAmountToTransfer);
    }

    onSubmitPartialWithdrawalForm(form: NgForm) {
        let amount = this.currencyToNumber(form.value.amountToTransfer);
        if (amount > this.renflexAccount.availableBalance) {
            this.bsModalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Notice",
                    body: `The maximum amount you can fund is ₦${this.renflexAccount.availableBalance} `,
                    confirmText: "Yes",
                    cancelText: "No",
                },
            });
        } else if (amount < +this.investment.contributionAmount) {
            this.bsModalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Notice",
                    body: `The minimum amount you can fund is ₦${+this.investment.contributionAmount} `,
                    confirmText: "Yes",
                    cancelText: "No",
                },
            });
        } else {
            let payload: any = {};
            payload.debitAccountId = this.renflexAccount.id;
            payload.creditAccountId = this.investment._id;
            payload.minorAmount = amount * 100;
            payload.narration = `Funded ${this.investment.planName} from regular savings`;
            this.loading = true;
            this.investmentService.intrabankTransfer(payload).subscribe(
                (data) => {
                    this.loading = false;
                    this.investment.paymentMade = true;
                    this.done.emit(true);
                },
                (error) => {
                    this.loading = false;
                    this.toastr.error(error.error.message);
                }
            );
        }

    }


}
