import { ClientService } from "src/app/@core/client.service";
import { NgForm } from "@angular/forms";
import { Component, OnInit, Input, EventEmitter, Output } from "@angular/core";
import { BsModalService, BsModalRef } from "ngx-bootstrap/modal";
import { Client } from "src/app/@model/client";
import { ToastrService } from "ngx-toastr";

@Component({
    selector: "app-next-of-kin-modal",
    templateUrl: "./next-of-kin-modal.component.html",
    styleUrls: [
        "../../single-view/complete/complete.component.scss",
        "./next-of-kin-modal.component.scss",
    ],
})
export class NextOfKinModalComponent implements OnInit {


    @Output() done = new EventEmitter<Client>();
    @Input() isInPage = false;



    close: boolean;
    relationships: string[] = ["Spouse", "Sibling", "Parent"];
    @Input() nokRelationship = "";
    @Input() client: Client;

    @Input() nokFirstName: string;
    @Input() nokMiddleName: string;
    @Input() nokLastName: string;
    @Input() nokPhone: string;

    nokDetails = {};
    nextOfKinModalLoader: boolean = false;

    constructor(
        public bsModalRef: BsModalRef,
        private clientService: ClientService,
        private toastr: ToastrService
    ) { }

    ngOnInit(): void {
        if (!this.isInPage) {
            this.client = this.clientService.getClientLocallyForAccountSetup();
        }

    }

    closeModal() {
        this.bsModalRef.hide();
    }

    saveNextOfKin(form: NgForm) {
        // this.nextOfKinModalLoader = true;
        let payload: any = {};
        payload.nokFirstName = form.value.nokFirstName;
        payload.nokMiddleName = form.value.nokMiddleName;
        payload.nokSurname = form.value.nokLastName;
        payload.nokLastName = form.value.nokLastName;
        payload.nokPhoneNumber = form.value.nokPhone;
        payload.nokPhone = form.value.nokPhone;
        payload.nokRelationship = form.value.nokRelationship;
        // payload.id = this.client.bvn;
        payload.dob = this.client.dateOfBirth;
        payload.nokDone = true;
        Object.assign(this.client, payload);

        if (this.client.sex) {
            this.client.gender = this.client.sex.toUpperCase();
        }


        this.clientService.changeNOKState(true);

        this.clientService.saveClientLocallyForAccountSetup(this.client);

        this.toastr.success("Next of kin successfully added");

        form.resetForm();

        if (this.isInPage) {
            this.done.emit(this.client);
        } else {
            this.bsModalRef.hide();
        }

        // this.clientService.createClient(this.client).subscribe(

    }
}
