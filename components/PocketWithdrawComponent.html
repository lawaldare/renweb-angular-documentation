import { Location } from "@angular/common";
import { Component, OnInit } from "@angular/core";
import { Router } from "@angular/router";
import { ToastrService } from "ngx-toastr";
import { appRoutes } from "src/app/@config/routes";
import { ClientService } from "src/app/@core/client.service";
import { GeneralService } from "src/app/@core/general.service";
import { InvestmentService } from "src/app/@core/investment.service";
import { Client } from "src/app/@model/client";
import { InvestmentInterface } from "src/app/@model/interfaces";

@Component({
    selector: "app-pocket-withdraw",
    templateUrl: "./pocket-withdraw.component.html",
    styleUrls: ["./pocket-withdraw.component.scss"],
})
export class PocketWithdrawComponent implements OnInit {
    client: Client = {};
    investment: InvestmentInterface = {};
    pocketAccount: any;

    constructor(
        public clientService: ClientService,
        public router: Router,
        public locationRef: Location,
        public generalService: GeneralService,
        public investmentService: InvestmentService,
        private toastr: ToastrService
    ) { }

    ngOnInit(): void {
        this.client = this.clientService.getClientLocally();
        if (!this.client.nuban) {
            this.retrieveNuban();
        } else {
            this.getPocketAccount(this.client.nuban);
        }
        if (!this.client.bvnValidated) {
            this.router.navigate([appRoutes.accountSetUp]);
        }
        this.generalService.showLoader(true);
    }

    // Pocket account

    retrieveNuban() {
        this.clientService.getClientNuban(this.client.clientEncodedkey)
            .subscribe((data) => {
                this.client.nuban = data.data.nuban;
                this.getPocketAccount(this.client.nuban);
            }, () => {
                this.toastr.error("Couldn't retrieve your pocket account");
            })
    }

    getPocketAccount(id) {
        this.generalService.showLoader(true);
        this.investmentService.getSingleInvestment(id).subscribe(
            (data) => {
                this.pocketAccount = data.data.clientDeposit;

                if (!this.pocketAccount) {
                    return this.noPocketAccount();
                }

                this.investment = {
                    _id: this.pocketAccount.id,
                    availableBalance: this.pocketAccount.availableBalance,
                };

                this.generalService.showLoader(false);

            },
            (error) => {
                this.toastr.error("Couldn't retrieve your pocket account");
                this.generalService.showLoader(false);
            }
        );
    }


    noPocketAccount() {
        this.toastr.warning(
            "You don't have a pocket account to withdraw from, please fund one so it can be created."
        );
        return this.router.navigate([appRoutes.dashboard]);
    }
}
