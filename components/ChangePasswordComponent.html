import { Component, OnInit } from '@angular/core';
import { ResetModel, Client } from 'src/app/@model/client';
import { appRoutes } from 'src/app/@config/routes';
import { AuthService } from 'src/app/@core/auth.service';
import { Router, ActivatedRoute } from '@angular/router';
import { ToastrService } from 'ngx-toastr';
import { Location } from '@angular/common';
import { Base64 } from 'js-base64';


@Component({
    selector: 'app-change-password',
    templateUrl: './change-password.component.html',
    styleUrls: ["../auth/auth-mobile.scss", '../signup/signup.component.scss', './change-password.component.scss']
})
export class ChangePasswordComponent implements OnInit {

    client = new Client();
    routes = appRoutes;
    shouldShowPassword = true;
    passwordNextState = 'Show'; // it's either show or hide

    constructor(private authService: AuthService,
        private router: Router,
        private location: Location,
        private route: ActivatedRoute,
        private toastr: ToastrService) { }

    ngOnInit() {
        this.route.paramMap.subscribe(param => {
            this.client.emailOrPhone = Base64.decode(param.get('email'));
            this.client.otp = Base64.decode(param.get('otp'));
        })
    }

    passwordReset(client: ResetModel) {
        this.authService.changeState(true);
        this.authService.resetPassword(client)
            .subscribe(data => {
                this.authService.changeState(false);
                this.router.navigate([this.routes.login]);
                this.toastr.success('Password has been reset successfully');
            },
                error => {
                    this.authService.changeState(false);
                    this.location.back()
                    this.toastr.error(error.error.message || "Couldn't reset password, try again");
                })
    }

    /**
     * Changes the password type of the password field
     * it has just 2 states either "Hide" or "Show"
     * @param shouldShowPassword helps to know when to hide or show password
     * @param $event the triggered event
     */
    changePasswordState(shouldShowPassword: boolean, event) {
        if (!shouldShowPassword) {
            this.passwordNextState = 'Show';
            event.target.parentElement.children[1].type = 'password';
            this.shouldShowPassword = !shouldShowPassword;
            return
        }
        this.passwordNextState = 'Hide';
        event.target.parentElement.children[1].type = 'text';
        this.shouldShowPassword = !shouldShowPassword;
    }

}
