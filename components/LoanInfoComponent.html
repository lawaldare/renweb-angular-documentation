import { Component, OnInit, Output, EventEmitter, Input, AfterViewInit } from '@angular/core';
import { Client } from 'src/app/@model/client';
import { LoanService } from 'src/app/@core/loan.service';
import { ToastrService } from 'ngx-toastr';
import * as moment from 'moment';
import { BsModalService } from 'ngx-bootstrap/modal';
import { DialogBoxComponent } from 'src/app/@page/shared/dialog-box/dialog-box.component';
import { DashboardService } from 'src/app/@core/dashboard.service';


@Component({
    selector: 'app-loan-info',
    templateUrl: './loan-info.component.html',
    styleUrls: ["../../../invest/invest/invest.shared.scss",
        "../../../invest/plan-name/plan-name.component.scss",
        './loan-info.component.scss']
})
export class LoanInfoComponent implements OnInit, AfterViewInit {

    @Output() done: EventEmitter<boolean> = new EventEmitter<boolean
    >();
    @Input() client: Client;
    employmentStatus: string;

    bsConfig = { dateInputFormat: "DD/MM/YYYY", adaptivePosition: true };
    maxDate: Date;
    minDate: Date;
    stateCode: string = "";
    clientLGA: string = "";
    dob;
    dateFromCalender = "";
    states: any[];
    LGAs: any[];
    amountCurrency: string;
    modalRef: any;
    config: any;
    constructor(private loanService: LoanService,
        private modalService: BsModalService,
        private toastr: ToastrService, private dashboardService: DashboardService) {
        this.maxDate = new Date();
        this.maxDate.setDate(this.maxDate.getDate())
        // this.loanService.changeStepNumber(2);
    }

    ngOnInit(): void {
        this.config = this.dashboardService.getConfigLocally();
        if (this.client.employmentStatus === 'Full Time (Private)') {
            this.employmentStatus = 'salary';
        } else {
            this.employmentStatus = 'income'
        }

    }


    setDOB(event) {
        this.dateFromCalender = event;
        this.dob = this.dateFromCalender;
    }

    ngAfterViewInit(): void {

        //Called after ngAfterContentInit when the component's view has been initialized. Applies to components only.
        //Add 'implements AfterViewInit' to the class.
        this.getSupportedStates();
        if (this.client.dob) {
            this.dob = new Date(this.client.dob);
            this.dateFromCalender = this.dob;
        }
        if (this.client.salary) {
            this.amountCurrency = this.numberToCurrencyFormat(this.client.salary);
        }

        // this.loanService.changeStepNumber(2);

    }

    updatePersonalInfo() {
        if (this.dateFromCalender && this.dob) {
            const maximumDOBAllowed = moment().subtract(22, "years").toDate().getTime();
            if (maximumDOBAllowed < this.dob.getTime()) {
                this.modalRef = this.modalService.show(DialogBoxComponent, {
                    backdrop: "static",
                    initialState: {
                        type: "info",
                        header: "Oops!",
                        body: `You are not eligible for a loan, you are ${moment().diff(this.dob, 'years')} years old`,
                        confirmText: "Done",
                    },
                });
            } else if (this.client.employmentStatus === 'Full Time (Private)' && this.client.salary < this.config.plMinimumSalary) {
                this.modalRef = this.modalService.show(DialogBoxComponent, {
                    backdrop: "static",
                    initialState: {
                        type: "info",
                        header: "Oops!",
                        body: "You are not eligible to apply for a loan due to your monthly income",
                        confirmText: "Done",
                    },
                });
            } else if (this.client.employmentStatus === 'Self-Employed' && this.client.salary < this.config.mblMinimumSalary) {
                this.modalRef = this.modalService.show(DialogBoxComponent, {
                    backdrop: "static",
                    initialState: {
                        type: "info",
                        header: "Oops!",
                        body: "You are not eligible to apply for a loan due to your monthly income",
                        confirmText: "Done",
                    },
                });
            } else {
                this.client.dateOfBirth = moment(this.dob, 'DD-MMM-YYYY').format('YYYY-MM-DD');
                this.client.dob = this.client.dateOfBirth;
                Object.assign(this.client, { addressState: this.states.find(x => x.code === this.stateCode).name });
                Object.assign(this.client, { addressLGA: this.LGAs.find(x => x.name === this.clientLGA).name });
                this.done.emit();
            }
        } else {
            this.modalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: "info",
                    header: "Oops!",
                    body: `Please enter a valid date of birth`,
                    confirmText: "Done",
                },
            });
        }


    }

    getSupportedStates() {
        this.loanService.getClientStatesAndLGAs().subscribe((data: any) => {
            this.states = data.data;
            if (this.client.addressState) {
                this.stateCode = this.states.find(x => x.name === this.client.addressState).code;
                this.getSupportedStateLGAs(this.stateCode);
            }

        }, error => {
            this.toastr.error(error.message);
        })
    }

    getSupportedStateLGAs(stateCode: string) {
        this.LGAs = this.states.find(state => state.code === stateCode).lga;
    }

    // setLGA(lgaName: string) {
    //     this.client.addressLGA = lgaName;
    // }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "").replace("", "");
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57)) {
            return false;
        }
        return true;
    }

    checkIfAmountIsValid(amount) {
        Object.assign(this.client, { salary: this.currencyToNumber(amount) });
        this.amountCurrency = this.numberToCurrencyFormat(this.client.salary);
    }

}
