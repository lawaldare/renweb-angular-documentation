import { LoanService } from 'src/app/@core/loan.service';
import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { GeneralService } from 'src/app/@core/general.service';
import { Client } from 'src/app/@model/client';
import { ClientService } from 'src/app/@core/client.service';
import { ToastrService } from 'ngx-toastr';
import { DomSanitizer } from '@angular/platform-browser';
import { Subject, Observable } from 'rxjs';
import { WebcamImage, WebcamInitError, WebcamUtil } from 'ngx-webcam';

@Component({
    selector: 'app-upload-loan-img',
    templateUrl: './upload-loan-img.component.html',
    styleUrls: ["../../../invest/invest/invest.shared.scss",
        "../../../invest/plan-name/plan-name.component.scss",
        './upload-loan-img.component.scss']
})
export class UploadLoanImgComponent implements OnInit {

    @Input() client: Client;
    @Input() uploadType: string;
    @Input() appTitle: string;
    @Input() copy: string;
    @Input() loanID: string;
    @Output() done = new EventEmitter<boolean>();
    selectedPhoto: any;
    imageUrl: any;
    loading: boolean = false;
    buttonText: string = 'Take a selfie';
    isItCameraPhoto: boolean = false;
    loanEncodedKey: string;



    public showWebcam = false;
    public webcamImage: WebcamImage = null;
    private trigger: Subject<void> = new Subject<void>();
    public get triggerObservable(): Observable<void> {
        return this.trigger.asObservable();
    }

    constructor(private generalService: GeneralService,
        private clientService: ClientService,
        public sanitizer: DomSanitizer,
        private toastr: ToastrService, private loanService: LoanService) { }

    ngOnInit(): void {
    }


    public triggerSnapshot(): void {
        this.buttonText = 'Click to capture';
        this.showWebcam = true;
        this.trigger.next();
    }

    showVideo() {
        this.showWebcam = false;
        this.trigger.next();
    }


    public handleInitError(error: WebcamInitError): void {
        this.showWebcam = false;
        if (error.mediaStreamError && error.mediaStreamError.name === "NotAllowedError") {
            this.toastr.error("Camera access was not allowed by user!");
        } else if (error.mediaStreamError && error.mediaStreamError.name === "NotFoundError") {
            this.toastr.error("No camera found!")
        };
    }



    public handleImage(webcamImage: WebcamImage): void {
        this.webcamImage = webcamImage;
        this.imageUrl = webcamImage.imageAsDataUrl;
        this.selectedPhoto = webcamImage;
        this.isItCameraPhoto = true;
    }


    selectPhoto(event) {
        let file = event.target.files[0];
        if (file) {
            let acceptedTypes = ["image/jpeg", "image/png"];

            if (!acceptedTypes.includes(file.type)) {
                this.selectedPhoto = null;
                return this.toastr.error(
                    "Error, only jpeg or png images are allowed please."
                );
            }

            if (file.size > 2000000) {
                this.selectedPhoto = null;
                return this.toastr.error(
                    "Error, the maximum file size you can upload is 2MB."
                );
            }

            this.selectedPhoto = file;
            this.isItCameraPhoto = false;

            this.imageUrl = this.sanitizer.bypassSecurityTrustResourceUrl(
                URL.createObjectURL(this.selectedPhoto)
            );
        }
    }

    uploadSelfieImage() {
        this.loanEncodedKey = this.loanService.getLoanEncodedKey();
        this.loading = true;
        let data = {
            documentType: this.uploadType,
            type: "loan_account",
            documentMime: this.selectedPhoto._mimeType.split("/")[1],
            encodedKey: this.loanEncodedKey,
            document: this.selectedPhoto._imageAsDataUrl.toString().split("base64,")[1]
        }

        this.clientService.uploadProfilePicture(data).subscribe(
            (response) => {
                this.loanService.sendLoanDocsStatus({
                    documentType: this.uploadType,
                    loanId: this.loanID,
                    uploaded: true,
                    type: 'loan_account'
                }).subscribe(data => {
                    this.loading = false;
                    this.toastr.success(
                        `Great, ${this.uploadType} uploaded successfully.`
                    );
                    this.done.emit()
                }, error => {
                    console.log(error);
                })
            },
            (error) => {
                this.loading = false;
                this.toastr.error(
                    error.error.message ||
                    "An error occurred, please try again later."
                );
            }
        );
    }



    uploadImageUploaded() {
        this.loanEncodedKey = this.loanService.getLoanEncodedKey();
        this.loading = true;
        let reader = new FileReader();
        reader.readAsDataURL(this.selectedPhoto);
        reader.onload = () => {
            let data = {
                documentType: this.uploadType,
                type: "loan_account",
                documentMime: this.selectedPhoto.type.split("/")[1],
                encodedKey: this.loanEncodedKey,
                document: reader.result.toString().split("base64,")[1],
            };

            this.clientService.uploadProfilePicture(data).subscribe(
                (response) => {
                    this.loanService.sendLoanDocsStatus({
                        documentType: this.uploadType,
                        loanId: this.loanID,
                        uploaded: true,
                        type: 'loan_account'
                    }).subscribe(data => {
                        this.loading = false;
                        this.toastr.success(
                            `Great, ${this.uploadType} uploaded successfully.`
                        );
                        this.done.emit()
                    }, error => {
                        console.log(error);
                    })
                },
                (error) => {
                    this.loading = false;
                    this.toastr.error(
                        error.error.message ||
                        "An error occurred, please try again later."
                    );
                }
            );
        };

        reader.onerror = (error) => {
            this.loading = false;
            this.toastr.error("An error occurred, please try again.");
        };
    }

    uploadDoc() {
        if (this.isItCameraPhoto) {
            this.uploadSelfieImage();
        } else {
            this.uploadImageUploaded();
        }
    }






}
