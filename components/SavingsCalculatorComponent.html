import { Component, OnInit } from "@angular/core";
import { ToastrService } from "ngx-toastr";
import { InvestmentService } from "src/app/@core/investment.service";
import { Router } from "@angular/router";
import { appRoutes } from "src/app/@config/routes";

@Component({
    selector: "app-savings-calculator",
    templateUrl: "./savings-calculator.component.html",
    styleUrls: [
        "../../@page/page/page.component.scss",
        "../investment-calculator/investment-calculator.component.scss",
        "./savings-calculator.component.scss",

export class SavingsCalculatorComponent implements OnInit {
    amount: number;
    minAmount = 5000;
    timeOut: any;

    amountCurrency: string;
    monthlyInterest: number;
    interestRate: number = 10;
    totalAmountToBePaid: number;
    tenure: number;

    interestRatesArr = [
        {
            min: 5000,
            max: 200000 - 1,
            rates: {
                365: 10,
                270: 10,
                180: 10,
                90: 10,
            },
        },
        {
            min: 200000,
            max: 1000000 - 1,
            rates: {
                365: 10,
                270: 10,
                180: 10,
                90: 10,
            },
        },
        {
            min: 1000000,
            max: 10000000,
            rates: {
                365: 10,
                270: 10,
                180: 10,
                90: 10,
            },
        },
    ];

    rates: number[];
    MILLISECOND_IN_A_DAY = 86400000;
    maturityDate: number;
    interest: number;
    selectedIndex: number;
    routes = appRoutes;
    frequentAmount: number;
    frequency: number; //in months so monthly is 1, weekly is 1/4, daily is 1/30
    savingsType = "target";
    investments = ["smart goal", "regular savings"];
    activeInvestment = this.investments[0];
    tenures = ["monthly", "weekly", "daily"];
    activeTenure = this.tenures[0];

    constructor(
        private toastr: ToastrService,
        private router: Router,
        private investmentService: InvestmentService
    ) {}

    ngOnInit() {
        this.monthlyInterest = 10;
        this.amount = this.minAmount * 1000;
        this.frequency = 1;
        this.tenure = 1;
        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
        this.getInterestRates(this.amount);
    }

    selectTenure(index, durationInDays) {
         
        this.selectedIndex = index;
        this.maturityDate =
            new Date().getTime() + durationInDays * this.MILLISECOND_IN_A_DAY;
        this.interest =
            ((this.monthlyInterest / 100) * this.amount * durationInDays) / 30;

        const tenureBoxes = document.querySelectorAll(".tenure-box");
        this.clearSelectedBoxes();
        tenureBoxes[index].classList.add("selected");
    }

    clearSelectedBoxes() {
        const tenureBoxes = document.querySelectorAll(".tenure-box");
        tenureBoxes.forEach((tenureBox) => {
            tenureBox.classList.replace("selected", "-");
        });
    }

    numberChecks(amount) {
        this.getInterestRates(amount);
        this.clearSelectedBoxes();
    }

    /**
     * Get Investment all investment amount ranges and corresponding rates
     */
    getInvestmentRates() {
        // TODO: find out API structure for calculator
    }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "");
    }

    checkIfAmountIsValid() {
        clearTimeout(this.timeOut);
        this.amount = this.currencyToNumber(this.amountCurrency);
        this.amountCurrency = this.numberToCurrencyFormat(this.amount);
        const amount = this.amount;

        this.timeOut = setTimeout(() => {
            if (amount < this.minAmount) {
                this.amount = this.minAmount;
                this.amountCurrency = this.numberToCurrencyFormat(this.amount);
                this.toastr.warning(
                    "The minimum amount you can be invested is " +
                        this.minAmount
                );
            }
        }, 1000);
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57)) {
            return false;
        }
        return true;
    }

    getInterestRates(amount: number) {
        delete this.maturityDate;
        delete this.interest;
        this.interestRatesArr.forEach((rateObj) => {
            if (amount >= rateObj.min && amount <= rateObj.max) {
                this.rates = Object.values(rateObj.rates).sort(
                    (a, b) => +b - +a
                );
            }
        });
    }

    investNow(amount, tenure, frequentAmount) {
        if (amount && tenure && this.frequency) {
            this.investmentService.setCalcType(this.savingsType);
            this.investmentService.saveCalcValues({
                amount,
                tenure,
                frequentAmount,
                frequency: this.frequency,
            });
        }
        this.router.navigate([appRoutes.register]);
    }
}
