import { Component, OnInit } from "@angular/core";
import { BsModalRef } from "ngx-bootstrap/modal";
import { ClientService } from "src/app/@core/client.service";
import { ToastrService } from "ngx-toastr";
import { InvestmentService } from "src/app/@core/investment.service";
import { GeneralService } from "src/app/@core/general.service";
import { getInvestmentsSuccessfullResponse } from "../invest/invest/investment.mockdata";

@Component({
    selector: "app-withdraw",
    templateUrl: "./withdraw.component.html",
    providers: [BsModalRef],
    styleUrls: [
        "../page/page.component.scss",
        "../invest/invest/invest.component.scss",
        "./withdraw.component.scss"
    ]
})
export class WithdrawComponent implements OnInit {
    state = 0;

    client: any;
    date = new Date();
    calculatorDetails: any;
    investments = [];
    constructor(
        private bsModalRef: BsModalRef,
        private clientService: ClientService,
        private toastr: ToastrService,
        public investmentService: InvestmentService,
        private generalService: GeneralService
    ) {}

    ngOnInit(): void {
        this.client = this.clientService.getClientLocally();
        this.calculatorDetails = this.investmentService.getCalcValues();
        this.getInvestments();
    }

    getInvestments() {
        this.generalService.showLoader(true);
        this.investmentService.getInvestments().subscribe(
            data => {
                this.generalService.showLoader(false);
                if (data.data.clientDeposits.length > 0) this.state = -1;
                this.investments = this.attachDueInvestmentKey(
                    data.data.clientDeposits
                );
            },
            error => {
                this.generalService.showLoader(false);
                this.state = -1;
                this.toastr.warning("This is test data");
                this.investments = this.attachDueInvestmentKey(
                    getInvestmentsSuccessfullResponse.data.clientDeposits
                );
            }
        );
    }

    /**
     * Check if the plan is due for withdrawal
     * @param date the maturity date for the specified plan
     */
    attachDueInvestmentKey(investments: Array<any>): Array<any> {
        investments.forEach(inv => {
            const now = new Date();
            const presentDate = new Date(
                `${now.getFullYear()}-${now.getMonth()}-${now.getDate()}`
            ).getTime();
            const invMaturityDate = new Date(inv.maturityDate).getTime();
            inv.isWithdrwalDue = presentDate > invMaturityDate ? true : false;
        });
        return investments;
    }
}
