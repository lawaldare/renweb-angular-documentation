import { ToastrService } from 'ngx-toastr';
import { NgForm } from '@angular/forms';
import { Component, OnInit, TemplateRef, Input } from '@angular/core';
import { InvestmentService } from 'src/app/@core/investment.service';
import { GeneralService } from 'src/app/@core/general.service';
import { ActivatedRoute, Router } from '@angular/router';
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';
import { Client, DialogModalButton } from 'src/app/@model/client';
import { ClientService } from 'src/app/@core/client.service';
import { DialogBoxComponent } from '../../shared/dialog-box/dialog-box.component';
import { appRoutes } from 'src/app/@config/routes';

@Component({
    selector: 'app-partial',
    templateUrl: './partial.component.html',
    styleUrls: ['../../invest/plan-name/plan-name.component.scss', '../withdraw/withdraw.component.scss', './partial.component.scss']
})
export class PartialComponent implements OnInit {

    investment: any
    selectedBankCode: string;
    selectedBankLogo: string;
    password: string;
    timeOut: any;
    dialogOption = new DialogModalButton();
    bank: any;
    accountDetails = { accountName: "", accountNumber: "", bank: "" };
    accountNumber: string;
    banks: any[];
    showLoader: boolean = false;
    client: Client;
    modalRef: BsModalRef;
    availableBalance: number;
    availableBalanceAfterWithdrawal: number;
    amountToTransferDub: number;
    partialWithdrawalFormValue: any;
    passwordNextState: string = 'Show';
    shouldShowPassword: boolean = true;
    showDepositCalculator: boolean = false;
    isTargetSavings: boolean;
    amount: any;
    tenure: any;
    notValid: boolean = true;
    amountToTransfer: string;
    tempAmountToTransfer: number;
    invest: {
        planName?: string;
        tenor?: number;
        amount?: number;
        interest?: any;
        totalPayOut?: number;
    } = {};

    // private _amountToTransfer: number;
    // get amountToTransfer(): number {
    //     return this._amountToTransfer;
    // }

    // set amountToTransfer(amount: number) {
    //     this._amountToTransfer = amount;
    //     this.calculateRemainingBalance(this.amountToTransfer);
    //     this.checkIfAmountIsValid();
    // }

    @Input() id: string;
    @Input() isPocketWithdrawal: boolean;




    constructor(private investmentService: InvestmentService, private generalService: GeneralService, private route: ActivatedRoute, private toastr: ToastrService, private modalService: BsModalService, private clientService: ClientService, private router: Router) { }

    ngOnInit(): void {
        this.getBanks();
        this.getSingleInvestment(this.id);
        // this.route.parent.paramMap.subscribe((params) => {
        //     this.id = params.get("id");
        //     this.getSingleInvestment(this.id);
        //     console.log(this.id);

        //     this.client = this.clientService.getClientLocally();
        //     this.getMambuDetails();
        // });
        this.checkingForClosedModal();
    }

    getBanks() {
        this.banks = this.investmentService.getBanks();
    }


    getMambuDetails() {
        this.generalService.showLoader(true);
        this.clientService.getClientFromMambu().subscribe(
            (data) => {

                if (data.data) {
                    if (data.data.bankLiquidationDetails) {
                        this.accountDetails.bank = data.data.bankLiquidationDetails.nameOfBank;
                        this.accountDetails.accountNumber = data.data.bankLiquidationDetails.accountNumber;

                        this.selectedBankCode = this.banks.find(
                            (x) => x.Name.toLowerCase() === this.bank.toLowerCase()
                        ).Code;
                        this.getBankLogo(this.selectedBankCode);
                    }
                }
                this.generalService.showLoader(false);
            },
            (error) => {
                this.toastr.error(
                    "Oops",
                    error.error.message || "Error occured while adding to bank"
                );
                this.generalService.showLoader(false);
            }
        );
    }


    getSingleInvestment(id: string) {
        this.generalService.showLoader(true);
        this.investmentService.getSingleInvestment(id).subscribe(
            (data) => {

                this.investment = data.data.clientDeposit
                this.checkingAccountType(this.investment);
                this.generalService.showLoader(false);
                this.availableBalance = this.investment.availableBalance;
                this.availableBalanceAfterWithdrawal = this.investment.availableBalance;
            },
            (error) => {
                this.generalService.showLoader(false);
            }
        );
    }

    checkingAccountType(obj: any) {
        if (obj.accountType === 'SAVINGS_PLAN') {
            this.isTargetSavings = true;
        } else if (obj.accountType === 'FIXED_DEPOSIT') {
            this.isTargetSavings = false;
        }
    }

    getBankLogo(code: string) {
        this.banks.forEach(x => {
            if (x.Code === code) {
                this.selectedBankLogo = x.logo;
            }
        })
    }

    closeModal() {
        this.modalRef.hide();
    }

    onSubmitPartialWithdrawalForm(form: NgForm) {
        let amount = this.currencyToNumber(form.value.amountToTransfer);
        this.availableBalanceAfterWithdrawal = this.investment.availableBalance - amount;
        this.amount = amount;
        if (!this.isPocketWithdrawal) {
            this.modalRef = this.modalService.show(DialogBoxComponent, {
                backdrop: "static",
                initialState: {
                    type: 'confirm',
                    header: 'Great!',
                    body: 'A new fixed deposit plan will be created with your remaining balance',
                    confirmText: 'continue',
                    dialogOption: this.dialogOption
                }
            });
        } else {
            this.router.navigate([appRoutes.investment, this.id, 'withdraw', 'partial', this.amount, (this.investment.duration / 30)]);
        }
    }


    calculateRemainingBalance(num) {
        num = this.currencyToNumber(this.amountToTransfer);
        if (num <= this.investment.availableBalance) {
            this.availableBalance = this.investment.availableBalance - num;
        } else {
            this.availableBalance = 0;
        }
    }

    isNumberKey(evt) {
        const charCode = evt.which ? evt.which : evt.keyCode;
        if (charCode > 31 && (charCode < 48 || charCode > 57)) {
            return false;
        }
        return true;
    }

    numberToCurrencyFormat(amount: number): string {
        if (!amount) {
            return "";
        }
        return `₦${amount.toFixed().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,")}`;
    }

    currencyToNumber(amount): number {
        if (typeof amount !== "string") {
            amount = amount.toString();
        }
        return +amount.replace(/,|_/g, "").replace("₦", "");
    }



    checkIfAmountIsValid() {
        clearTimeout(this.timeOut);
        this.tempAmountToTransfer = this.currencyToNumber(this.amountToTransfer);
        this.amountToTransfer = this.numberToCurrencyFormat(this.tempAmountToTransfer);
        const amount = this.tempAmountToTransfer;
        this.amountToTransferDub = this.tempAmountToTransfer;
        this.toastr.clear();
        this.timeOut = setTimeout(() => {
            if (amount > this.investment.availableBalance) {
                this.amountToTransfer = this.investment.availableBalance;
                this.notValid = false;
                this.toastr.warning(
                    "The maximum amount you can withdraw is " +
                    this.investment.availableBalance
                );
            } else if (this.tempAmountToTransfer < 100 && !this.isPocketWithdrawal) {
                this.tempAmountToTransfer = 100;
                this.amountToTransfer = this.numberToCurrencyFormat(this.tempAmountToTransfer);
                this.notValid = false;
                this.availableBalance = this.investment.availableBalance - 100;
                this.toastr.warning(
                    "The minimum amount you can withdraw is ₦100"
                );
            }
            else if (!this.isPocketWithdrawal) {
                if (this.amountToTransferDub > (this.investment.availableBalance - 10000)) {
                    this.tempAmountToTransfer = 100;
                    this.amountToTransfer = this.numberToCurrencyFormat(this.tempAmountToTransfer);
                    this.availableBalance = this.investment.availableBalance - 100;
                    this.toastr.warning(
                        "Your Available balance should not be less than ₦10,000"
                    );
                }
            }
        }, 1650);
    }
    disableButton() {
        this.notValid = false;
        const amount = this.tempAmountToTransfer;
        this.amountToTransferDub = this.tempAmountToTransfer;
        if (amount > this.investment.availableBalance) {
            this.notValid = true;
        } else if (this.tempAmountToTransfer < 100) {
            this.notValid = true;
        }
        else if (!this.isPocketWithdrawal) {
            if (this.amountToTransferDub > (this.investment.availableBalance - 10000)) {
                this.notValid = true;
            } else {
                this.notValid = false;
            }
        } else {
            this.notValid = false;
        }
    }

    checkingForClosedModal() {

        this.modalService.onHide.subscribe(() => {
            if (this.dialogOption.isConfirmed && !this.isPocketWithdrawal) {
                this.showDepositCalculator = true;
            }
        });
    }

    depositCompleted(calculatorValues: Object) {
        const tenor = calculatorValues['tenor'];
        this.router.navigate([appRoutes.investment, this.id, 'withdraw', 'partial', this.amount, tenor]);
    }

    assignCalculatorVariables(calculatorValues: Object) {
        const tenor = calculatorValues['tenor'];
        this.router.navigate([appRoutes.investment, this.id, 'withdraw', 'partial', this.amount, tenor]);

    }
}
