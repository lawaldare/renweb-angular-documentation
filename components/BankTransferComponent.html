import {
    Component,
    OnInit,
    Output,
    EventEmitter,
    OnDestroy,
    Input,
} from "@angular/core";
import { appRoutes } from "src/app/@config/routes";
import * as moment from "moment";
import { Router } from "@angular/router";
import { GeneralService } from "src/app/@core/general.service";
import { InvestmentInterface } from "src/app/@model/interfaces";
import { ToastrService } from "ngx-toastr";
import { Client } from "src/app/@model/client";
import { ClientService } from "src/app/@core/client.service";

@Component({
    selector: "app-bank-transfer",
    templateUrl: "./bank-transfer.component.html",
    styleUrls: [
        "../../invest/plan-name/plan-name.component.scss",
        "./bank-transfer.component.scss",
    ],
})
export class BankTransferComponent implements OnInit, OnDestroy {
    routes = appRoutes;
    client: Client = {};
    timerInterval;
    timer;
    timeLeft;
    @Input() investment: InvestmentInterface;
    @Output() back = new EventEmitter();
    @Output() done = new EventEmitter();

    constructor(
        private router: Router,
        private generalService: GeneralService,
        private toastr: ToastrService,
        private clientService: ClientService
    ) {}

    ngOnInit(): void {
        this.timer = new Date(moment().add(10, "minutes").toDate()).getTime();
        // this.getTimer();
        // this.timerInterval = setInterval(() => {
        //     this.getTimer();
        // }, 1000);
        this.client = this.clientService.getClientLocally();
    }

    getTimer() {
        let now = new Date().getTime();
        let diff = this.timer - now;
        let minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
        let seconds = Math.floor((diff % (1000 * 60)) / 1000);

        if (minutes == 0 && seconds == 0) {
            this.done.emit();
        }

        this.timeLeft = `${minutes < 10 ? `0${minutes}` : minutes}:${
            seconds < 10 ? `0${seconds}` : seconds
        }`;
    }

    paymentDone() {
        this.investment.paymentMade = true;
        this.done.emit();
    }

    ngOnDestroy() {
        clearInterval(this.timerInterval);
    }

    copyText(text: number){
        this.generalService.copyText(text.toString());
    }
}
