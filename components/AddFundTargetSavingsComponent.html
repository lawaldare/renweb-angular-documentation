import { Component, OnInit } from '@angular/core';
import { Client, DialogModalButton } from 'src/app/@model/client';
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';
import { InvestmentInterface } from 'src/app/@model/interfaces';
import { ClientService } from 'src/app/@core/client.service';
import { Router, ActivatedRoute } from '@angular/router';
import { GeneralService } from 'src/app/@core/general.service';
import { InvestmentService } from 'src/app/@core/investment.service';
import { ToastrService } from 'ngx-toastr';
import { appRoutes } from 'src/app/@config/routes';
import { Location } from "@angular/common";
import { DialogBoxComponent } from '../../shared/dialog-box/dialog-box.component';


@Component({
    selector: 'app-add-fund-target-savings',
    templateUrl: './add-fund-target-savings.component.html',
    styleUrls: ['./add-fund-target-savings.component.scss']
})
export class AddFundTargetSavingsComponent implements OnInit {

    client: Client = {};
    modalRef: BsModalRef;
    investment: InvestmentInterface = {};
    state = 0;
    dialogOption = new DialogModalButton();
    states = {
        planAmount: 0,
        planFund: 1,
        bankTransfer: 2,
    };

    constructor(
        public clientService: ClientService,
        public router: Router,
        public locationRef: Location,
        public generalService: GeneralService,
        public investmentService: InvestmentService,
        private modalService: BsModalService,
        private toastr: ToastrService,
        private route: ActivatedRoute,
    ) { }

    ngOnInit(): void {
        this.client = this.clientService.getClientLocally();
        if (!this.client.bvnValidated) {
            this.router.navigate([appRoutes.accountSetUp]);
        }
        this.route.parent.paramMap.subscribe((params) => {
            this.investment._id = params.get("id");
        });

        this.checkForClosedModal();

    }


    incrementState() {
        this.state++;
    }


    amountDone(data) {
        this.investment = { ...this.investment, ...data };
        this.incrementState();
    }

    paymentDone() {
        if (!this.investment.paymentMade) {
            return this.incrementState();
        }

        this.modalRef = this.modalService.show(DialogBoxComponent, {
            backdrop: "static",
            initialState: {
                type: "info",
                id: "payment",
                header: "Awesome!",
                body: `You have successfully funded your smart goal.`,
                confirmText: "Done",
                dialogOption: this.dialogOption,
            },
        });
    }

    decrementState() {
        if (this.state > 0) this.state--;
        else this.locationRef.back();
    }

    checkForClosedModal() {
        this.modalService.onHide.subscribe(() => {
            if (!this.dialogOption.isConfirmed) {
                return (this.dialogOption.isConfirmed = false);
            }

            return this.router.navigate([appRoutes.dashboard]);
        });
    }


}

<ng-container>
    <app-plan-amount [hidden]="state !== states.planAmount" (done)="amountDone($event)"></app-plan-amount>

    <app-plan-fund [investment]="investment" (done)='paymentDone()' *ngIf="state==states.planFund"
        [availableOptions]="['debitCard']" pageTitle="Add fund to your smart goal" [showNotification]="false">
    </app-plan-fund>

    <app-bank-transfer *ngIf="state==states.bankTransfer" (back)="decrementState()" (done)='paymentDone()'
        [investment]="investment">
    </app-bank-transfer>

</ng-container>
