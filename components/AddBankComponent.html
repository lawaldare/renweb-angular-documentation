import { Component, OnInit, Output, EventEmitter } from '@angular/core';
import { BsModalRef } from 'ngx-bootstrap/modal';
import { ClientService } from 'src/app/@core/client.service';
import { GeneralService } from 'src/app/@core/general.service';
import { InvestmentService } from 'src/app/@core/investment.service';
import { ToastrService } from 'ngx-toastr';
import { BankData } from 'src/app/@page/single-view/banks';
import { LoanInterface } from '../create-loan.component';
import { Input } from '@angular/core';

@Component({
    selector: 'app-add-bank',
    templateUrl: './add-bank.component.html',
    styleUrls: ["../../../invest/invest/invest.shared.scss",
        "../../../invest/plan-name/plan-name.component.scss",
        './add-bank.component.scss'],
    providers: [BsModalRef]
})
export class AddBankComponent implements OnInit {

    @Output() done = new EventEmitter<any>();
    @Input() loanDetail: LoanInterface;


    clientSubscription: any;
    accountDetails: any = {};
    banks = BankData;
    investmentSubscription: any;
    showBank = false;


    constructor(private clientService: ClientService,
        private investmentService: InvestmentService,
        private generalService: GeneralService,
        private toastr: ToastrService) { }

    ngOnInit(): void {
        this.getMambuDetails();
    }

    getMambuDetails() {
        this.generalService.showLoader(true);
        this.clientSubscription = this.clientService.getClientFromMambu()
            .subscribe(
                (data) => {
                    if (data.data) {
                        if (data.data.bankLiquidationDetails) {
                            Object.assign(this.accountDetails, { bank: data.data.bankLiquidationDetails.nameOfBank });
                            Object.assign(this.accountDetails, { accountNumber: data.data.bankLiquidationDetails.accountNumber });

                            this.getAccountName(this.banks.find(
                                (x) => x.Name.toLowerCase() === this.accountDetails.bank.toLowerCase()
                            ).Code, this.accountDetails.accountNumber);
                        }
                    } else {
                        this.showBank = true;
                        this.generalService.showLoader(false);
                    }


                },
                (error) => {
                    this.showBank = true;
                    this.generalService.showLoader(false);
                }
            );
    }
    getAccountName(bankCode, accountNumber) {
        this.generalService.showLoader(true);
        this.investmentSubscription = this.investmentService
            .getAccountName(
                bankCode,
                accountNumber
            )
            .subscribe(
                (data: any) => {
                    this.generalService.showLoader(false);
                    if (data.message === "Invalid Account Number") {
                        this.toastr.error("Invalid bank and account number combination");

                        return;
                    }
                    this.accountDetails.accountName = data.data.account_name;
                    this.showBank = true;

                },
                (error) => {
                    this.generalService.showLoader(false);
                    this.showBank = true;
                    this.toastr.error(
                        "Error occured during account name enquiry"
                    );
                }
            );
    }


    getAccountDetails(accountDetails) {
        this.done.emit(accountDetails);
    }

}
