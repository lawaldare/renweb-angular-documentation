    // Okra implementations

    openOkra() {
        Okra.buildWithOptions({
            name: environment.okra.clientName,
            env: environment.okra.env,
            key: environment.okra.publicKey,
            token: environment.okra.clientToken,
            limit: environment.okra.limit,
            color: environment.okra.color,
            logo: environment.okra.logo,
            manual: false,
            ussd: false,
            products: environment.okra.products,
            onSuccess: (data) => {
                let payload: any = {};
                payload.record_id = data.record_id;
                payload.loan_id = this.loanId;
                payload.client_id = this.clientId;
                payload.customer_id = data.customer_id;
                payload.channel = 'renweb';

                console.log(payload);

                this.loanService.storeOkraRecord(payload).subscribe(data => {
                    this.modalRef.hide();
                    localStorage.removeItem('act_de');
                    this.done.emit();
                }, error => {
                    this.toastr.error('oops', error.message);
                })
            },
            onClose: () => {
                this.modalRef.hide();
                if (this.mbsBanks.find(bank => bank.bankName === this.clientBankName)) {
                    this.initiateMBS();
                } else {
                    this.loading = false;
                    this.mobileBankingState = 'manual';
                }
            },
            onError: () => {
                this.loading = true;
                if (this.mbsBanks.find(bank => bank.bankName === this.clientBankName)) {
                    this.initiateMBS();
                } else {
                    this.loading = false;
                    this.mobileBankingState = 'manual';
                }
            }
        })
    }

    // Mono implementations

    openMono() {
        let connect;
        var config = {
            key: environment.monoPublicKey,
            onSuccess: (data) => {
                let payload: any = {};
                payload.code = data.code;
                payload.loanId = this.loanId;
                payload.clientId = this.clientId;
                payload.channel = 'renweb';

                console.log(payload);

                this.loanService.storeMonoRecord(payload).subscribe(data => {
                    this.modalRef.hide();
                    localStorage.removeItem('act_de');
                    this.done.emit();
                }, error => {
                    this.toastr.error('oops', error.message);
                })
            },

            onClose: () => {
                this.noMono = true;
            }
        };

        connect = new MonoConnect(config);
        connect.setup();
        connect.open();
    }

    changePasswordState(shouldShowBSPassword: boolean, event) {
        if (!shouldShowBSPassword) {
            this.passwordNextState = 'Show';
            event.target.parentElement.children[1].type = 'password';
            this.shouldShowBSPassword = !shouldShowBSPassword;
            return;
        }
        this.passwordNextState = 'Hide';
        event.target.parentElement.children[1].type = 'text';
        this.shouldShowBSPassword = !shouldShowBSPassword;
    }


    getMBSbanks() {
        this.loanService.getMBSBanks().subscribe((data: any) => {
            this.mbsBanks = data.mbs;
        }, error => {
            this.mbsBanks = mbsSupportedBankData.mbs;
        })
    }

    initiateMBS() {
        this.loading = true;
        this.loanService.initiateMBS(this.loanId).subscribe((data: any) => {
            this.loading = false;
            if (data.success !== 'fail') {
                this.toastr.success('We just sent a request to your bank, please check your text messages for your ticket number and password.')
                this.mobileBankingState = 'mbs-page';
            } else {
                this.mobileBankingState = 'manual';
            }

        }, error => {
            this.loading = false;
            this.mobileBankingState = 'manual';
        })
    }

    confirmMBS() {
        this.loading = true;
        const obj = {
            loan_id: this.loanId, ticket_number: this.ticketNumber, password: this.bsPassword
        }
        this.loanService.confirmMBS(obj).subscribe((data: { status: string }) => {
            this.loading = false;
            if (!data.status.toLowerCase().includes('fail')) {
                this.done.emit(true);
                localStorage.removeItem('act_de');
            } else {
                this.mobileBankingState = 'manual';
            }
        }, error => {
            this.showLastButtonOption = true;
            this.toastr.error(error.error.message);
            this.loading = false;
        })
    }

    tryLastOption() {
        this.mobileBankingState = 'manual';
    }

    sendBankStatement() {
        this.done.emit();
        localStorage.removeItem('act_de');
