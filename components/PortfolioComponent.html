import { Component, OnInit } from "@angular/core";
import { BsModalRef } from "ngx-bootstrap/modal";
import { ChartType } from "chart.js";
import { Label } from "ng2-charts";

@Component({
    selector: "app-portfolio",
    templateUrl: "./portfolio.component.html",
    styleUrls: ["./portfolio.component.scss"],
})
export class PortfolioComponent implements OnInit {
    portfolioBalance: number = 0;
    totalTargetSavingsBalance: number = 0;
    targetSavingsEarnings: number = 0;
    totalFixedDepositBalance: number = 0;
    fixedDepositEarnings: number = 0;
    totalRenflexSavingsBalance: number = 0;
    renflexSavingsEarnings: number = 0;

    financialData: any = {};

    doughnutChartLabels: Label[];
    doughnutChartData = [];
    doughnutChartType: ChartType = "doughnut";
    chartColors: Array<any> = [
        {
            backgroundColor: ["#55C5CF", "#055E83", "#F15822"],
            borderColor: ["#55C5CF", "#055E83", "#F15822"],
        },
    ];
    chartOptions = {
        legend: {
            display: false,
        },
        tooltips: {
            callbacks: {
                title: function (tooltipItem, data) {
                    return data["labels"][tooltipItem[0]["index"]];
                },
                label: function (tooltipItem, data) {
                    return data["datasets"][0]["data"][
                        tooltipItem["index"]
                    ].toLocaleString();
                },
            },
        },
    };
    hasData: boolean = false;

    constructor(public modalRef: BsModalRef) { }

    ngOnInit(): void {
        this.financialData = [
            {
                title: "Smart Goal",
                subTitle: "Interest earned",
                total: this.totalTargetSavingsBalance,
                earnings: this.targetSavingsEarnings,
                color: "#55C5CF",
            },
            {
                title: "Fixed Deposit",
                subTitle: "Interest earned",
                total: this.totalFixedDepositBalance,
                earnings: this.fixedDepositEarnings,
                color: "#055E83",
            },
            {
                title: "Renmoney Account",
                subTitle: "Interest earned",
                total: this.totalRenflexSavingsBalance,
                earnings: this.renflexSavingsEarnings,
                color: "#F15822",
            },
        ];

        this.doughnutChartLabels = this.financialData.map((item) => item.title);
        this.doughnutChartData = this.financialData.map((item) => item.total);

        setTimeout(() => {
            this.hasData = this.doughnutChartData.reduce((a, b) => a + b) > 0;
        }, 100);
    }
}
