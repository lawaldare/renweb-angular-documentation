import { appRoutes } from "src/app/@config/routes";
import { LiquidationModalComponent } from "./../../modals/liquidation-modal/liquidation-modal.component";
import { NextOfKinModalComponent } from "./../../modals/next-of-kin-modal/next-of-kin-modal.component";
import { InputBvnModalComponent } from "./../../modals/input-bvn-modal/input-bvn-modal.component";
import { Component, OnInit, OnDestroy } from "@angular/core";
import { Client } from "src/app/@model/client";
import { ClientService } from "src/app/@core/client.service";
import { Location } from "@angular/common";
import { BsModalRef, BsModalService } from "ngx-bootstrap/modal";
import { Router } from "@angular/router";
import { GeneralService } from "src/app/@core/general.service";
import { ToastrService } from "ngx-toastr";

@Component({
    selector: "app-account-set-up",
    templateUrl: "./account-set-up.component.html",
    styleUrls: [
        "../../invest/plan-name/plan-name.component.scss",
        "./account-set-up.component.scss",
    ],
})
export class AccountSetUpComponent implements OnInit, OnDestroy {
    bsModalRef: BsModalRef;
    client = new Client();
    // bvnValidated: boolean = false;
    haveLiquidationBank: boolean = false;
    haveNokDone: boolean = false;
    nokDetails: any = {};
    isClientOnMambu = false;
    timeout: any;
    bvnValidated: any;


    constructor(
        private clientService: ClientService,
        private location: Location,
        private modalService: BsModalService,
        private generalService: GeneralService,
        private toastr: ToastrService,
        private router: Router
    ) {
        this.clientService.bvnValidated$.subscribe(data => {
            this.bvnValidated = data;
        })

        this.clientService.nokDone$.subscribe(data => {
            this.haveNokDone = data;
        })

        this.clientService.haveLiquidationBank$.subscribe(data => {
            this.haveLiquidationBank = data;
        })
    }

    ngOnInit(): void {

        let client_temp = {};

        Object.assign(client_temp, this.clientService.getClientLocally());

        if (this.clientService.getClientLocallyForAccountSetup()) {
            this.client = this.clientService.getClientLocallyForAccountSetup();
        } else {
            this.clientService.saveClientLocallyForAccountSetup(client_temp);
            this.client = this.clientService.getClientLocallyForAccountSetup();
        }



        if (this.client.bvnValidated) {
            this.bvnValidated = true;
        }

        if (this.client.nokDone) {
            this.haveNokDone = true;
        }

        if (this.client.accountNumber) {
            this.haveLiquidationBank = true;
        }
        // this.getDataFromMambu();

        // this.checkingForClosedModal();
    }

    ngOnDestroy(): void {
        //Called once, before the instance is destroyed.
        //Add 'implements OnDestroy' to the class.
        clearTimeout(this.timeout)
    }

    getDataFromMambu() {

        this.generalService.showLoader(true);
        this.clientService.getClientFromMambu().subscribe(
            (data: any) => {
                this.generalService.showLoader(false);
                if (data.data) {
                    if (data.data.bankLiquidationDetails) {
                        this.haveLiquidationBank = true;
                    }
                    if (data.data.nokDetails) {
                        // this.haveNokDone = true;
                        this.nokDetails = data.data.nokDetails;
                    }
                } else {
                    this.isClientOnMambu = false;
                }

            },
            (error) => {
                this.generalService.showLoader(false);
                this.isClientOnMambu = false;
            }
        );
    }

    goBack() {
        this.location.back();
    }

    goToDashboard() {
        this.router.navigate([appRoutes.dashboard]);
    }

    openBVNModalComponent() {
        this.bsModalRef = this.modalService.show(InputBvnModalComponent, {
            backdrop: "static",
            initialState: {
                client: this.client,
                isBvnValidated: this.bvnValidated,
            },
        });
    }

    openNextOfKinModalComponent() {
        if (this.bvnValidated) {
            this.bsModalRef = this.modalService.show(NextOfKinModalComponent, {
                backdrop: "static",
                initialState: {
                    client: this.client,
                    ...this.nokDetails,
                    nokPhoneNumber: this.nokDetails.nokPhone,
                    nokSurname: this.nokDetails.nokLastName,
                },
            });
        }
    }

    openLiquidationModalComponent() {
        if (this.bvnValidated) {
            this.bsModalRef = this.modalService.show(
                LiquidationModalComponent,
                {
                    backdrop: "static",
                    initialState: { client: this.client },
                }
            );
        }
    }

    // checkingForClosedModal() {
    //     this.modalService.onHide.subscribe(() => {
    //         // this.generalService.showLoader(true);
    //         // this.getDataFromMambu();


    //     });
    // }
}
