import { Component, OnInit, Input } from "@angular/core";
import { BsModalRef } from "ngx-bootstrap/modal";
import { DashboardService } from "src/app/@core/dashboard.service";
import { Client } from "src/app/@model/client";
import { ToastrService } from "ngx-toastr";
import { ClientService } from "src/app/@core/client.service";
import { InvestmentService } from "src/app/@core/investment.service";
import { appRoutes } from "src/app/@config/routes";

@Component({
    selector: "app-bvn",
    templateUrl: "./bvn-modal.component.html",
    styleUrls: ["./bvn-modal.component.scss"],
})
export class BvnModalComponent implements OnInit {
    calculateData: boolean;
    routes = appRoutes;
    calculatorDetails: any;

    client = new Client();
    verify = {
        bvn: "",
        dateOfBirth: "",
    };
    loading = false;

    constructor(
        public bsModalRef: BsModalRef,
        private clientService: ClientService,
        private bvnService: DashboardService,
        private toastr: ToastrService,
        private investmentService: InvestmentService
    ) { }

    ngOnInit() {
    }

    verifyBVN(verify) {
        this.loading = true;
        this.bvnService.getDataFromBVN({ bvn: verify.bvn }).subscribe(
            (response) => {
                if (
                    new Date(response.data.DOB).getTime() ==
                    verify.dateOfBirth.getTime()
                ) {
                    //TODO: update the is User's BVN
                    if (response.data.middleName) {
                        this.client.middleName = response.data.middleName;
                    }

                    this.client.dateOfBirth = verify.dateOfBirth
                        .toISOString()
                        .split("T")[0];
                    this.client.address = response.data.residentialAddress;
                    this.client.gender = response.data.gender.toUpperCase();

                    this.client.bvn = verify.bvn;
                    this.client.addressState = "Lagos";
                    this.client.addressLGA = "Oshodi-Isolo";
                    this.client.id = verify.bvn.toString();
                    // this.client.documentType = "BVN";
                    // this.client.documentId = this.client.bvn;
                    this.createClientOnMambu(this.client);
                    this.toastr.success('BVN successfully validated!');
                } else {
                    this.toastr.error(
                        "Your bvn doesn't match the date of birth you provided"
                    );
                    this.loading = false;
                }
            },
            (error) => {
                this.loading = false;
                this.toastr.error(
                    error.message
                        ? "You entered an invalid BVN"
                        : "Something went wrong while trying to verify your bvn"
                );
            }
        );
    }

    createClientOnMambu(client) {
        this.clientService.updateClient(client).subscribe(
            (data: any) => {
                this.client.bvnValidated = true;
                // this.client.clientEncodedKey = data.data.clientEncodedKey;
                this.getUserData();
            },
            (error) => {
                this.loading = false;
                this.toastr.error(
                    "Something went wrong while creating the client"
                );
            }
        );
    }

    getUserData() {
        this.clientService.getClient().subscribe(
            (data) => {
                this.loading = false;
                this.bsModalRef.hide();
                data.data["gender"] = data.data.sex;
                this.clientService.saveClientLocally(data.data);
            },
            (error) => {
                this.loading = false;
                this.toastr.error(
                    "Something went wrong while getting the client"
                );
            }
        );
    }

}
