import { Component, OnInit } from '@angular/core';
import { Router, ActivatedRoute } from '@angular/router';
import { InvestmentService } from 'src/app/@core/investment.service';
import { ClientService } from 'src/app/@core/client.service';
import { GeneralService } from 'src/app/@core/general.service';
import { ToastrService } from 'ngx-toastr';

@Component({
    selector: 'app-history',
    templateUrl: './history.component.html',
    styleUrls: ['./history.component.scss']
})
export class HistoryComponent implements OnInit {

    id: string;
    investment: any
    balancePlusEarnings: number;
    duration: number;
    transactions: any;

    constructor(
        private route: ActivatedRoute,
        private investmentService: InvestmentService,
        private generalService: GeneralService,
        private toastr: ToastrService
    ) { }

    ngOnInit(): void {
        this.route.parent.paramMap.subscribe((params) => {
            this.id = params.get("id");
            this.getSingleInvestment(this.id);
            this.getInvestmentHistory(this.id);
        });
    }


    getInvestmentHistory(id: string) {
        // this.loading = true;
        this.investmentService.getInvestmentHistory(id).subscribe(
            (data) => {
                // this.loading = false;
                this.transactions = data.data.clientTransactions;
            },
            (error) => {
                // this.loading = false;
            }
        );
    }



    sendStatement() {
        this.investmentService.sendStatement(this.id).subscribe(data => {
            this.toastr.success('Statement sent successfully')
        }, error => {
            this.toastr.error('There is a problem sending your statement')
        })
    }


    getSingleInvestment(id: string) {
        this.generalService.showLoader(true);
        this.investmentService.getSingleInvestment(id).subscribe(
            (data) => {
                this.investment = this.attachDueInvestmentKey(
                    data.data.clientDeposit
                );
                this.investment.withHoldingTax = +this.investment.earnings * 0.1;
                if (this.investment.accountType === 'SAVINGS_PLAN') {

                    this.investment.proposedInvestmentAmount = +this.investment.availableBalance - +this.investment.earnings + this.investment.withHoldingTax;
                }
                this.balancePlusEarnings = this.investment.proposedInvestmentAmount + +this.investment.earnings;
                this.duration = +this.investment.duration / 30;
                const interest = this.investment.proposedInvestmentAmount * (+this.investment.interestRate * (this.investment.duration / 30) / (100 * 12));

                this.investment.expectedPayout = this.investment.proposedInvestmentAmount + interest * 0.9;

                this.generalService.showLoader(false);
            },
            (error) => {
                this.generalService.showLoader(false);
            }
        );
    }

    attachDueInvestmentKey(inv) {
        const now = new Date();
        const presentDate = new Date(
            `${now.getFullYear()}-${now.getMonth()}-${now.getDate()}`
        ).getTime();
        const invMaturityDate = new Date(inv.maturityDate).getTime();
        inv.isWithdrwalDue = presentDate > invMaturityDate ? true : false;
        return inv;
    }

}
