import { Component, OnInit, Input, Output, EventEmitter } from "@angular/core";
import { InvestmentInterface } from "src/app/@model/interfaces";

@Component({
    selector: "app-plan-review",
    templateUrl: "./plan-review.component.html",
    styleUrls: [
        "../plan-name/plan-name.component.scss",
        "./plan-review.component.scss",
    ],
})
export class PlanReviewComponent implements OnInit {
    @Input() investment: InvestmentInterface = {};
    @Output() done: EventEmitter<boolean> = new EventEmitter<boolean>();

    constructor() {}

    ngOnInit(): void {}

    looksGood() {
        this.done.emit(true);
    }
}
