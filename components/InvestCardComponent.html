import { Component, OnInit, Input, HostListener } from "@angular/core";
import { Router } from "@angular/router";
import { appRoutes } from "src/app/@config/routes";

@Component({
    selector: "app-invest-card",
    templateUrl: "./invest-card.component.html",
    styleUrls: ["./invest-card.component.scss"]
})
export class InvestCardComponent implements OnInit {
    @Input("savings") inv;
    @Input() isWithdrawPage = false;
    @Input() type;
    timeFinishedPercent: number;
    constructor(private router: Router) { }

    ngOnInit(): void {
        this.setRulePercentage();
    }

    @HostListener("click")
    goToInvestmentPage() {
        if (!this.isWithdrawPage) {
            this.router.navigate([appRoutes.investment, this.inv.id]);
        }
        if (this.inv.isWithdrwalDue) {
            this.router.navigate([appRoutes.investment, this.inv.id]);
        }
    }

    setRulePercentage() {
        const now = new Date().getTime();
        const invStartDate = new Date(this.inv.startDate).getTime();

        this.timeFinishedPercent =
            ((now - invStartDate) / 86400000 / this.inv.duration) * 100;



    }
}
