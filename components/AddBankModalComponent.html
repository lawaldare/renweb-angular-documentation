import { LoanService } from 'src/app/@core/loan.service';
import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { BankData } from '../../single-view/banks';
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';
import { ClientService } from 'src/app/@core/client.service';
import { InvestmentService } from 'src/app/@core/investment.service';
import { ToastrService } from 'ngx-toastr';
import { GeneralService } from 'src/app/@core/general.service';
import { Client } from 'src/app/@model/client';
import { DashboardService } from 'src/app/@core/dashboard.service';
import { LoanInterface } from '../../create/loan/create-loan.component';
import { environment } from 'src/environments/environment';

@Component({
    selector: 'app-add-bank-modal',
    templateUrl: './add-bank-modal.component.html',
    styleUrls: ['./add-bank-modal.component.scss'],

})
export class AddBankModalComponent implements OnInit {

    @Input() isInPage = false;
    @Output() done = new EventEmitter<any>();
    @Input() loanDetail: LoanInterface;



    @Input() accountDetails: {
        bank?: string,
        accountNumber?: string,
        bankCode?: string;
        accountName?: string
    };

    banks = BankData;
    status: boolean = false;
    loading: boolean;
    statusMessage: string;
    accountNameFetched: boolean;
    timeout: any;
    client: Client;
    accountName: string;
    bankCode: string;
    bank: string;
    accountNumber: string;
    id: string;
    loanType: string;
    teamList: any;
    loan: any;


    constructor(
        public bsModalRef: BsModalRef,
        private clientService: ClientService,
        private investmentService: InvestmentService,
        private toastr: ToastrService,
        private generalService: GeneralService,
        private loanService: LoanService,
        private dashboardService: DashboardService
    ) { }

    ngOnInit(): void {
        this.id = 'addBank';

        if (this.accountDetails.bank) {
            debugger;
            this.bankCode = this.banks.find(
                (x) => x.Name.toLowerCase() === this.accountDetails.bank.toLowerCase()
            ).Code;
            this.accountNumber = this.accountDetails.accountNumber;
            this.accountName = this.accountDetails.accountName;
        } else {
            this.accountDetails = {};
        }

        this.teamList = this.loanService.getTeamLists();


    }


    update() {
        this.loading = true;
        const client: any = {};
        this.client = this.clientService.getClientLocally();
        client.firstName = this.client.firstName;
        client.lastName = this.client.lastName;
        if (this.client.middleName) {
            client.middleName = this.client.middleName;
        }
        client.dateOfBirth = this.client.dob;
        if (this.isInPage) {
            this.client.dateOfBirth = this.client.dob;
        }
        if (this.client.address) {
            client.address = this.client.address;
        }
        if (this.client.sex) {
            client.gender = this.client.sex.toUpperCase();
        }


        client.bvn = this.client.bvn;
        client.id = client.bvn;
        client.documentType = "BVN";

        if (this.accountName.toLowerCase().split(" ").includes(this.client.firstName.toLowerCase().trim()) || this.accountName.toLowerCase().split(" ").includes(this.client.lastName.toLowerCase().trim())) {
            this.bank = this.banks.find((x) => x.Code === this.bankCode).Name;
            if (this.isInPage) {
                this.client = this.loanService.getFormerLoanDetails().client;
                this.client.bank = this.bank;
                this.client.accountNumber = this.accountNumber
                this.accountDetails.accountName = this.accountName;
                this.accountDetails.bank = this.bank;
                this.accountDetails.accountNumber = this.accountNumber;
                this.accountDetails.bankCode = this.bankCode;
                this.loanService.saveLoanClient(this.client);
                this.updateLoanAccount();
            } else {
                this.updateBankData(this.bank, this.accountNumber, client);
            }
        } else {
            this.loading = false;
            this.statusMessage =
                "Bank details do not match the loan applicant. Please enter your correct bank details ";
            this.toastr.error("Bank details do not match the loan applicant. Please enter your correct bank details ");
        }
    }

    updateBankData(bankName, accountNumber, client) {
        client.bank = bankName;
        client.accountNumber = accountNumber;
        this.clientService.updateClient(client).subscribe(
            (data) => {
                this.loading = false;
                this.bsModalRef.hide();
            },
            (error) => {
                this.loading = false;
                this.toastr.error(
                    error.error.message || "Error occured while adding to bank"
                );
            }
        );
    }

    changeBank(bankCode) {
        this.accountName = "";
        this.accountNumber = "";
        this.bankCode = bankCode;
    }



    getAccountName(bankCode) {
        delete this.accountName;
        delete this.statusMessage;

        clearTimeout(this.timeout);
        delete this.statusMessage;
        this.timeout = setTimeout(() => {
            if (this.accountNumber.length === 10 && bankCode) {
                this.loading = true;
                this.investmentService.getAccountName(bankCode, this.accountNumber).subscribe((data: any) => {
                    this.status = true
                    if (data.message === "Invalid Account Number") {
                        this.statusMessage = "Invalid bank and account number combination";
                        this.loading = false;
                        return;
                    }
                    this.accountName = data.data.account_name;
                    this.accountNameFetched = true;
                    this.loading = false;
                },
                    (error) => {
                        this.loading = false;
                        this.statusMessage = "Kindly try again";
                        this.toastr.error(
                            "Error occured during account name enquiry"
                        );
                    }
                );
            }
        }, 1000);
    }

    getBankAccountName() {
        this.loading = true;
        this.investmentService.getAccountName(this.bankCode, this.accountNumber).subscribe((data: any) => {
            if (data.message === "Invalid Account Number") {
                this.statusMessage = "Invalid bank and account number combination";
                this.loading = false;
                return;
            }
            this.accountName = data.data.account_name;
            this.accountNameFetched = true;
            this.loading = false;
        },
            (error) => {
                this.loading = false;
                this.statusMessage = "Kindly try again";
                this.toastr.error(
                    "Error occured during account name enquiry"
                );
            }
        );
    }

    updateLoanAccount() {
        const loanObj = {
            "id": this.loanDetail.loanId,
            "type": "loans",
            "fields": [
                {
                    "id": "Beneficiary_Account_Number_Loan_",
                    "value": this.accountDetails.accountNumber
                },
                {
                    "id": "bank_name",
                    "value": this.accountDetails.bank
                },
                {
                    "id": "Repayment_Account_No_Loan_Accoun",
                    "value": this.accountDetails.accountNumber
                },
                {
                    "id": "Repayment_Bank_Loan_Accounts",
                    "value": this.accountDetails.bank
                },
                {
                    "id": "accountNameValidated",
                    "value": this.accountDetails.accountName
                },
                {
                    "id": "accountNumberValidated",
                    "value": this.accountDetails.accountNumber
                }
            ]
        };



        this.loanService.updateAccount(loanObj).subscribe((data: any) => {
            this.loading = false;
            this.done.emit(this.accountDetails);
        }, error => {
            this.loading = false;
            this.toastr.error(error?.message, 'Oops!');
        });
    }





}
