import { Component, OnInit } from "@angular/core";
import { BsModalRef } from "ngx-bootstrap/modal";

@Component({
    selector: "app-loan-repayments-modal",
    templateUrl: "./loan-repayments-modal.component.html",
    styleUrls: ["./loan-repayments-modal.component.scss"],
})
export class LoanRepaymentsModalComponent implements OnInit {
    balance: number = 0;
    loan: any = {};
    repayments: any = [];
    colors = {
        PENDING: "#55C5CF",
        "PARTIALLY PAID": "#55C5CF",
        PAID: "#055E83",
        LATE: "#F15822",
    };

    constructor(public modalRef: BsModalRef) {}

    ngOnInit(): void {}
}
