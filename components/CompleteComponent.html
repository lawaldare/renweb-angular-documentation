import { appRoutes } from "src/app/@config/routes";
import { NgForm } from "@angular/forms";
import { Client } from "src/app/@model/client";
import { Component, OnInit, TemplateRef } from "@angular/core";
import { ClientService } from "src/app/@core/client.service";
import { GeneralService } from "src/app/@core/general.service";
import { ToastrService } from "ngx-toastr";
import { InvestmentService } from "src/app/@core/investment.service";
import { BsModalService, BsModalRef } from "ngx-bootstrap/modal";
import { DialogBoxComponent } from "../../shared/dialog-box/dialog-box.component";
import { DialogModalButton } from "src/app/@model/client";
import { ActivatedRoute, Router } from "@angular/router";
import { AddBankModalComponent } from "../../modals/add-bank-modal/add-bank-modal.component";

@Component({
    selector: "app-complete",
    templateUrl: "./complete.component.html",
    styleUrls: [
        "../../invest/plan-name/plan-name.component.scss",
        "../withdraw/withdraw.component.scss",
        "./complete.component.scss",
    ],
})
export class CompleteComponent implements OnInit {
    selectedBankCode: string;
    selectedBankLogo: string;
    amount: number;
    tenure: number;
    passwordNextState: string = "Show";
    shouldShowPassword: boolean = true;
    password: string;
    accountNumber: string;
    bank: any;
    renflexAccount: any;
    banks: any[];
    statusMessage: string;
    status: boolean = false;
    timeout: any;
    accountDetails = {
        accountName: "",
        accountNumber: "",
        bank: "",
        bankCode: "",
    };
    loading: boolean = false;
    modalRef: BsModalRef;
    dialogOption = new DialogModalButton();
    accountNameFetched: boolean = false;
    client: Client;
    id: string;
    investment: any;
    showLoader: boolean = false;
    isFullWithdrawalType: boolean;
    passwordModalLoader: boolean = false;
    investments: any;
    pocketWithdrawal: boolean = false;
    addBankId;
    pocketAccount: any

    constructor(
        private clientService: ClientService,
        private toastr: ToastrService,
        private generalService: GeneralService,
        private investmentService: InvestmentService,
        private modalService: BsModalService,
        private route: ActivatedRoute,
        private router: Router
    ) {
        this.accountNumber = "";
    }

    ngOnInit(): void {
        this.getBanks();
        this.getMambuDetails();
        this.route.parent.paramMap.subscribe((params) => {
            this.id = params.get("id");
            this.getAllInvestments();
            this.client = this.clientService.getClientLocally();
            if (!this.client.nuban) {
                this.retrieveNuban();
            } else {
                this.getPocketAccount(this.client.nuban);
            }

        });
        this.route.paramMap.subscribe((params) => {

            this.amount = +params.get("amount");
            this.tenure = +params.get("tenure");
            if (Number.isNaN(this.tenure)) {
                this.pocketWithdrawal = true;
            }
        });
        this.checkingForClosedModal();
        this.checkWithdrawalType();
    }

    checkWithdrawalType() {
        if (location.href.includes("/full")) {
            this.isFullWithdrawalType = true;
        } else {
            this.isFullWithdrawalType = false;
        }
    }

    getAllInvestments() {
        this.loading = true;
        this.investmentService.getInvestments()
            .subscribe((data) => {
                this.loading = false;
                this.investments = data.data.clientDeposits;
                this.investment = this.investments.find(x => x.id === this.id);
                // this.checkForRegularSavings(this.investments);
            }, () => {
                this.loading = false;
            });
    }

    // pocket account

    retrieveNuban() {
        this.clientService.getClientNuban(this.client.clientEncodedkey)
            .subscribe((data) => {
                this.client.nuban = data.data.nuban;
                this.getPocketAccount(this.client.nuban);
            }, () => {
                this.toastr.error("Couldn't retrieve your pocket account");
            })
    }

    getPocketAccount(id) {
        // this.generalService.showLoader(true);
        this.investmentService.getSingleInvestment(id).subscribe(
            (data) => {
                this.renflexAccount = data.data.clientDeposit;

                if (!this.renflexAccount) {
                    this.investmentService.createRegularSavings().subscribe(
                        (data) => {
                            // this.toastr.success("Pocket account created!");
                        },
                        (error) => {
                            this.toastr.error(error.error.message);
                        }
                    );
                }

                const renflexDetails = {
                    _id: this.renflexAccount.id,
                    availableBalance: this.renflexAccount.availableBalance,
                };

                Object.assign(this.investment, renflexDetails);

                // this.generalService.showLoader(false);

            },
            (error) => {
                this.toastr.error("Couldn't retrieve your pocket account");
                // this.generalService.showLoader(false);
            }
        );
    }




    getBanks() {
        this.banks = this.investmentService.getBanks();
    }

    getMambuDetails() {
        // this.loading = true;
        this.clientService.getClientFromMambu().subscribe(
            (data) => {
                if (data.data) {
                    if (data.data.bankLiquidationDetails) {
                        this.accountDetails.bank =
                            data.data.bankLiquidationDetails.nameOfBank;
                        this.accountDetails.accountNumber =
                            data.data.bankLiquidationDetails.accountNumber;
                        this.accountNumber =
                            data.data.bankLiquidationDetails.accountNumber;
                        // this.loading = false;
                        this.getAccountName(
                            this.banks.find(
                                (x) =>
                                    x.Name.toLowerCase() ===
                                    this.accountDetails.bank.toLowerCase()
                            ).Code
                        );
                        this.selectedBankCode = this.banks.find(
                            (x) =>
                                x.Name.toLowerCase() ===
                                this.accountDetails.bank.toLowerCase()
                        ).Code;
                        this.getBankLogo(this.selectedBankCode);
                    } else {
                        // this.loading = false;
                    }
                } else {
                    // this.loading = false;
                }

            },
            (error) => {
                this.toastr.error(
                    "Oops",
                    error.error.message || "Error occured while adding to bank"
                );
                // this.loading = false;
            }
        );
    }

    getBankLogo(code: string) {
        this.banks.forEach((x) => {
            if (x.Code === code) {
                this.selectedBankLogo = x.logo;
            }
        });
    }

    openAddBankModal() {
        this.modalRef = this.modalService.show(AddBankModalComponent, {
            backdrop: "static",
            initialState: {
                client: this.client,
                id: this.addBankId,
                accountDetails: this.accountDetails,
            },
        });
    }

    getAccountName(bankCode) {
        clearTimeout(this.timeout);
        delete this.statusMessage;
        this.timeout = setTimeout(() => {
            if (
                this.accountDetails.accountNumber.length === 10 &&
                this.accountDetails.bank
            ) {
                this.loading = true;
                this.investmentService
                    .getAccountName(bankCode, this.accountDetails.accountNumber)
                    .subscribe(
                        (data: any) => {

                            if (data.message === "Invalid Account Number") {
                                this.statusMessage =
                                    "Invalid bank and account number combination";
                                this.loading = false;
                                return;
                            }
                            this.accountDetails.accountName =
                                data.data.account_name;
                            this.accountNameFetched = true;
                            this.loading = false;
                        },
                        (error) => {
                            this.loading = false;
                            this.statusMessage = "Kindly try again";
                            this.toastr.error(
                                "Error occured during account name enquiry"
                            );
                        }
                    );
            }
        }, 1650);
    }

    update() {
        this.loading = true;
        const client: any = {};
        client.firstName = this.client.firstName;
        client.lastName = this.client.lastName;
        client.middleName = this.client.middleName;
        client.dateOfBirth = this.client.dob;
        client.address = this.client.address;
        client.gender = this.client.sex.toUpperCase();

        client.bvn = this.client.bvn;
        client.id = client.bvn;
        // client.documentType = "BVN";

        if (
            this.accountDetails.accountName
                .toLowerCase()
                .split(" ")
                .includes(
                    this.client.firstName.toLowerCase() ||
                    this.client.middleName.toLowerCase() ||
                    this.client.lastName.toLowerCase()
                )
        ) {
            this.updateBankData(
                this.banks.find((x) => x.Code === this.accountDetails.bankCode)
                    .Name,
                this.accountDetails.accountNumber,
                client
            );
        } else {
            this.loading = false;
            this.statusMessage =
                "Name doesn't match any of your provided names";
            this.toastr.error("Name doesn't match any of your provided names");
        }
    }

    updateBankData(bankName, accountNumber, client) {
        client.bank = bankName;
        client.accountNumber = accountNumber;
        this.clientService.updateClient(client).subscribe(
            (data) => {
                this.modalRef.hide();
                this.getUserData();
                this.getMambuDetails();
            },
            (error) => {
                this.loading = false;
                this.toastr.error(
                    "Oops",
                    error.error.message || "Error occured while adding to bank"
                );
            }
        );
    }

    getUserData() {
        this.clientService.getClient().subscribe(
            (data) => {
                this.loading = false;
                data.data["gender"] = data.data.sex;
                this.clientService.saveClientLocally(data.data);
            },
            (error) => {
                this.loading = false;
                this.toastr.error(
                    "Something went wrong while getting the client"
                );
            }
        );
    }

    transfer() {
        this.loading = true;
        let payload: any = {};
        payload.creditAccountId = this.renflexAccount.id;
        payload.debitAccountId = this.id;
        payload.narration = `Withdrawn from ${this.investment.name}`;
        if (this.isFullWithdrawalType) {
            payload.minorAmount = this.investment.availableBalance * 100;
        } else {
            payload.minorAmount = this.amount * 100;
        }

        this.investmentService.intrabankTransfer(payload).subscribe(
            (data) => {
                this.toastr.success("Funds transferred successfully");
                this.loading = false;
                this.router.navigate([appRoutes.dashboard]);
            },
            (error) => {
                this.toastr.error(error.error.message);
                this.loading = false;
            }
        );
    }

    openModal(fullWithdrawalPassword: TemplateRef<any>) {
        if (!this.accountNameFetched) {
            this.loading = true;
            this.getAccountName(
                this.banks.find(
                    (x) =>
                        x.Name.toLowerCase() ===
                        this.accountDetails.bank.toLowerCase()
                ).Code
            );
        } else {
            this.modalRef = this.modalService.show(fullWithdrawalPassword);
        }
    }

    onSubmitFullWithdrawalPasswordForm(form: NgForm) {
        this.loading = true;
        let payload: any = { ...form.value };
        if (this.isFullWithdrawalType) {
            payload.amount = this.investment.availableBalance;
            payload.tenor = this.investment.duration;
            payload.productType = "fixedDepositFull";
        } else {
            if (!this.pocketWithdrawal) {
                if (
                    this.tenure === 3 ||
                    this.tenure === 6 ||
                    this.tenure === 9 ||
                    this.tenure === 12
                ) {
                    payload.amount = this.amount;
                    payload.tenor = this.tenure * 30;
                    payload.productType = "fixedDepositPart";
                } else {
                    this.toastr.error("Invalid Tenure");
                    this.loading = false;
                    this.modalRef.hide();
                    return;
                }
            }
        }
        payload.clientId = this.client.bvn;
        payload.accountId = this.id;
        payload.bankCode = this.selectedBankCode;
        payload.accountNumber = this.accountNumber;
        this.clientService.validatePassword(form.value.password).subscribe(
            (response) => {
                this.modalRef.hide();
                if (this.investment.accountState === "MATURED" && this.investment.accountType === "FIXED_DEPOSIT") {
                    this.investmentService.maturedWithdrawal(payload).subscribe(
                        (data) => {
                            this.modalRef.hide();
                            this.loading = false;
                            this.modalRef = this.modalService.show(
                                DialogBoxComponent,
                                {
                                    backdrop: "static",
                                    class: "modal-position",
                                    initialState: {
                                        type: "info",
                                        id: "redirect",
                                        header: "Transfer initiated",
                                        body:
                                            "You will receive your funds shortly",
                                        confirmText: "okay",
                                        dialogOption: this.dialogOption,
                                    },
                                }
                            );
                        },
                        (error) => {
                            this.toastr.error(
                                "Oops",
                                error.error.message ||
                                "Error occured while initiating a withdrawal"
                            );
                            this.loading = false;
                        }
                    );
                } else if (this.investment.accountState === "MATURED" && this.investment.accountType === "SAVINGS_PLAN") {
                    this.investmentService.maturedTargetSavingsWithdrawal(payload).subscribe(
                        (data) => {
                            this.modalRef.hide();
                            this.loading = false;
                            this.modalRef = this.modalService.show(
                                DialogBoxComponent,
                                {
                                    backdrop: "static",
                                    class: "modal-position",
                                    initialState: {
                                        type: "info",
                                        header: "Transfer initiated",
                                        body:
                                            "You will receive your funds shortly",
                                        confirmText: "okay",
                                        dialogOption: this.dialogOption,
                                    },
                                }
                            );
                        },
                        (error) => {
                            this.toastr.error(
                                "Oops",
                                error.error.message ||
                                "Error occured while initiating a withdrawal"
                            );
                            this.loading = false;
                        }
                    );
                } else if (this.investment.accountState !== "MATURED" && this.investment.accountType === "FIXED_DEPOSIT") {
                    this.investmentService.preMaturedWithdrawal(payload).subscribe(
                        (data) => {
                            this.modalRef.hide();
                            this.loading = false;
                            this.modalRef = this.modalService.show(
                                DialogBoxComponent,
                                {
                                    backdrop: "static",
                                    class: "modal-position",
                                    initialState: {
                                        type: "info",
                                        header: "Transfer initiated",
                                        body:
                                            "You will receive your funds shortly",
                                        confirmText: "okay",
                                        dialogOption: this.dialogOption,
                                    },
                                }
                            );
                        },
                        (error) => {
                            this.toastr.error(
                                "Oops",
                                error.error.message ||
                                "Error occured while initiating a withdrawal"
                            );
                            this.loading = false;
                        }
                    );
                } else if (this.investment.accountState !== "MATURED" && this.investment.accountType === "REGULAR_SAVINGS") {
                    this.withdrawFromPocket(payload);
                }
            },
            (error) => {
                this.loading = false;
                form.resetForm();
                this.toastr.error(error.error.message);
                this.showLoader = false;
            }
        );
    }

    withdrawFromPocket(payload) {
        payload = {
            ...payload,
            productType: "regularSavings",
            amount: this.amount,
            accountNumber: this.accountDetails.accountNumber,
            tenor: 0,
            accountType: "REGULAR_SAVINGS",
        };

        this.investmentService.withdrawRenflexAccount(payload).subscribe(
            (data) => {
                this.modalRef.hide();
                this.loading = false;
                this.modalRef = this.modalService.show(DialogBoxComponent, {
                    backdrop: "static",
                    class: "modal-position",
                    initialState: {
                        type: "info",
                        header: "Transfer initiated",
                        body: "You will receive your funds shortly",
                        confirmText: "okay",
                        dialogOption: this.dialogOption,
                    },
                });
            },
            (error) => {
                this.toastr.error(
                    "Oops",
                    error.error.message ||
                    "Error occured while initiating a withdrawal"
                );
                this.loading = false;
            }
        );
    }

    checkingForClosedModal() {
        this.modalService.onHide.subscribe(() => {
            if (this.dialogOption.isConfirmed && !this.addBankId) {
                this.router.navigate([appRoutes.dashboard]);
            } else if (this.dialogOption.isConfirmed && this.addBankId) {
                delete this.addBankId;
            }
            this.dialogOption.isConfirmed = false;
            this.getBankLogo(this.accountDetails.bankCode);
        });
    }

    changePasswordState(shouldShowPassword: boolean, event) {
        if (!shouldShowPassword) {
            this.passwordNextState = "Show";
            event.target.parentElement.children[1].type = "password";
            this.shouldShowPassword = !shouldShowPassword;
            return;
        }
        this.passwordNextState = "Hide";
        event.target.parentElement.children[1].type = "text";
        this.shouldShowPassword = !shouldShowPassword;
    }
}
