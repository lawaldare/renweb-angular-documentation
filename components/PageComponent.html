import { Component, OnInit, OnDestroy, AfterViewInit, HostListener, ViewChild, ElementRef } from "@angular/core";
import { MenuItem } from "../sidebar/sidebar.component";
import { PAGE_MENU, MOBILE_PAGE_MENU, MOBILE_SIDE_PAGE_MENU } from "./page.menu-items";
import { BsModalRef, BsModalService } from "ngx-bootstrap/modal";
import { Client } from "src/app/@model/client";
import {
    BvnModalComponent,
    FundInvestmentModalComponent,
    PlanNameModalComponent
} from "../modals";
import { ToastrService } from "ngx-toastr";
import { ClientService } from "src/app/@core/client.service";
import { Router, NavigationEnd, ActivatedRoute } from "@angular/router";
import { Subscription } from "rxjs";
import { InvestmentService } from "src/app/@core/investment.service";
import { CalculatorModalComponent } from "../modals/calculator-modal/calculator-modal.component";
import { appRoutes } from "src/app/@config/routes";
import { GeneralService } from "src/app/@core/general.service";
import { DashboardService } from 'src/app/@core/dashboard.service';
import { AuthService } from 'src/app/@core/auth.service';
import { LoanService } from 'src/app/@core/loan.service';
import { RenConfig } from '../create/loan/loan-calculator/loan-config-mockdata';
import { gsap } from 'gsap'
import { environment } from 'src/environments/environment';
import * as FullStory from '@fullstory/browser'


@Component({
    selector: "app-page",
    templateUrl: "./page.component.html",
    styleUrls: ["./page.component.scss"]
})
export class PageComponent implements OnInit, OnDestroy, AfterViewInit {

    @ViewChild('navbar', { static: true }) navbar: ElementRef<HTMLElement>;
    @ViewChild('sideBar', { static: true }) sideBar: ElementRef<HTMLElement>;

    menuItems: MenuItem[] = PAGE_MENU;
    mobileMenuItems: MenuItem[] = MOBILE_PAGE_MENU;
    mobileSidebarMenuItems: MenuItem[] = MOBILE_SIDE_PAGE_MENU;
    bsModalRef: BsModalRef;
    client = new Client();
    routeSubscription: Subscription;
    calculatorDetails: { amount: any; tenure: any };
    loading: boolean = true;
    loadingSubsciption: Subscription;
    mobileSidebar: boolean = true;
    dashboardRoute = appRoutes.dashboard;
    timer: any;
    fiveMinutes = 5 * 60; // 5 minutes
    activityInterval: any;
    constructor(
        private modalService: BsModalService,
        private clientService: ClientService,
        private authService: AuthService,
        private router: Router,
        private generalService: GeneralService,
        private route: ActivatedRoute,
        private dashboardService: DashboardService,
        private investmentService: InvestmentService,
        private loanService: LoanService,
        private toastr: ToastrService
    ) {
        this.timer = this.fiveMinutes;

        // this.client = this.clientService.getClientLocally();

        // FullStory.init({
        //     orgId: 'YTQEK',
        //     // devMode: environment.production
        // })

        // FullStory.identify(this.client.id, {
        //     displayName: this.client.name,
        //     email: this.client.email
        // })
    }

    ngOnInit() {
        this.client.bvnValidated = false;
        // this.checkForClosedModal();
        this.checkForRouteChanges();

        // this.showTheRightModal();
        this.getUserData();

        if (!environment.production) {
            this.initAnimation();
        }
    }


    initAnimation() {
        const timeline = gsap.timeline({ defaults: { opacity: 0, duration: 1 } })
        timeline.from(this.navbar.nativeElement.childNodes, { y: -100, stagger: 0.5 })
        timeline.from(this.sideBar.nativeElement, { x: -30 })
    }



    ngAfterViewInit() {
        this.checkForLoadingState();
        this.checkForIdleTime();
        this.dashboardService.saveConfig(RenConfig);

        // this.getConfigDetails();
    }

    checkForLoadingState() {
        this.loadingSubsciption = this.generalService.state$.subscribe(data => {
            this.loading = data;
        });
    }

    getConfigDetails() {
        this.dashboardService.getConfigDetails().subscribe(data => {
            this.dashboardService.saveConfig(data);
        })
    }

    ngOnDestroy() {
        clearInterval(this.activityInterval);
        this.loadingSubsciption.unsubscribe();
        this.routeSubscription.unsubscribe();
        this.bsModalRef.hide();
    }

    checkForRouteChanges() {
        this.routeSubscription = this.router.events.subscribe(evt => {
            if (evt instanceof NavigationEnd) {
                this.getUserData();

                window.scrollTo({
                    top: 10,
                    behavior: "smooth"
                });
            }
            return;
        });
    }

    getUserData() {
        this.clientService.getClient().subscribe(data => {
            data.data["gender"] = data.data.sex;
            this.clientService.saveClientLocally(data.data);

            const formerLoanDetail = this.loanService.getFormerLoanDetails();
            if (formerLoanDetail.client) {
                if (!location.href.includes(appRoutes.continueLoanFromDashboard)) {
                    if (formerLoanDetail.client.bvn !== data.data.bvn) {
                        // this.loanService.clearFormerLoanDetail();
                    }
                }

            }
        });
    }

    openSideBar() {
        this.mobileSidebar = !this.mobileSidebar
    }

    showTheRightModal() {
        this.client = this.clientService.getClientLocally();
        this.calculatorDetails = this.investmentService.getCalcValues();

        if (!this.client.bvnValidated) {
            this.openBVNModalComponent();
        } else if (
            this.client.mambuStatus !== "OnMambu" &&
            location.href.includes("/onboard") &&
            this.calculatorDetails
        ) {
            this.openFundInvestmentModal();
        } else if (
            this.client.mambuStatus !== "OnMambu" &&
            !this.calculatorDetails &&
            location.href.includes("/onboard")
        ) {
            this.openCalculatorComponent();
        }
    }

    checkForClosedModal() {
        this.modalService.onHide.subscribe(() => {
            if (!location.href.includes("showCalc")) {
                this.showTheRightModal();
            } else {
                this.openCalculatorComponent();
            }
        });
    }

    openBVNModalComponent() {
        this.bsModalRef = this.modalService.show(BvnModalComponent, {
            backdrop: "static",
            initialState: { client: this.client }
        });
    }

    openPlanNameModalComponent() {
        this.bsModalRef = this.modalService.show(PlanNameModalComponent, {
            backdrop: "static",
            initialState: { client: this.client }
        });
    }

    openFundInvestmentModal() {
        this.bsModalRef = this.modalService.show(FundInvestmentModalComponent, {
            backdrop: "static",
            initialState: {
                client: this.client,
                calculatorDetails: this.calculatorDetails
            }
        });
    }

    openCalculatorComponent() {
        this.bsModalRef = this.modalService.show(CalculatorModalComponent, {
            backdrop: "static",
            initialState: {
                client: this.client,
                calculatorDetails: this.calculatorDetails
            }
        });
    }



    @HostListener('mousemove')
    @HostListener('click')
    resetTimer() {
        this.timer = this.fiveMinutes;
        return null;
    }

    checkForIdleTime() {

        this.activityInterval = setInterval(() => {
            this.timer--;
            if (this.timer === 0) {
                this.logOut();
            }
        }, 1000);
    }

    logOut() {
        this.authService.logOut()
        location.reload();
    }
}
