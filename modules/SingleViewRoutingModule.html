import { PartialWithdrawalComponent } from './partial-withdrawal/partial-withdrawal.component';
import { RolloverComponent } from './rollover/rollover.component';
import { CompleteComponent } from './complete/complete.component';
import { HistoryComponent } from './history/history.component';
import { DetailComponent } from './detail/detail.component';
import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';
import { AuthGuard } from 'src/app/@guards';
import { WithdrawComponent } from './withdraw/withdraw.component';
import { SingleViewComponent } from './single-view/single-view.component';
import { AddFundTargetSavingsComponent } from './add-fund-target-savings/add-fund-target-savings.component';


const routes: Routes = [
    {
        path: '',
        component: SingleViewComponent,
        canActivateChild: [AuthGuard],
        children: [
            {
                path: 'detail',
                component: DetailComponent
            },
            {
                path: 'detail-email',
                component: DetailComponent
            },
            {
                path: 'history',
                component: HistoryComponent
            },
            {
                path: 'rollover',
                component: RolloverComponent
            },
            {
                path: 'add-fund',
                component: AddFundTargetSavingsComponent
            },
            {
                path: 'withdraw/partial',
                component: PartialWithdrawalComponent
            },
            {
                path: 'withdraw/partial/:amount/:tenure',
                component: CompleteComponent
            },
            {
                path: 'withdraw/full',
                component: CompleteComponent
            },

            {
                path: 'withdraw/:balance',
                component: WithdrawComponent
            },
            {
                path: "",
                pathMatch: "full",
                redirectTo: "detail"
            }


        ]

    }
]

@NgModule({
    imports: [RouterModule.forChild(routes)],
    exports: [RouterModule],
    providers: [],

})
export class SingleViewRoutingModule { }

export const routedComponents = [DetailComponent, HistoryComponent, WithdrawComponent, CompleteComponent, SingleViewComponent, RolloverComponent, PartialWithdrawalComponent, AddFundTargetSavingsComponent];
