import { Component, OnInit, Input } from '@angular/core';

@Component({
  selector: 'app-sidebar',
  templateUrl: './sidebar.component.html',
  styleUrls: ['./sidebar.component.scss']
})
export class SidebarComponent implements OnInit {

  @Input() menuItems: MenuItem[];
  constructor() { }

  ngOnInit() {
  }

}


export interface MenuItem {
  link ?: string; // router link to be redirected to
  url ?: string; // full url to open a window to
  viewBox ?: string; // view box for svg element
  icon ?: string; // id of ison in the sprite diagram
  title ?: string; // title of the route to be redirected to
  selected ?: boolean; // is the menu selected
}
